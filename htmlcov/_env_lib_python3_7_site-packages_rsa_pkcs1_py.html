15"""Functions for PKCS#1 version 1.5 encryption and signing 

16 

17This module implements certain functionality from PKCS#1 version 1.5. For a 

18very clear example, read http://www.di-mgt.com.au/rsa_alg.html#pkcs1schemes 

19 

20At least 8 bytes of random padding is used when encrypting a message. This makes 

21these methods much more secure than the ones in the ``rsa`` module.

84def _pad_for_encryption(message: bytes, target_length: int) -> bytes: 

85 r"""Pads the message for encryption, returning the padded message. 

86 

87 :return: 00 02 RANDOM_DATA 00 MESSAGE 

88 

89 >>> block = _pad_for_encryption(b'hello', 16) 

90 >>> len(block) 

91 16 

92 >>> block[0:2] 

93 b'\x00\x02' 

94 >>> block[-6:] 

95 b'\x00hello' 

96 

97 """ 

98 

99 max_msglength = target_length - 11 

100 msglength = len(message) 

101 

102 if msglength > max_msglength: 

103 raise OverflowError('%i bytes needed for message, but there is only' 

104 ' space for %i' % (msglength, max_msglength)) 

105 

106 # Get random padding 

107 padding = b'' 

108 padding_length = target_length - msglength - 3 

109 

110 # We remove 0-bytes, so we'll end up with less padding than we've asked for, 

111 # so keep adding data until we're at the correct length. 

112 while len(padding) < padding_length: 

113 needed_bytes = padding_length - len(padding) 

114 

115 # Always read at least 8 bytes more than we need, and trim off the rest 

116 # after removing the 0-bytes. This increases the chance of getting 

117 # enough bytes, especially when needed_bytes is small 

118 new_padding = os.urandom(needed_bytes + 5) 

119 new_padding = new_padding.replace(b'\x00', b'') 

120 padding = padding + new_padding[:needed_bytes] 

121 

122 assert len(padding) == padding_length 

123 

124 return b''.join([b'\x00\x02', 

125 padding, 

126 b'\x00', 

127 message])

130def _pad_for_signing(message: bytes, target_length: int) -> bytes: 

131 r"""Pads the message for signing, returning the padded message. 

132 

133 The padding is always a repetition of FF bytes. 

134 

135 :return: 00 01 PADDING 00 MESSAGE 

136 

137 >>> block = _pad_for_signing(b'hello', 16) 

138 >>> len(block) 

139 16 

140 >>> block[0:2] 

141 b'\x00\x01' 

142 >>> block[-6:] 

143 b'\x00hello' 

144 >>> block[2:-6] 

145 b'\xff\xff\xff\xff\xff\xff\xff\xff' 

146 

147 """ 

148 

149 max_msglength = target_length - 11 

150 msglength = len(message) 

151 

152 if msglength > max_msglength: 

153 raise OverflowError('%i bytes needed for message, but there is only' 

154 ' space for %i' % (msglength, max_msglength)) 

155 

156 padding_length = target_length - msglength - 3 

157 

158 return b''.join([b'\x00\x01', 

159 padding_length * b'\xff', 

160 b'\x00', 

161 message])

164def encrypt(message: bytes, pub_key: key.PublicKey) -> bytes: 

165 """Encrypts the given message using PKCS#1 v1.5 

166 

167 :param message: the message to encrypt. Must be a byte string no longer than 

168 ``k-11`` bytes, where ``k`` is the number of bytes needed to encode 

169 the ``n`` component of the public key. 

170 :param pub_key: the :py:class:`rsa.PublicKey` to encrypt with. 

171 :raise OverflowError: when the message is too large to fit in the padded 

172 block. 

173 

174 >>> from rsa import key, common 

175 >>> (pub_key, priv_key) = key.newkeys(256) 

176 >>> message = b'hello' 

177 >>> crypto = encrypt(message, pub_key) 

178 

179 The crypto text should be just as long as the public key 'n' component: 

180 

181 >>> len(crypto) == common.byte_size(pub_key.n) 

182 True 

183 

184 """ 

185 

186 keylength = common.byte_size(pub_key.n) 

187 padded = _pad_for_encryption(message, keylength) 

188 

189 payload = transform.bytes2int(padded) 

190 encrypted = core.encrypt_int(payload, pub_key.e, pub_key.n) 

191 block = transform.int2bytes(encrypted, keylength) 

192 

193 return block

196def decrypt(crypto: bytes, priv_key: key.PrivateKey) -> bytes: 

197 r"""Decrypts the given message using PKCS#1 v1.5 

198 

199 The decryption is considered 'failed' when the resulting cleartext doesn't 

200 start with the bytes 00 02, or when the 00 byte between the padding and 

201 the message cannot be found. 

202 

203 :param crypto: the crypto text as returned by :py:func:`rsa.encrypt` 

204 :param priv_key: the :py:class:`rsa.PrivateKey` to decrypt with. 

205 :raise DecryptionError: when the decryption fails. No details are given as 

206 to why the code thinks the decryption fails, as this would leak 

207 information about the private key. 

208 

209 

210 >>> import rsa 

211 >>> (pub_key, priv_key) = rsa.newkeys(256) 

212 

213 It works with strings: 

214 

215 >>> crypto = encrypt(b'hello', pub_key) 

216 >>> decrypt(crypto, priv_key) 

217 b'hello' 

218 

219 And with binary data: 

220 

221 >>> crypto = encrypt(b'\x00\x00\x00\x00\x01', pub_key) 

222 >>> decrypt(crypto, priv_key) 

223 b'\x00\x00\x00\x00\x01' 

224 

225 Altering the encrypted information will *likely* cause a 

226 :py:class:`rsa.pkcs1.DecryptionError`. If you want to be *sure*, use 

227 :py:func:`rsa.sign`. 

228 

229 

230 .. warning:: 

231 

232 Never display the stack trace of a 

233 :py:class:`rsa.pkcs1.DecryptionError` exception. It shows where in the 

234 code the exception occurred, and thus leaks information about the key. 

235 It's only a tiny bit of information, but every bit makes cracking the 

236 keys easier. 

237 

238 >>> crypto = encrypt(b'hello', pub_key) 

239 >>> crypto = crypto[0:5] + b'X' + crypto[6:] # change a byte 

240 >>> decrypt(crypto, priv_key) 

241 Traceback (most recent call last): 

242 ... 

243 rsa.pkcs1.DecryptionError: Decryption failed 

244 

245 """ 

246 

247 blocksize = common.byte_size(priv_key.n) 

248 encrypted = transform.bytes2int(crypto) 

249 decrypted = priv_key.blinded_decrypt(encrypted) 

250 cleartext = transform.int2bytes(decrypted, blocksize) 

251 

252 # Detect leading zeroes in the crypto. These are not reflected in the 

253 # encrypted value (as leading zeroes do not influence the value of an 

254 # integer). This fixes CVE-2020-13757. 

255 if len(crypto) > blocksize: 

256 # This is operating on public information, so doesn't need to be constant-time. 

257 raise DecryptionError('Decryption failed') 

258 

259 # If we can't find the cleartext marker, decryption failed. 

260 cleartext_marker_bad = not compare_digest(cleartext[:2], b'\x00\x02') 

261 

262 # Find the 00 separator between the padding and the message 

263 sep_idx = cleartext.find(b'\x00', 2) 

264 

265 # sep_idx indicates the position of the `\x00` separator that separates the 

266 # padding from the actual message. The padding should be at least 8 bytes 

267 # long (see https://tools.ietf.org/html/rfc8017#section-7.2.2 step 3), which 

268 # means the separator should be at least at index 10 (because of the 

269 # `\x00\x02` marker that preceeds it). 

270 sep_idx_bad = sep_idx < 10 

271 

272 anything_bad = cleartext_marker_bad | sep_idx_bad 

273 if anything_bad: 

274 raise DecryptionError('Decryption failed') 

275 

276 return cleartext[sep_idx + 1:]

279def sign_hash(hash_value: bytes, priv_key: key.PrivateKey, hash_method: str) -> bytes: 

280 """Signs a precomputed hash with the private key. 

281 

282 Hashes the message, then signs the hash with the given key. This is known 

283 as a "detached signature", because the message itself isn't altered. 

284 

285 :param hash_value: A precomputed hash to sign (ignores message). 

286 :param priv_key: the :py:class:`rsa.PrivateKey` to sign with 

287 :param hash_method: the hash method used on the message. Use 'MD5', 'SHA-1', 

288 'SHA-224', SHA-256', 'SHA-384' or 'SHA-512'. 

289 :return: a message signature block. 

290 :raise OverflowError: if the private key is too small to contain the 

291 requested hash. 

292 

293 """ 

294 

295 # Get the ASN1 code for this hash method 

296 if hash_method not in HASH_ASN1: 

297 raise ValueError('Invalid hash method: %s' % hash_method) 

298 asn1code = HASH_ASN1[hash_method] 

299 

300 # Encrypt the hash with the private key 

301 cleartext = asn1code + hash_value 

302 keylength = common.byte_size(priv_key.n) 

303 padded = _pad_for_signing(cleartext, keylength) 

304 

305 payload = transform.bytes2int(padded) 

306 encrypted = priv_key.blinded_encrypt(payload) 

307 block = transform.int2bytes(encrypted, keylength) 

308 

309 return block

312def sign(message: bytes, priv_key: key.PrivateKey, hash_method: str) -> bytes: 

313 """Signs the message with the private key. 

314 

315 Hashes the message, then signs the hash with the given key. This is known 

316 as a "detached signature", because the message itself isn't altered. 

317 

318 :param message: the message to sign. Can be an 8-bit string or a file-like 

319 object. If ``message`` has a ``read()`` method, it is assumed to be a 

320 file-like object. 

321 :param priv_key: the :py:class:`rsa.PrivateKey` to sign with 

322 :param hash_method: the hash method used on the message. Use 'MD5', 'SHA-1', 

323 'SHA-224', SHA-256', 'SHA-384' or 'SHA-512'. 

324 :return: a message signature block. 

325 :raise OverflowError: if the private key is too small to contain the 

326 requested hash. 

327 

328 """ 

329 

330 msg_hash = compute_hash(message, hash_method) 

331 return sign_hash(msg_hash, priv_key, hash_method)

334def verify(message: bytes, signature: bytes, pub_key: key.PublicKey) -> str: 

335 """Verifies that the signature matches the message. 

336 

337 The hash method is detected automatically from the signature. 

338 

339 :param message: the signed message. Can be an 8-bit string or a file-like 

340 object. If ``message`` has a ``read()`` method, it is assumed to be a 

341 file-like object. 

342 :param signature: the signature block, as created with :py:func:`rsa.sign`. 

343 :param pub_key: the :py:class:`rsa.PublicKey` of the person signing the message. 

344 :raise VerificationError: when the signature doesn't match the message. 

345 :returns: the name of the used hash. 

346 

347 """ 

348 

349 keylength = common.byte_size(pub_key.n) 

350 encrypted = transform.bytes2int(signature) 

351 decrypted = core.decrypt_int(encrypted, pub_key.e, pub_key.n) 

352 clearsig = transform.int2bytes(decrypted, keylength) 

353 

354 # Get the hash method 

355 method_name = _find_method_hash(clearsig) 

356 message_hash = compute_hash(message, method_name) 

357 

358 # Reconstruct the expected padded hash 

359 cleartext = HASH_ASN1[method_name] + message_hash 

360 expected = _pad_for_signing(cleartext, keylength) 

361 

362 if len(signature) != keylength: 

363 raise VerificationError('Verification failed') 

364 

365 # Compare with the signed one 

366 if expected != clearsig: 

367 raise VerificationError('Verification failed') 

368 

369 return method_name

391def yield_fixedblocks(infile: typing.BinaryIO, blocksize: int) -> typing.Iterator[bytes]: 

392 """Generator, yields each block of ``blocksize`` bytes in the input file. 

393 

394 :param infile: file to read and separate in blocks. 

395 :param blocksize: block size in bytes. 

396 :returns: a generator that yields the contents of each block 

397 """

400 block = infile.read(blocksize) 

401 

402 read_bytes = len(block) 

403 if read_bytes == 0: 

404 break 

405 

406 yield block

412def compute_hash(message: typing.Union[bytes, typing.BinaryIO], method_name: str) -> bytes: 

413 """Returns the message digest. 

414 

415 :param message: the signed message. Can be an 8-bit string or a file-like 

416 object. If ``message`` has a ``read()`` method, it is assumed to be a 

417 file-like object. 

418 :param method_name: the hash method, must be a key of 

419 :py:const:`HASH_METHODS`. 

420 

421 """ 

422 

423 if method_name not in HASH_METHODS: 

424 raise ValueError('Invalid hash method: %s' % method_name) 

425 

426 method = HASH_METHODS[method_name] 

427 hasher = method() 

428 

429 if isinstance(message, bytes): 

430 hasher.update(message) 

431 else: 

432 assert hasattr(message, 'read') and hasattr(message.read, '__call__') 

433 # read as 1K blocks 

434 for block in yield_fixedblocks(message, 1024): 

435 hasher.update(block) 

436 

437 return hasher.digest()
