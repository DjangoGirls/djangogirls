211 def repr_failure(self, excinfo): 

212 import doctest 

213 

214 failures = None 

215 if excinfo.errisinstance((doctest.DocTestFailure, doctest.UnexpectedException)): 

216 failures = [excinfo.value] 

217 elif excinfo.errisinstance(MultipleDoctestFailures): 

218 failures = excinfo.value.failures 

219 

220 if failures is not None: 

221 reprlocation_lines = [] 

222 for failure in failures: 

223 example = failure.example 

224 test = failure.test 

225 filename = test.filename 

226 if test.lineno is None: 

227 lineno = None 

228 else: 

229 lineno = test.lineno + example.lineno + 1 

230 message = type(failure).__name__ 

231 reprlocation = ReprFileLocation(filename, lineno, message) 

232 checker = _get_checker() 

233 report_choice = _get_report_choice( 

234 self.config.getoption("doctestreport") 

235 ) 

236 if lineno is not None: 

237 lines = failure.test.docstring.splitlines(False) 

238 # add line numbers to the left of the error message 

239 lines = [ 

240 "%03d %s" % (i + test.lineno + 1, x) 

241 for (i, x) in enumerate(lines) 

242 ] 

243 # trim docstring error lines to 10 

244 lines = lines[max(example.lineno - 9, 0) : example.lineno + 1] 

245 else: 

246 lines = [ 

247 "EXAMPLE LOCATION UNKNOWN, not showing all tests of that example" 

248 ] 

249 indent = ">>>" 

250 for line in example.source.splitlines(): 

251 lines.append("??? %s %s" % (indent, line)) 

252 indent = "..." 

253 if isinstance(failure, doctest.DocTestFailure): 

254 lines += checker.output_difference( 

255 example, failure.got, report_choice 

256 ).split("\n") 

257 else: 

258 inner_excinfo = ExceptionInfo(failure.exc_info) 

259 lines += ["UNEXPECTED EXCEPTION: %s" % repr(inner_excinfo.value)] 

260 lines += traceback.format_exception(*failure.exc_info) 

261 reprlocation_lines.append((reprlocation, lines)) 

262 return ReprFailDoctest(reprlocation_lines) 

263 else: 

264 return super(DoctestItem, self).repr_failure(excinfo)

392def _get_checker(): 

393 """ 

394 Returns a doctest.OutputChecker subclass that takes in account the 

395 ALLOW_UNICODE option to ignore u'' prefixes in strings and ALLOW_BYTES 

396 to strip b'' prefixes. 

397 Useful when the same doctest should run in Python 2 and Python 3. 

398 

399 An inner class is used to avoid importing "doctest" at the module 

400 level. 

401 """ 

402 if hasattr(_get_checker, "LiteralsOutputChecker"): 

403 return _get_checker.LiteralsOutputChecker() 

404 

405 import doctest 

406 import re 

407 

408 class LiteralsOutputChecker(doctest.OutputChecker): 

409 """ 

410 Copied from doctest_nose_plugin.py from the nltk project: 

411 https://github.com/nltk/nltk 

412 

413 Further extended to also support byte literals. 

414 """ 

415 

416 _unicode_literal_re = re.compile(r"(\W|^)[uU]([rR]?[\'\"])", re.UNICODE) 

417 _bytes_literal_re = re.compile(r"(\W|^)[bB]([rR]?[\'\"])", re.UNICODE) 

418 

419 def check_output(self, want, got, optionflags): 

420 res = doctest.OutputChecker.check_output(self, want, got, optionflags) 

421 if res: 

422 return True 

423 

424 allow_unicode = optionflags & _get_allow_unicode_flag() 

425 allow_bytes = optionflags & _get_allow_bytes_flag() 

426 if not allow_unicode and not allow_bytes: 

427 return False 

428 

429 else: # pragma: no cover 

430 

431 def remove_prefixes(regex, txt): 

432 return re.sub(regex, r"\1\2", txt) 

433 

434 if allow_unicode: 

435 want = remove_prefixes(self._unicode_literal_re, want) 

436 got = remove_prefixes(self._unicode_literal_re, got) 

437 if allow_bytes: 

438 want = remove_prefixes(self._bytes_literal_re, want) 

439 got = remove_prefixes(self._bytes_literal_re, got) 

440 res = doctest.OutputChecker.check_output(self, want, got, optionflags) 

441 return res 

442 

443 _get_checker.LiteralsOutputChecker = LiteralsOutputChecker 

444 return _get_checker.LiteralsOutputChecker()

481def _fix_spoof_python2(runner, encoding): 

482 """ 

483 Installs a "SpoofOut" into the given DebugRunner so it properly deals with unicode output. This 

484 should patch only doctests for text files because they don't have a way to declare their 

485 encoding. Doctests in docstrings from Python modules don't have the same problem given that 

486 Python already decoded the strings. 

487 

488 This fixes the problem related in issue #2434. 

489 """ 

490 from _pytest.compat import _PY2 

491 

492 if not _PY2: 

493 return 

494 

495 from doctest import _SpoofOut 

496 

497 class UnicodeSpoof(_SpoofOut): 

498 def getvalue(self): 

499 result = _SpoofOut.getvalue(self) 

500 if encoding and isinstance(result, bytes): 

501 result = result.decode(encoding) 

502 return result 

503 

504 runner._fakeout = UnicodeSpoof()
