260 def supports_transactions(self): 

261 """Confirm support for transactions.""" 

262 with self.connection.cursor() as cursor: 

263 cursor.execute('CREATE TABLE ROLLBACK_TEST (X INT)') 

264 self.connection.set_autocommit(False) 

265 cursor.execute('INSERT INTO ROLLBACK_TEST (X) VALUES (8)') 

266 self.connection.rollback() 

267 self.connection.set_autocommit(True) 

268 cursor.execute('SELECT COUNT(X) FROM ROLLBACK_TEST') 

269 count, = cursor.fetchone() 

270 cursor.execute('DROP TABLE ROLLBACK_TEST') 

271 return count == 0
