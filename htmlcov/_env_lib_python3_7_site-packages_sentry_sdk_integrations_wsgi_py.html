281def _make_wsgi_event_processor(environ, use_x_forwarded_for): 

282 # type: (Dict[str, str], bool) -> EventProcessor 

283 # It's a bit unfortunate that we have to extract and parse the request data 

284 # from the environ so eagerly, but there are a few good reasons for this. 

285 # 

286 # We might be in a situation where the scope/hub never gets torn down 

287 # properly. In that case we will have an unnecessary strong reference to 

288 # all objects in the environ (some of which may take a lot of memory) when 

289 # we're really just interested in a few of them. 

290 # 

291 # Keeping the environment around for longer than the request lifecycle is 

292 # also not necessarily something uWSGI can deal with: 

293 # https://github.com/unbit/uwsgi/issues/1950 

294 

295 client_ip = get_client_ip(environ) 

296 request_url = get_request_url(environ, use_x_forwarded_for) 

297 query_string = environ.get("QUERY_STRING") 

298 method = environ.get("REQUEST_METHOD") 

299 env = dict(_get_environ(environ)) 

300 headers = _filter_headers(dict(_get_headers(environ)))
