2Blizzard Mipmap Format (.blp) 

3Jerome Leclanche <jerome@leclan.ch> 

4 

5The contents of this file are hereby released in the public domain (CC0) 

6Full text of the CC0 license: 

7 https://creativecommons.org/publicdomain/zero/1.0/ 

8 

9BLP1 files, used mostly in Warcraft III, are not fully supported. 

10All types of BLP2 files used in World of Warcraft are supported. 

11 

12The BLP file structure consists of a header, up to 16 mipmaps of the 

13texture 

14 

15Texture sizes must be powers of two, though the two dimensions do 

16not have to be equal; 512x256 is valid, but 512x200 is not. 

17The first mipmap (mipmap #0) is the full size image; each subsequent 

18mipmap halves both dimensions. The final mipmap should be 1x1. 

19 

20BLP files come in many different flavours: 

21* JPEG-compressed (type == 0) - only supported for BLP1. 

22* RAW images (type == 1, encoding == 1). Each mipmap is stored as an 

23 array of 8-bit values, one per pixel, left to right, top to bottom. 

24 Each value is an index to the palette. 

25* DXT-compressed (type == 1, encoding == 2): 

26- DXT1 compression is used if alpha_encoding == 0. 

27 - An additional alpha bit is used if alpha_depth == 1. 

28 - DXT3 compression is used if alpha_encoding == 1. 

29 - DXT5 compression is used if alpha_encoding == 7. 

30"""

48def unpack_565(i): 

49 return (((i >> 11) & 0x1F) << 3, ((i >> 5) & 0x3F) << 2, (i & 0x1F) << 3)

52def decode_dxt1(data, alpha=False): 

53 """ 

54 input: one "row" of data (i.e. will produce 4*width pixels) 

55 """ 

56 

57 blocks = len(data) // 8 # number of blocks in row 

58 ret = (bytearray(), bytearray(), bytearray(), bytearray()) 

59 

60 for block in range(blocks): 

61 # Decode next 8-byte block. 

62 idx = block * 8 

63 color0, color1, bits = struct.unpack_from("<HHI", data, idx) 

64 

65 r0, g0, b0 = unpack_565(color0) 

66 r1, g1, b1 = unpack_565(color1) 

67 

68 # Decode this block into 4x4 pixels 

69 # Accumulate the results onto our 4 row accumulators 

70 for j in range(4): 

71 for i in range(4): 

72 # get next control op and generate a pixel 

73 

74 control = bits & 3 

75 bits = bits >> 2 

76 

77 a = 0xFF 

78 if control == 0: 

79 r, g, b = r0, g0, b0 

80 elif control == 1: 

81 r, g, b = r1, g1, b1 

82 elif control == 2: 

83 if color0 > color1: 

84 r = (2 * r0 + r1) // 3 

85 g = (2 * g0 + g1) // 3 

86 b = (2 * b0 + b1) // 3 

87 else: 

88 r = (r0 + r1) // 2 

89 g = (g0 + g1) // 2 

90 b = (b0 + b1) // 2 

91 elif control == 3: 

92 if color0 > color1: 

93 r = (2 * r1 + r0) // 3 

94 g = (2 * g1 + g0) // 3 

95 b = (2 * b1 + b0) // 3 

96 else: 

97 r, g, b, a = 0, 0, 0, 0 

98 

99 if alpha: 

100 ret[j].extend([r, g, b, a]) 

101 else: 

102 ret[j].extend([r, g, b]) 

103 

104 return ret

107def decode_dxt3(data): 

108 """ 

109 input: one "row" of data (i.e. will produce 4*width pixels) 

110 """ 

111 

112 blocks = len(data) // 16 # number of blocks in row 

113 ret = (bytearray(), bytearray(), bytearray(), bytearray()) 

114 

115 for block in range(blocks): 

116 idx = block * 16 

117 block = data[idx : idx + 16] 

118 # Decode next 16-byte block. 

119 bits = struct.unpack_from("<8B", block) 

120 color0, color1 = struct.unpack_from("<HH", block, 8) 

121 

122 (code,) = struct.unpack_from("<I", block, 12) 

123 

124 r0, g0, b0 = unpack_565(color0) 

125 r1, g1, b1 = unpack_565(color1) 

126 

127 for j in range(4): 

128 high = False # Do we want the higher bits? 

129 for i in range(4): 

130 alphacode_index = (4 * j + i) // 2 

131 a = bits[alphacode_index] 

132 if high: 

133 high = False 

134 a >>= 4 

135 else: 

136 high = True 

137 a &= 0xF 

138 a *= 17 # We get a value between 0 and 15 

139 

140 color_code = (code >> 2 * (4 * j + i)) & 0x03 

141 

142 if color_code == 0: 

143 r, g, b = r0, g0, b0 

144 elif color_code == 1: 

145 r, g, b = r1, g1, b1 

146 elif color_code == 2: 

147 r = (2 * r0 + r1) // 3 

148 g = (2 * g0 + g1) // 3 

149 b = (2 * b0 + b1) // 3 

150 elif color_code == 3: 

151 r = (2 * r1 + r0) // 3 

152 g = (2 * g1 + g0) // 3 

153 b = (2 * b1 + b0) // 3 

154 

155 ret[j].extend([r, g, b, a]) 

156 

157 return ret

160def decode_dxt5(data): 

161 """ 

162 input: one "row" of data (i.e. will produce 4 * width pixels) 

163 """ 

164 

165 blocks = len(data) // 16 # number of blocks in row 

166 ret = (bytearray(), bytearray(), bytearray(), bytearray()) 

167 

168 for block in range(blocks): 

169 idx = block * 16 

170 block = data[idx : idx + 16] 

171 # Decode next 16-byte block. 

172 a0, a1 = struct.unpack_from("<BB", block) 

173 

174 bits = struct.unpack_from("<6B", block, 2) 

175 alphacode1 = bits[2] | (bits[3] << 8) | (bits[4] << 16) | (bits[5] << 24) 

176 alphacode2 = bits[0] | (bits[1] << 8) 

177 

178 color0, color1 = struct.unpack_from("<HH", block, 8) 

179 

180 (code,) = struct.unpack_from("<I", block, 12) 

181 

182 r0, g0, b0 = unpack_565(color0) 

183 r1, g1, b1 = unpack_565(color1) 

184 

185 for j in range(4): 

186 for i in range(4): 

187 # get next control op and generate a pixel 

188 alphacode_index = 3 * (4 * j + i) 

189 

190 if alphacode_index <= 12: 

191 alphacode = (alphacode2 >> alphacode_index) & 0x07 

192 elif alphacode_index == 15: 

193 alphacode = (alphacode2 >> 15) | ((alphacode1 << 1) & 0x06) 

194 else: # alphacode_index >= 18 and alphacode_index <= 45 

195 alphacode = (alphacode1 >> (alphacode_index - 16)) & 0x07 

196 

197 if alphacode == 0: 

198 a = a0 

199 elif alphacode == 1: 

200 a = a1 

201 elif a0 > a1: 

202 a = ((8 - alphacode) * a0 + (alphacode - 1) * a1) // 7 

203 elif alphacode == 6: 

204 a = 0 

205 elif alphacode == 7: 

206 a = 255 

207 else: 

208 a = ((6 - alphacode) * a0 + (alphacode - 1) * a1) // 5 

209 

210 color_code = (code >> 2 * (4 * j + i)) & 0x03 

211 

212 if color_code == 0: 

213 r, g, b = r0, g0, b0 

214 elif color_code == 1: 

215 r, g, b = r1, g1, b1 

216 elif color_code == 2: 

217 r = (2 * r0 + r1) // 3 

218 g = (2 * g0 + g1) // 3 

219 b = (2 * b0 + b1) // 3 

220 elif color_code == 3: 

221 r = (2 * r1 + r0) // 3 

222 g = (2 * g1 + g0) // 3 

223 b = (2 * b1 + b0) // 3 

224 

225 ret[j].extend([r, g, b, a]) 

226 

227 return ret
