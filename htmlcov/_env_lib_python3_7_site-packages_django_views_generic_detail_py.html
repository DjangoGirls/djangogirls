8class SingleObjectMixin(ContextMixin): 

9 """ 

10 Provide the ability to retrieve a single object for further manipulation. 

11 """ 

12 model = None 

13 queryset = None 

14 slug_field = 'slug' 

15 context_object_name = None 

16 slug_url_kwarg = 'slug' 

17 pk_url_kwarg = 'pk' 

18 query_pk_and_slug = False

20 def get_object(self, queryset=None): 

21 """ 

22 Return the object the view is displaying. 

23 

24 Require `self.queryset` and a `pk` or `slug` argument in the URLconf. 

25 Subclasses can override this to return any object. 

26 """ 

27 # Use a custom queryset if provided; this is required for subclasses 

28 # like DateDetailView 

29 if queryset is None: 

30 queryset = self.get_queryset() 

31 

32 # Next, try looking up by primary key. 

33 pk = self.kwargs.get(self.pk_url_kwarg) 

34 slug = self.kwargs.get(self.slug_url_kwarg) 

35 if pk is not None: 

36 queryset = queryset.filter(pk=pk) 

37 

38 # Next, try looking up by slug. 

39 if slug is not None and (pk is None or self.query_pk_and_slug): 

40 slug_field = self.get_slug_field() 

41 queryset = queryset.filter(**{slug_field: slug}) 

42 

43 # If none of those are defined, it's an error. 

44 if pk is None and slug is None: 

45 raise AttributeError("Generic detail view %s must be called with " 

46 "either an object pk or a slug." 

47 % self.__class__.__name__) 

48 

49 try: 

50 # Get the single item from the filtered queryset 

51 obj = queryset.get() 

52 except queryset.model.DoesNotExist: 

53 raise Http404(_("No %(verbose_name)s found matching the query") % 

54 {'verbose_name': queryset.model._meta.verbose_name}) 

55 return obj 

56 

57 def get_queryset(self): 

58 """ 

59 Return the `QuerySet` that will be used to look up the object. 

60 

61 This method is called by the default implementation of get_object() and 

62 may not be called if get_object() is overridden. 

63 """ 

64 if self.queryset is None: 

65 if self.model: 

66 return self.model._default_manager.all() 

67 else: 

68 raise ImproperlyConfigured( 

69 "%(cls)s is missing a QuerySet. Define " 

70 "%(cls)s.model, %(cls)s.queryset, or override " 

71 "%(cls)s.get_queryset()." % { 

72 'cls': self.__class__.__name__ 

73 } 

74 ) 

75 return self.queryset.all()

77 def get_slug_field(self): 

78 """Get the name of a slug field to be used to look up by slug.""" 

79 return self.slug_field

83 if self.context_object_name: 

84 return self.context_object_name

110class SingleObjectTemplateResponseMixin(TemplateResponseMixin): 

111 template_name_field = None 

112 template_name_suffix = '_detail' 

113 

114 def get_template_names(self): 

115 """ 

116 Return a list of template names to be used for the request. May not be 

117 called if render_to_response() is overridden. Return the following list: 

118 

119 * the value of ``template_name`` on the view (if provided) 

120 * the contents of the ``template_name_field`` field on the 

121 object instance that the view is operating upon (if available) 

122 * ``<app_label>/<model_name><template_name_suffix>.html`` 

123 """ 

124 try: 

125 names = super().get_template_names() 

126 except ImproperlyConfigured: 

127 # If template_name isn't specified, it's not a problem -- 

128 # we just start with an empty list. 

129 names = [] 

130 

131 # If self.template_name_field is set, grab the value of the field 

132 # of that name from the object; this is the most specific template 

133 # name, if given. 

134 if self.object and self.template_name_field: 

135 name = getattr(self.object, self.template_name_field, None) 

136 if name: 

137 names.insert(0, name) 

138 

139 # The least-specific option is the default <app>/<model>_detail.html; 

140 # only use this if the object in question is a model. 

141 if isinstance(self.object, models.Model): 

142 object_meta = self.object._meta 

143 names.append("%s/%s%s.html" % ( 

144 object_meta.app_label, 

145 object_meta.model_name, 

146 self.template_name_suffix 

147 )) 

148 elif hasattr(self, 'model') and self.model is not None and issubclass(self.model, models.Model): 

149 names.append("%s/%s%s.html" % ( 

150 self.model._meta.app_label, 

151 self.model._meta.model_name, 

152 self.template_name_suffix 

153 )) 

154 

155 # If we still haven't managed to find any template names, we should 

156 # re-raise the ImproperlyConfigured to alert the user. 

157 if not names: 

158 raise 

159 

160 return names
