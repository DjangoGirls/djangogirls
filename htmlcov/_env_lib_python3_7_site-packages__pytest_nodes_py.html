21def _splitnode(nodeid): 

22 """Split a nodeid into constituent 'parts'. 

23 

24 Node IDs are strings, and can be things like: 

25 '' 

26 'testing/code' 

27 'testing/code/test_excinfo.py' 

28 'testing/code/test_excinfo.py::TestFormattedExcinfo::()' 

29 

30 Return values are lists e.g. 

31 [] 

32 ['testing', 'code'] 

33 ['testing', 'code', 'test_excinfo.py'] 

34 ['testing', 'code', 'test_excinfo.py', 'TestFormattedExcinfo', '()'] 

35 """ 

36 if nodeid == "": 

37 # If there is no root node at all, return an empty list so the caller's logic can remain sane 

38 return [] 

39 parts = nodeid.split(SEP) 

40 # Replace single last element 'test_foo.py::Bar::()' with multiple elements 'test_foo.py', 'Bar', '()' 

41 parts[-1:] = parts[-1].split("::") 

42 return parts

397def get_fslocation_from_item(item): 

398 """Tries to extract the actual location from an item, depending on available attributes:

401 * "obj": a Python object that the item wraps.
