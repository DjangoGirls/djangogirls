59def APP(self, marker): 

60 # 

61 # Application marker. Store these in the APP dictionary. 

62 # Also look for well-known application markers. 

63 

64 n = i16(self.fp.read(2)) - 2 

65 s = ImageFile._safe_read(self.fp, n) 

66 

67 app = "APP%d" % (marker & 15) 

68 

69 self.app[app] = s # compatibility 

70 self.applist.append((app, s)) 

71 

72 if marker == 0xFFE0 and s[:4] == b"JFIF": 

73 # extract JFIF information 

74 self.info["jfif"] = version = i16(s, 5) # version 

75 self.info["jfif_version"] = divmod(version, 256) 

76 # extract JFIF properties 

77 try: 

78 jfif_unit = s[7] 

79 jfif_density = i16(s, 8), i16(s, 10) 

80 except Exception: 

81 pass 

82 else: 

83 if jfif_unit == 1: 

84 self.info["dpi"] = jfif_density 

85 self.info["jfif_unit"] = jfif_unit 

86 self.info["jfif_density"] = jfif_density 

87 elif marker == 0xFFE1 and s[:5] == b"Exif\0": 

88 if "exif" not in self.info: 

89 # extract EXIF information (incomplete) 

90 self.info["exif"] = s # FIXME: value will change 

91 elif marker == 0xFFE2 and s[:5] == b"FPXR\0": 

92 # extract FlashPix information (incomplete) 

93 self.info["flashpix"] = s # FIXME: value will change 

94 elif marker == 0xFFE2 and s[:12] == b"ICC_PROFILE\0": 

95 # Since an ICC profile can be larger than the maximum size of 

96 # a JPEG marker (64K), we need provisions to split it into 

97 # multiple markers. The format defined by the ICC specifies 

98 # one or more APP2 markers containing the following data: 

99 # Identifying string ASCII "ICC_PROFILE\0" (12 bytes) 

100 # Marker sequence number 1, 2, etc (1 byte) 

101 # Number of markers Total of APP2's used (1 byte) 

102 # Profile data (remainder of APP2 data) 

103 # Decoders should use the marker sequence numbers to 

104 # reassemble the profile, rather than assuming that the APP2 

105 # markers appear in the correct sequence. 

106 self.icclist.append(s) 

107 elif marker == 0xFFED and s[:14] == b"Photoshop 3.0\x00": 

108 # parse the image resource block 

109 offset = 14 

110 photoshop = self.info.setdefault("photoshop", {}) 

111 while s[offset : offset + 4] == b"8BIM": 

112 try: 

113 offset += 4 

114 # resource code 

115 code = i16(s, offset) 

116 offset += 2 

117 # resource name (usually empty) 

118 name_len = s[offset] 

119 # name = s[offset+1:offset+1+name_len] 

120 offset += 1 + name_len 

121 offset += offset & 1 # align 

122 # resource data block 

123 size = i32(s, offset) 

124 offset += 4 

125 data = s[offset : offset + size] 

126 if code == 0x03ED: # ResolutionInfo 

127 data = { 

128 "XResolution": i32(data, 0) / 65536, 

129 "DisplayedUnitsX": i16(data, 4), 

130 "YResolution": i32(data, 8) / 65536, 

131 "DisplayedUnitsY": i16(data, 12), 

132 } 

133 photoshop[code] = data 

134 offset += size 

135 offset += offset & 1 # align 

136 except struct.error: 

137 break # insufficient data 

138 

139 elif marker == 0xFFEE and s[:5] == b"Adobe": 

140 self.info["adobe"] = i16(s, 5) 

141 # extract Adobe custom properties 

142 try: 

143 adobe_transform = s[1] 

144 except Exception: 

145 pass 

146 else: 

147 self.info["adobe_transform"] = adobe_transform 

148 elif marker == 0xFFE2 and s[:4] == b"MPF\0": 

149 # extract MPO information 

150 self.info["mp"] = s[4:] 

151 # offset is current location minus buffer size 

152 # plus constant header size 

153 self.info["mpoffset"] = self.fp.tell() - n + 4 

154 

155 # If DPI isn't in JPEG header, fetch from EXIF 

156 if "dpi" not in self.info and "exif" in self.info: 

157 try: 

158 exif = self.getexif() 

159 resolution_unit = exif[0x0128] 

160 x_resolution = exif[0x011A] 

161 try: 

162 dpi = float(x_resolution[0]) / x_resolution[1] 

163 except TypeError: 

164 dpi = x_resolution 

165 if resolution_unit == 3: # cm 

166 # 1 dpcm = 2.54 dpi 

167 dpi *= 2.54 

168 self.info["dpi"] = int(dpi + 0.5), int(dpi + 0.5) 

169 except (KeyError, SyntaxError, ValueError, ZeroDivisionError): 

170 # SyntaxError for invalid/unreadable EXIF 

171 # KeyError for dpi not included 

172 # ZeroDivisionError for invalid dpi rational value 

173 # ValueError for x_resolution[0] being an invalid float 

174 self.info["dpi"] = 72, 72

188def SOF(self, marker): 

189 # 

190 # Start of frame marker. Defines the size and mode of the 

191 # image. JPEG is colour blind, so we use some simple 

192 # heuristics to map the number of layers to an appropriate 

193 # mode. Note that this could be made a bit brighter, by 

194 # looking for JFIF and Adobe APP markers. 

195 

196 n = i16(self.fp.read(2)) - 2 

197 s = ImageFile._safe_read(self.fp, n) 

198 self._size = i16(s, 3), i16(s, 1) 

199 

200 self.bits = s[0] 

201 if self.bits != 8: 

202 raise SyntaxError(f"cannot handle {self.bits}-bit layers") 

203 

204 self.layers = s[5] 

205 if self.layers == 1: 

206 self.mode = "L" 

207 elif self.layers == 3: 

208 self.mode = "RGB" 

209 elif self.layers == 4: 

210 self.mode = "CMYK" 

211 else: 

212 raise SyntaxError(f"cannot handle {self.layers}-layer images") 

213 

214 if marker in [0xFFC2, 0xFFC6, 0xFFCA, 0xFFCE]: 

215 self.info["progressive"] = self.info["progression"] = 1 

216 

217 if self.icclist: 

218 # fixup icc profile 

219 self.icclist.sort() # sort by sequence number 

220 if self.icclist[0][13] == len(self.icclist): 

221 profile = [] 

222 for p in self.icclist: 

223 profile.append(p[14:]) 

224 icc_profile = b"".join(profile) 

225 else: 

226 icc_profile = None # wrong number of fragments 

227 self.info["icc_profile"] = icc_profile 

228 self.icclist = [] 

229 

230 for i in range(6, len(s), 3): 

231 t = s[i : i + 3] 

232 # 4-tuples: id, vsamp, hsamp, qtable 

233 self.layer.append((t[0], t[1] // 16, t[1] & 15, t[2]))

236def DQT(self, marker): 

237 # 

238 # Define quantization table. Note that there might be more 

239 # than one table in each marker. 

240 

241 # FIXME: The quantization tables can be used to estimate the 

242 # compression quality. 

243 

244 n = i16(self.fp.read(2)) - 2 

245 s = ImageFile._safe_read(self.fp, n) 

246 while len(s): 

247 v = s[0] 

248 precision = 1 if (v // 16 == 0) else 2 # in bytes 

249 qt_length = 1 + precision * 64 

250 if len(s) < qt_length: 

251 raise SyntaxError("bad quantization table marker") 

252 data = array.array("B" if precision == 1 else "H", s[1:qt_length]) 

253 if sys.byteorder == "little" and precision > 1: 

254 data.byteswap() # the values are always big-endian 

255 self.quantization[v & 15] = data 

256 s = s[qt_length:]

343 def _open(self): 

344 

345 s = self.fp.read(3) 

346 

347 if not _accept(s): 

348 raise SyntaxError("not a JPEG file") 

349 s = b"\xFF" 

350 

351 # Create attributes 

352 self.bits = self.layers = 0 

353 

354 # JPEG specifics (internal) 

355 self.layer = [] 

356 self.huffman_dc = {} 

357 self.huffman_ac = {} 

358 self.quantization = {} 

359 self.app = {} # compatibility 

360 self.applist = [] 

361 self.icclist = [] 

362 self._xmp = None 

363 

364 while True: 

365 

366 i = s[0] 

367 if i == 0xFF: 

368 s = s + self.fp.read(1) 

369 i = i16(s) 

370 else: 

371 # Skip non-0xFF junk 

372 s = self.fp.read(1) 

373 continue 

374 

375 if i in MARKER: 

376 name, description, handler = MARKER[i] 

377 if handler is not None: 

378 handler(self, i) 

379 if i == 0xFFDA: # start of scan 

380 rawmode = self.mode 

381 if self.mode == "CMYK": 

382 rawmode = "CMYK;I" # assume adobe conventions 

383 self.tile = [("jpeg", (0, 0) + self.size, 0, (rawmode, ""))] 

384 # self.__offset = self.fp.tell() 

385 break 

386 s = self.fp.read(1) 

387 elif i == 0 or i == 0xFFFF: 

388 # padded marker or junk; move on 

389 s = b"\xff" 

390 elif i == 0xFF00: # Skip extraneous data (escaped 0xFF) 

391 s = self.fp.read(1) 

392 else: 

393 raise SyntaxError("no marker found")

395 def load_read(self, read_bytes): 

396 """ 

397 internal: read more image data 

398 For premature EOF and LOAD_TRUNCATED_IMAGES adds EOI marker 

399 so libjpeg can finish decoding 

400 """ 

401 s = self.fp.read(read_bytes) 

402 

403 if not s and ImageFile.LOAD_TRUNCATED_IMAGES: 

404 # Premature EOF. 

405 # Pretend file is finished adding EOI marker 

406 return b"\xFF\xD9" 

407 

408 return s

410 def draft(self, mode, size): 

411 

412 if len(self.tile) != 1: 

413 return 

414 

415 # Protect from second call 

416 if self.decoderconfig: 

417 return 

418 

419 d, e, o, a = self.tile[0] 

420 scale = 1 

421 original_size = self.size 

422 

423 if a[0] == "RGB" and mode in ["L", "YCbCr"]: 

424 self.mode = mode 

425 a = mode, "" 

426 

427 if size: 

428 scale = min(self.size[0] // size[0], self.size[1] // size[1]) 

429 for s in [8, 4, 2, 1]: 

430 if scale >= s: 

431 break 

432 e = ( 

433 e[0], 

434 e[1], 

435 (e[2] - e[0] + s - 1) // s + e[0], 

436 (e[3] - e[1] + s - 1) // s + e[1], 

437 ) 

438 self._size = ((self.size[0] + s - 1) // s, (self.size[1] + s - 1) // s) 

439 scale = s 

440 

441 self.tile = [(d, e, o, a)] 

442 self.decoderconfig = (scale, 0) 

443 

444 box = (0, 0, original_size[0] / scale, original_size[1] / scale) 

445 return (self.mode, box)

479 def getxmp(self): 

480 """ 

481 Returns a dictionary containing the XMP tags. 

482 :returns: XMP tags in a dictionary. 

483 """ 

484 

485 if self._xmp is None: 

486 self._xmp = {} 

487 

488 for segment, content in self.applist: 

489 if segment == "APP1": 

490 marker, xmp_tags = content.rsplit(b"\x00", 1) 

491 if marker == b"http://ns.adobe.com/xap/1.0/": 

492 root = xml.etree.ElementTree.fromstring(xmp_tags)

495 child.split("}")[1]: value 

496 for child, value in element.attrib.items() 

497 }

507def _getmp(self): 

508 # Extract MP information. This method was inspired by the "highly 

509 # experimental" _getexif version that's been in use for years now, 

510 # itself based on the ImageFileDirectory class in the TIFF plugin. 

511 

512 # The MP record essentially consists of a TIFF file embedded in a JPEG 

513 # application marker. 

514 try: 

515 data = self.info["mp"] 

516 except KeyError: 

517 return None 

518 file_contents = io.BytesIO(data) 

519 head = file_contents.read(8) 

520 endianness = ">" if head[:4] == b"\x4d\x4d\x00\x2a" else "<" 

521 # process dictionary 

522 try: 

523 info = TiffImagePlugin.ImageFileDirectory_v2(head) 

524 file_contents.seek(info.next) 

525 info.load(file_contents) 

526 mp = dict(info) 

527 except Exception as e: 

528 raise SyntaxError("malformed MP Index (unreadable directory)") from e 

529 # it's an error not to have a number of images 

530 try: 

531 quant = mp[0xB001] 

532 except KeyError as e: 

533 raise SyntaxError("malformed MP Index (no number of images)") from e 

534 # get MP entries 

535 mpentries = [] 

536 try: 

537 rawmpentries = mp[0xB002] 

538 for entrynum in range(0, quant): 

539 unpackedentry = struct.unpack_from( 

540 f"{endianness}LLLHH", rawmpentries, entrynum * 16 

541 ) 

542 labels = ("Attribute", "Size", "DataOffset", "EntryNo1", "EntryNo2") 

543 mpentry = dict(zip(labels, unpackedentry)) 

544 mpentryattr = { 

545 "DependentParentImageFlag": bool(mpentry["Attribute"] & (1 << 31)), 

546 "DependentChildImageFlag": bool(mpentry["Attribute"] & (1 << 30)), 

547 "RepresentativeImageFlag": bool(mpentry["Attribute"] & (1 << 29)), 

548 "Reserved": (mpentry["Attribute"] & (3 << 27)) >> 27, 

549 "ImageDataFormat": (mpentry["Attribute"] & (7 << 24)) >> 24, 

550 "MPType": mpentry["Attribute"] & 0x00FFFFFF, 

551 } 

552 if mpentryattr["ImageDataFormat"] == 0: 

553 mpentryattr["ImageDataFormat"] = "JPEG" 

554 else: 

555 raise SyntaxError("unsupported picture format in MPO") 

556 mptypemap = { 

557 0x000000: "Undefined", 

558 0x010001: "Large Thumbnail (VGA Equivalent)", 

559 0x010002: "Large Thumbnail (Full HD Equivalent)", 

560 0x020001: "Multi-Frame Image (Panorama)", 

561 0x020002: "Multi-Frame Image: (Disparity)", 

562 0x020003: "Multi-Frame Image: (Multi-Angle)", 

563 0x030000: "Baseline MP Primary Image", 

564 } 

565 mpentryattr["MPType"] = mptypemap.get(mpentryattr["MPType"], "Unknown") 

566 mpentry["Attribute"] = mpentryattr 

567 mpentries.append(mpentry) 

568 mp[0xB002] = mpentries 

569 except KeyError as e: 

570 raise SyntaxError("malformed MP Index (bad MP Entry)") from e 

571 # Next we should try and parse the individual image unique ID list; 

572 # we don't because I've never seen this actually used in a real MPO 

573 # file and so can't test it. 

574 return mp

609def convert_dict_qtables(qtables): 

610 qtables = [qtables[key] for key in range(len(qtables)) if key in qtables] 

611 for idx, table in enumerate(qtables): 

612 qtables[idx] = [table[i] for i in zigzag_index] 

613 return qtables

616def get_sampling(im): 

617 # There's no subsampling when images have only 1 layer 

618 # (grayscale images) or when they are CMYK (4 layers), 

619 # so set subsampling to the default value. 

620 # 

621 # NOTE: currently Pillow can't encode JPEG to YCCK format. 

622 # If YCCK support is added in the future, subsampling code will have 

623 # to be updated (here and in JpegEncode.c) to deal with 4 layers. 

624 if not hasattr(im, "layers") or im.layers in (1, 4): 

625 return -1 

626 sampling = im.layer[0][1:3] + im.layer[1][1:3] + im.layer[2][1:3] 

627 return samplings.get(sampling, -1)

630def _save(im, fp, filename): 

631 

632 try: 

633 rawmode = RAWMODE[im.mode] 

634 except KeyError as e: 

635 raise OSError(f"cannot write mode {im.mode} as JPEG") from e 

636 

637 info = im.encoderinfo 

638 

639 dpi = [round(x) for x in info.get("dpi", (0, 0))] 

640 

641 quality = info.get("quality", -1) 

642 subsampling = info.get("subsampling", -1) 

643 qtables = info.get("qtables") 

644 

645 if quality == "keep": 

646 quality = -1 

647 subsampling = "keep" 

648 qtables = "keep" 

649 elif quality in presets: 

650 preset = presets[quality] 

651 quality = -1 

652 subsampling = preset.get("subsampling", -1) 

653 qtables = preset.get("quantization") 

654 elif not isinstance(quality, int): 

655 raise ValueError("Invalid quality setting") 

656 else: 

657 if subsampling in presets: 

658 subsampling = presets[subsampling].get("subsampling", -1) 

659 if isinstance(qtables, str) and qtables in presets: 

660 qtables = presets[qtables].get("quantization") 

661 

662 if subsampling == "4:4:4": 

663 subsampling = 0 

664 elif subsampling == "4:2:2": 

665 subsampling = 1 

666 elif subsampling == "4:2:0": 

667 subsampling = 2 

668 elif subsampling == "4:1:1": 

669 # For compatibility. Before Pillow 4.3, 4:1:1 actually meant 4:2:0. 

670 # Set 4:2:0 if someone is still using that value. 

671 subsampling = 2 

672 elif subsampling == "keep": 

673 if im.format != "JPEG": 

674 raise ValueError("Cannot use 'keep' when original image is not a JPEG") 

675 subsampling = get_sampling(im)

677 def validate_qtables(qtables): 

678 if qtables is None: 

679 return qtables 

680 if isinstance(qtables, str): 

681 try: 

682 lines = [ 

683 int(num) 

684 for line in qtables.splitlines() 

685 for num in line.split("#", 1)[0].split() 

686 ] 

687 except ValueError as e: 

688 raise ValueError("Invalid quantization table") from e 

689 else: 

690 qtables = [lines[s : s + 64] for s in range(0, len(lines), 64)] 

691 if isinstance(qtables, (tuple, list, dict)): 

692 if isinstance(qtables, dict): 

693 qtables = convert_dict_qtables(qtables) 

694 elif isinstance(qtables, tuple): 

695 qtables = list(qtables) 

696 if not (0 < len(qtables) < 5): 

697 raise ValueError("None or too many quantization tables") 

698 for idx, table in enumerate(qtables): 

699 try: 

700 if len(table) != 64: 

701 raise TypeError 

702 table = array.array("H", table) 

703 except TypeError as e: 

704 raise ValueError("Invalid quantization table") from e 

705 else: 

706 qtables[idx] = list(table) 

707 return qtables 

708 

709 if qtables == "keep": 

710 if im.format != "JPEG": 

711 raise ValueError("Cannot use 'keep' when original image is not a JPEG") 

712 qtables = getattr(im, "quantization", None) 

713 qtables = validate_qtables(qtables) 

714 

715 extra = b"" 

716 

717 icc_profile = info.get("icc_profile") 

718 if icc_profile: 

719 ICC_OVERHEAD_LEN = 14 

720 MAX_BYTES_IN_MARKER = 65533 

721 MAX_DATA_BYTES_IN_MARKER = MAX_BYTES_IN_MARKER - ICC_OVERHEAD_LEN 

722 markers = [] 

723 while icc_profile: 

724 markers.append(icc_profile[:MAX_DATA_BYTES_IN_MARKER]) 

725 icc_profile = icc_profile[MAX_DATA_BYTES_IN_MARKER:] 

726 i = 1 

727 for marker in markers: 

728 size = struct.pack(">H", 2 + ICC_OVERHEAD_LEN + len(marker)) 

729 extra += ( 

730 b"\xFF\xE2" 

731 + size 

732 + b"ICC_PROFILE\0" 

733 + o8(i) 

734 + o8(len(markers)) 

735 + marker 

736 ) 

737 i += 1 

738 

739 # "progressive" is the official name, but older documentation 

740 # says "progression" 

741 # FIXME: issue a warning if the wrong form is used (post-1.1.7) 

742 progressive = info.get("progressive", False) or info.get("progression", False) 

743 

744 optimize = info.get("optimize", False) 

745 

746 exif = info.get("exif", b"") 

747 if isinstance(exif, Image.Exif): 

748 exif = exif.tobytes() 

749 

750 # get keyword arguments 

751 im.encoderconfig = ( 

752 quality, 

753 progressive, 

754 info.get("smooth", 0), 

755 optimize, 

756 info.get("streamtype", 0), 

757 dpi[0], 

758 dpi[1], 

759 subsampling, 

760 qtables, 

761 extra, 

762 exif, 

763 ) 

764 

765 # if we optimize, libjpeg needs a buffer big enough to hold the whole image 

766 # in a shot. Guessing on the size, at im.size bytes. (raw pixel size is 

767 # channels*size, this is a value that's been used in a django patch. 

768 # https://github.com/matthewwithanm/django-imagekit/issues/50 

769 bufsize = 0 

770 if optimize or progressive: 

771 # CMYK can be bigger 

772 if im.mode == "CMYK": 

773 bufsize = 4 * im.size[0] * im.size[1] 

774 # keep sets quality to -1, but the actual value may be high. 

775 elif quality >= 95 or quality == -1: 

776 bufsize = 2 * im.size[0] * im.size[1] 

777 else: 

778 bufsize = im.size[0] * im.size[1] 

779 

780 # The EXIF info needs to be written as one block, + APP1, + one spare byte. 

781 # Ensure that our buffer is big enough. Same with the icc_profile block. 

782 bufsize = max(ImageFile.MAXBLOCK, bufsize, len(exif) + 5, len(extra) + 1) 

783 

784 ImageFile._save(im, fp, [("jpeg", (0, 0) + im.size, 0, rawmode)], bufsize)

799def jpeg_factory(fp=None, filename=None): 

800 im = JpegImageFile(fp, filename)

802 mpheader = im._getmp() 

803 if mpheader[45057] > 1:
