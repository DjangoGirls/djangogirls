204 def last_executed_query(self, cursor, sql, params): 

205 """ 

206 Return a string of the query last executed by the given cursor, with 

207 placeholders replaced with actual values. 

208 

209 `sql` is the raw query containing placeholders and `params` is the 

210 sequence of parameters. These are used by default, but this method 

211 exists for database backends to provide a better implementation 

212 according to their own quoting schemes. 

213 """ 

214 # Convert params to contain string values. 

215 def to_string(s): 

216 return force_text(s, strings_only=True, errors='replace') 

217 if isinstance(params, (list, tuple)): 

218 u_params = tuple(to_string(val) for val in params) 

219 elif params is None: 

220 u_params = () 

221 else: 

222 u_params = {to_string(k): to_string(v) for k, v in params.items()} 

223 

224 return "QUERY = %r - PARAMS = %r" % (sql, u_params)

609 def window_frame_start(self, start): 

610 if isinstance(start, int): 

611 if start < 0: 

612 return '%d %s' % (abs(start), self.PRECEDING) 

613 elif start == 0: 

614 return self.CURRENT_ROW 

615 elif start is None: 

616 return self.UNBOUNDED_PRECEDING 

617 raise ValueError("start argument must be a negative integer, zero, or None, but got '%s'." % start) 

618 

619 def window_frame_end(self, end): 

620 if isinstance(end, int): 

621 if end == 0: 

622 return self.CURRENT_ROW 

623 elif end > 0: 

624 return '%d %s' % (end, self.FOLLOWING) 

625 elif end is None: 

626 return self.UNBOUNDED_FOLLOWING 

627 raise ValueError("end argument must be a positive integer, zero, or None, but got '%s'." % end)
