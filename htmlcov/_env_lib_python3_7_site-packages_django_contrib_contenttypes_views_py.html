9def shortcut(request, content_type_id, object_id): 

10 """ 

11 Redirect to an object's page based on a content-type ID and an object ID. 

12 """ 

13 # Look up the object, making sure it's got a get_absolute_url() function. 

14 try: 

15 content_type = ContentType.objects.get(pk=content_type_id) 

16 if not content_type.model_class(): 

17 raise Http404( 

18 _("Content type %(ct_id)s object has no associated model") % 

19 {'ct_id': content_type_id} 

20 ) 

21 obj = content_type.get_object_for_this_type(pk=object_id) 

22 except (ObjectDoesNotExist, ValueError): 

23 raise Http404( 

24 _("Content type %(ct_id)s object %(obj_id)s doesn't exist") % 

25 {'ct_id': content_type_id, 'obj_id': object_id} 

26 ) 

27 

28 try: 

29 get_absolute_url = obj.get_absolute_url 

30 except AttributeError: 

31 raise Http404( 

32 _("%(ct_name)s objects don't have a get_absolute_url() method") % 

33 {'ct_name': content_type.name} 

34 ) 

35 absurl = get_absolute_url() 

36 

37 # Try to figure out the object's domain, so we can do a cross-site redirect 

38 # if necessary. 

39 

40 # If the object actually defines a domain, we're done. 

41 if absurl.startswith(('http://', 'https://', '//')): 

42 return HttpResponseRedirect(absurl) 

43 

44 # Otherwise, we need to introspect the object's relationships for a 

45 # relation to the Site object 

46 object_domain = None 

47 

48 if apps.is_installed('django.contrib.sites'): 

49 Site = apps.get_model('sites.Site') 

50 

51 opts = obj._meta 

52 

53 # First, look for a many-to-many relationship to Site. 

54 for field in opts.many_to_many: 

55 if field.remote_field.model is Site: 

56 try: 

57 # Caveat: In the case of multiple related Sites, this just 

58 # selects the *first* one, which is arbitrary. 

59 object_domain = getattr(obj, field.name).all()[0].domain 

60 except IndexError: 

61 pass 

62 if object_domain is not None: 

63 break 

64 

65 # Next, look for a many-to-one relationship to Site. 

66 if object_domain is None: 

67 for field in obj._meta.fields: 

68 if field.remote_field and field.remote_field.model is Site: 

69 try: 

70 site = getattr(obj, field.name) 

71 except Site.DoesNotExist: 

72 continue 

73 if site is not None: 

74 object_domain = site.domain 

75 if object_domain is not None: 

76 break 

77 

78 # Fall back to the current site (if possible). 

79 if object_domain is None: 

80 try: 

81 object_domain = Site.objects.get_current(request).domain 

82 except Site.DoesNotExist: 

83 pass 

84 

85 else: 

86 # Fall back to the current request's site. 

87 object_domain = RequestSite(request).domain 

88 

89 # If all that malarkey found an object domain, use it. Otherwise, fall back 

90 # to whatever get_absolute_url() returned. 

91 if object_domain is not None: 

92 protocol = request.scheme 

93 return HttpResponseRedirect('%s://%s%s' % (protocol, object_domain, absurl)) 

94 else: 

95 return HttpResponseRedirect(absurl)
