20def serve(request, path, document_root=None, show_indexes=False): 

21 """ 

22 Serve static files below a given point in the directory structure. 

23 

24 To use, put a URL pattern such as:: 

25 

26 from django.views.static import serve 

27 

28 url(r'^(?P<path>.*)$', serve, {'document_root': '/path/to/my/files/'}) 

29 

30 in your URLconf. You must provide the ``document_root`` param. You may 

31 also set ``show_indexes`` to ``True`` if you'd like to serve a basic index 

32 of the directory. This index view will use the template hardcoded below, 

33 but if you'd like to override it, you can create a template called 

34 ``static/directory_index.html``. 

35 """ 

36 path = posixpath.normpath(path).lstrip('/') 

37 fullpath = safe_join(document_root, path) 

38 if os.path.isdir(fullpath): 

39 if show_indexes: 

40 return directory_index(path, fullpath) 

41 raise Http404(_("Directory indexes are not allowed here.")) 

42 if not os.path.exists(fullpath): 

43 raise Http404(_('"%(path)s" does not exist') % {'path': fullpath}) 

44 # Respect the If-Modified-Since header. 

45 statobj = os.stat(fullpath) 

46 if not was_modified_since(request.META.get('HTTP_IF_MODIFIED_SINCE'), 

47 statobj.st_mtime, statobj.st_size): 

48 return HttpResponseNotModified() 

49 content_type, encoding = mimetypes.guess_type(fullpath) 

50 content_type = content_type or 'application/octet-stream' 

51 response = FileResponse(open(fullpath, 'rb'), content_type=content_type) 

52 response["Last-Modified"] = http_date(statobj.st_mtime) 

53 if stat.S_ISREG(statobj.st_mode): 

54 response["Content-Length"] = statobj.st_size 

55 if encoding: 

56 response["Content-Encoding"] = encoding 

57 return response

86def directory_index(path, fullpath): 

87 try: 

88 t = loader.select_template([ 

89 'static/directory_index.html', 

90 'static/directory_index', 

91 ]) 

92 except TemplateDoesNotExist: 

93 t = Engine(libraries={'i18n': 'django.templatetags.i18n'}).from_string(DEFAULT_DIRECTORY_INDEX_TEMPLATE) 

94 c = Context() 

95 else: 

96 c = {} 

97 files = [] 

98 for f in os.listdir(fullpath): 

99 if not f.startswith('.'): 

100 if os.path.isdir(os.path.join(fullpath, f)): 

101 f += '/' 

102 files.append(f) 

103 c.update({ 

104 'directory': path + '/', 

105 'file_list': files, 

106 }) 

107 return HttpResponse(t.render(c))

110def was_modified_since(header=None, mtime=0, size=0): 

111 """ 

112 Was something modified since the user last downloaded it? 

113 

114 header 

115 This is the value of the If-Modified-Since header. If this is None, 

116 I'll just return True. 

117 

118 mtime 

119 This is the modification time of the item we're talking about. 

120 

121 size 

122 This is the size of the item we're talking about. 

123 """ 

124 try: 

125 if header is None: 

126 raise ValueError 

127 matches = re.match(r"^([^;]+)(; length=([0-9]+))?$", header, 

128 re.IGNORECASE) 

129 header_mtime = parse_http_date(matches.group(1)) 

130 header_len = matches.group(3) 

131 if header_len and int(header_len) != size: 

132 raise ValueError 

133 if int(mtime) > header_mtime: 

134 raise ValueError 

135 except (AttributeError, ValueError, OverflowError): 

136 return True 

137 return False
