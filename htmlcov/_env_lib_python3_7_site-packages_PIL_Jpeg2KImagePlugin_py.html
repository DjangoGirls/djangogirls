22def _parse_codestream(fp): 

23 """Parse the JPEG 2000 codestream to extract the size and component 

24 count from the SIZ marker segment, returning a PIL (size, mode) tuple.""" 

25 

26 hdr = fp.read(2) 

27 lsiz = struct.unpack(">H", hdr)[0] 

28 siz = hdr + fp.read(lsiz - 2) 

29 lsiz, rsiz, xsiz, ysiz, xosiz, yosiz, _, _, _, _, csiz = struct.unpack_from( 

30 ">HHIIIIIIIIH", siz 

31 ) 

32 ssiz = [None] * csiz 

33 xrsiz = [None] * csiz 

34 yrsiz = [None] * csiz 

35 for i in range(csiz): 

36 ssiz[i], xrsiz[i], yrsiz[i] = struct.unpack_from(">BBB", siz, 36 + 3 * i) 

37 

38 size = (xsiz - xosiz, ysiz - yosiz) 

39 if csiz == 1: 

40 if (yrsiz[0] & 0x7F) > 8: 

41 mode = "I;16" 

42 else: 

43 mode = "L" 

44 elif csiz == 2: 

45 mode = "LA" 

46 elif csiz == 3: 

47 mode = "RGB" 

48 elif csiz == 4: 

49 mode = "RGBA" 

50 else: 

51 mode = None 

52 

53 return (size, mode)

56def _parse_jp2_header(fp): 

57 """Parse the JP2 header box to extract size, component count and 

58 color space information, returning a (size, mode, mimetype) tuple.""" 

59 

60 # Find the JP2 header box 

61 header = None 

62 mimetype = None 

63 while True: 

64 lbox, tbox = struct.unpack(">I4s", fp.read(8)) 

65 if lbox == 1: 

66 lbox = struct.unpack(">Q", fp.read(8))[0] 

67 hlen = 16 

68 else: 

69 hlen = 8 

70 

71 if lbox < hlen: 

72 raise SyntaxError("Invalid JP2 header length") 

73 

74 if tbox == b"jp2h": 

75 header = fp.read(lbox - hlen) 

76 break 

77 elif tbox == b"ftyp": 

78 if fp.read(4) == b"jpx ": 

79 mimetype = "image/jpx" 

80 fp.seek(lbox - hlen - 4, os.SEEK_CUR) 

81 else: 

82 fp.seek(lbox - hlen, os.SEEK_CUR) 

83 

84 if header is None: 

85 raise SyntaxError("could not find JP2 header") 

86 

87 size = None 

88 mode = None 

89 bpc = None 

90 nc = None 

91 

92 hio = io.BytesIO(header) 

93 while True: 

94 lbox, tbox = struct.unpack(">I4s", hio.read(8)) 

95 if lbox == 1: 

96 lbox = struct.unpack(">Q", hio.read(8))[0] 

97 hlen = 16 

98 else: 

99 hlen = 8 

100 

101 content = hio.read(lbox - hlen) 

102 

103 if tbox == b"ihdr": 

104 height, width, nc, bpc, c, unkc, ipr = struct.unpack(">IIHBBBB", content) 

105 size = (width, height) 

106 if unkc: 

107 if nc == 1 and (bpc & 0x7F) > 8: 

108 mode = "I;16" 

109 elif nc == 1: 

110 mode = "L" 

111 elif nc == 2: 

112 mode = "LA" 

113 elif nc == 3: 

114 mode = "RGB" 

115 elif nc == 4: 

116 mode = "RGBA" 

117 break 

118 elif tbox == b"colr": 

119 meth, prec, approx = struct.unpack_from(">BBB", content) 

120 if meth == 1: 

121 cs = struct.unpack_from(">I", content, 3)[0] 

122 if cs == 16: # sRGB 

123 if nc == 1 and (bpc & 0x7F) > 8: 

124 mode = "I;16" 

125 elif nc == 1: 

126 mode = "L" 

127 elif nc == 3: 

128 mode = "RGB" 

129 elif nc == 4: 

130 mode = "RGBA" 

131 break 

132 elif cs == 17: # grayscale 

133 if nc == 1 and (bpc & 0x7F) > 8: 

134 mode = "I;16" 

135 elif nc == 1: 

136 mode = "L" 

137 elif nc == 2: 

138 mode = "LA" 

139 break 

140 elif cs == 18: # sYCC 

141 if nc == 3: 

142 mode = "RGB" 

143 elif nc == 4: 

144 mode = "RGBA" 

145 break 

146 

147 if size is None or mode is None: 

148 raise SyntaxError("Malformed jp2 header") 

149 

150 return (size, mode, mimetype)

157class Jpeg2KImageFile(ImageFile.ImageFile): 

158 format = "JPEG2000" 

159 format_description = "JPEG 2000 (ISO 15444)" 

160 

161 def _open(self): 

162 sig = self.fp.read(4) 

163 if sig == b"\xff\x4f\xff\x51": 

164 self.codec = "j2k" 

165 self._size, self.mode = _parse_codestream(self.fp) 

166 else: 

167 sig = sig + self.fp.read(8) 

168 

169 if sig == b"\x00\x00\x00\x0cjP \x0d\x0a\x87\x0a": 

170 self.codec = "jp2" 

171 header = _parse_jp2_header(self.fp) 

172 self._size, self.mode, self.custom_mimetype = header 

173 else: 

174 raise SyntaxError("not a JPEG 2000 file") 

175 

176 if self.size is None or self.mode is None: 

177 raise SyntaxError("unable to determine size/mode") 

178 

179 self._reduce = 0 

180 self.layers = 0 

181 

182 fd = -1 

183 length = -1 

184 

185 try: 

186 fd = self.fp.fileno() 

187 length = os.fstat(fd).st_size 

188 except Exception: 

189 fd = -1 

190 try: 

191 pos = self.fp.tell() 

192 self.fp.seek(0, io.SEEK_END) 

193 length = self.fp.tell() 

194 self.fp.seek(pos) 

195 except Exception: 

196 length = -1 

197 

198 self.tile = [ 

199 ( 

200 "jpeg2k", 

201 (0, 0) + self.size, 

202 0, 

203 (self.codec, self._reduce, self.layers, fd, length), 

204 ) 

205 ] 

206 

207 @property 

208 def reduce(self): 

209 # https://github.com/python-pillow/Pillow/issues/4343 found that the 

210 # new Image 'reduce' method was shadowed by this plugin's 'reduce' 

211 # property. This attempts to allow for both scenarios 

212 return self._reduce or super().reduce 

213 

214 @reduce.setter 

215 def reduce(self, value): 

216 self._reduce = value 

217 

218 def load(self): 

219 if self.tile and self._reduce: 

220 power = 1 << self._reduce 

221 adjust = power >> 1 

222 self._size = ( 

223 int((self.size[0] + adjust) / power), 

224 int((self.size[1] + adjust) / power), 

225 ) 

226 

227 # Update the reduce and layers settings 

228 t = self.tile[0] 

229 t3 = (t[3][0], self._reduce, self.layers, t[3][3], t[3][4]) 

230 self.tile = [(t[0], (0, 0) + self.size, t[2], t3)] 

231 

232 return ImageFile.ImageFile.load(self)
