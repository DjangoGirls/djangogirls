40def _save(im, fp, filename): 

41 fp.write(_MAGIC) # (2+2) 

42 sizes = im.encoderinfo.get( 

43 "sizes", 

44 [(16, 16), (24, 24), (32, 32), (48, 48), (64, 64), (128, 128), (256, 256)], 

45 ) 

46 width, height = im.size 

47 sizes = filter( 

48 lambda x: False 

49 if (x[0] > width or x[1] > height or x[0] > 256 or x[1] > 256) 

50 else True, 

51 sizes, 

52 ) 

53 sizes = list(sizes) 

54 fp.write(struct.pack("<H", len(sizes))) # idCount(2) 

55 offset = fp.tell() + len(sizes) * 16 

56 provided_images = {im.size: im for im in im.encoderinfo.get("append_images", [])} 

57 for size in sizes: 

58 width, height = size 

59 # 0 means 256 

60 fp.write(struct.pack("B", width if width < 256 else 0)) # bWidth(1) 

61 fp.write(struct.pack("B", height if height < 256 else 0)) # bHeight(1) 

62 fp.write(b"\0") # bColorCount(1) 

63 fp.write(b"\0") # bReserved(1) 

64 fp.write(b"\0\0") # wPlanes(2) 

65 fp.write(struct.pack("<H", 32)) # wBitCount(2) 

66 

67 image_io = BytesIO() 

68 tmp = provided_images.get(size) 

69 if not tmp: 

70 # TODO: invent a more convenient method for proportional scalings 

71 tmp = im.copy() 

72 tmp.thumbnail(size, Image.LANCZOS, reducing_gap=None) 

73 tmp.save(image_io, "png") 

74 image_io.seek(0) 

75 image_bytes = image_io.read() 

76 bytes_len = len(image_bytes) 

77 fp.write(struct.pack("<I", bytes_len)) # dwBytesInRes(4) 

78 fp.write(struct.pack("<I", offset)) # dwImageOffset(4) 

79 current = fp.tell() 

80 fp.seek(offset) 

81 fp.write(image_bytes) 

82 offset = offset + bytes_len 

83 fp.seek(current)

90class IcoFile: 

91 def __init__(self, buf): 

92 """ 

93 Parse image from file-like object containing ico file data 

94 """ 

95 

96 # check magic 

97 s = buf.read(6) 

98 if not _accept(s): 

99 raise SyntaxError("not an ICO file") 

100 

101 self.buf = buf 

102 self.entry = [] 

103 

104 # Number of items in file 

105 self.nb_items = i16(s, 4) 

106 

107 # Get headers for each item 

108 for i in range(self.nb_items): 

109 s = buf.read(16) 

110 

111 icon_header = { 

112 "width": s[0], 

113 "height": s[1], 

114 "nb_color": s[2], # No. of colors in image (0 if >=8bpp) 

115 "reserved": s[3], 

116 "planes": i16(s, 4), 

117 "bpp": i16(s, 6), 

118 "size": i32(s, 8), 

119 "offset": i32(s, 12), 

120 } 

121 

122 # See Wikipedia 

123 for j in ("width", "height"): 

124 if not icon_header[j]: 

125 icon_header[j] = 256 

126 

127 # See Wikipedia notes about color depth. 

128 # We need this just to differ images with equal sizes 

129 icon_header["color_depth"] = ( 

130 icon_header["bpp"] 

131 or ( 

132 icon_header["nb_color"] != 0 

133 and ceil(log(icon_header["nb_color"], 2)) 

134 ) 

135 or 256 

136 ) 

137 

138 icon_header["dim"] = (icon_header["width"], icon_header["height"]) 

139 icon_header["square"] = icon_header["width"] * icon_header["height"] 

140 

141 self.entry.append(icon_header) 

142 

143 self.entry = sorted(self.entry, key=lambda x: x["color_depth"]) 

144 # ICO images are usually squares 

145 # self.entry = sorted(self.entry, key=lambda x: x['width']) 

146 self.entry = sorted(self.entry, key=lambda x: x["square"]) 

147 self.entry.reverse() 

148 

149 def sizes(self): 

150 """ 

151 Get a list of all available icon sizes and color depths. 

152 """ 

153 return {(h["width"], h["height"]) for h in self.entry} 

154 

155 def getentryindex(self, size, bpp=False): 

156 for (i, h) in enumerate(self.entry): 

157 if size == h["dim"] and (bpp is False or bpp == h["color_depth"]): 

158 return i 

159 return 0 

160 

161 def getimage(self, size, bpp=False): 

162 """ 

163 Get an image from the icon 

164 """ 

165 return self.frame(self.getentryindex(size, bpp)) 

166 

167 def frame(self, idx): 

168 """ 

169 Get an image from frame idx 

170 """ 

171 

172 header = self.entry[idx] 

173 

174 self.buf.seek(header["offset"]) 

175 data = self.buf.read(8) 

176 self.buf.seek(header["offset"]) 

177 

178 if data[:8] == PngImagePlugin._MAGIC: 

179 # png frame 

180 im = PngImagePlugin.PngImageFile(self.buf) 

181 Image._decompression_bomb_check(im.size) 

182 else: 

183 # XOR + AND mask bmp frame 

184 im = BmpImagePlugin.DibImageFile(self.buf) 

185 Image._decompression_bomb_check(im.size) 

186 

187 # change tile dimension to only encompass XOR image 

188 im._size = (im.size[0], int(im.size[1] / 2)) 

189 d, e, o, a = im.tile[0] 

190 im.tile[0] = d, (0, 0) + im.size, o, a 

191 

192 # figure out where AND mask image starts 

193 mode = a[0] 

194 bpp = 8 

195 for k, v in BmpImagePlugin.BIT2MODE.items(): 

196 if mode == v[1]: 

197 bpp = k 

198 break 

199 

200 if 32 == bpp: 

201 # 32-bit color depth icon image allows semitransparent areas 

202 # PIL's DIB format ignores transparency bits, recover them. 

203 # The DIB is packed in BGRX byte order where X is the alpha 

204 # channel. 

205 

206 # Back up to start of bmp data 

207 self.buf.seek(o) 

208 # extract every 4th byte (eg. 3,7,11,15,...) 

209 alpha_bytes = self.buf.read(im.size[0] * im.size[1] * 4)[3::4] 

210 

211 # convert to an 8bpp grayscale image 

212 mask = Image.frombuffer( 

213 "L", # 8bpp 

214 im.size, # (w, h) 

215 alpha_bytes, # source chars 

216 "raw", # raw decoder 

217 ("L", 0, -1), # 8bpp inverted, unpadded, reversed 

218 ) 

219 else: 

220 # get AND image from end of bitmap 

221 w = im.size[0] 

222 if (w % 32) > 0: 

223 # bitmap row data is aligned to word boundaries 

224 w += 32 - (im.size[0] % 32) 

225 

226 # the total mask data is 

227 # padded row size * height / bits per char 

228 

229 and_mask_offset = o + int(im.size[0] * im.size[1] * (bpp / 8.0)) 

230 total_bytes = int((w * im.size[1]) / 8) 

231 

232 self.buf.seek(and_mask_offset) 

233 mask_data = self.buf.read(total_bytes) 

234 

235 # convert raw data to image 

236 mask = Image.frombuffer( 

237 "1", # 1 bpp 

238 im.size, # (w, h) 

239 mask_data, # source chars 

240 "raw", # raw decoder 

241 ("1;I", int(w / 8), -1), # 1bpp inverted, padded, reversed 

242 ) 

243 

244 # now we have two images, im is XOR image and mask is AND image 

245 

246 # apply mask image as alpha channel 

247 im = im.convert("RGBA") 

248 im.putalpha(mask) 

249 

250 return im

257class IcoImageFile(ImageFile.ImageFile): 

258 """ 

259 PIL read-only image support for Microsoft Windows .ico files. 

260 

261 By default the largest resolution image in the file will be loaded. This 

262 can be changed by altering the 'size' attribute before calling 'load'. 

263 

264 The info dictionary has a key 'sizes' that is a list of the sizes available 

265 in the icon file. 

266 

267 Handles classic, XP and Vista icon formats. 

268 

269 When saving, PNG compression is used. Support for this was only added in 

270 Windows Vista. 

271 

272 This plugin is a refactored version of Win32IconImagePlugin by Bryan Davis 

273 <casadebender@gmail.com>. 

274 https://code.google.com/archive/p/casadebender/wikis/Win32IconImagePlugin.wiki 

275 """ 

276 

277 format = "ICO" 

278 format_description = "Windows Icon" 

279 

280 def _open(self): 

281 self.ico = IcoFile(self.fp) 

282 self.info["sizes"] = self.ico.sizes() 

283 self.size = self.ico.entry[0]["dim"] 

284 self.load() 

285 

286 @property 

287 def size(self): 

288 return self._size 

289 

290 @size.setter 

291 def size(self, value): 

292 if value not in self.info["sizes"]: 

293 raise ValueError("This is not one of the allowed sizes of this image") 

294 self._size = value 

295 

296 def load(self): 

297 if self.im and self.im.size == self.size: 

298 # Already loaded 

299 return 

300 im = self.ico.getimage(self.size) 

301 # if tile is PNG, it won't really be loaded yet 

302 im.load() 

303 self.im = im.im 

304 self.mode = im.mode 

305 if im.size != self.size: 

306 warnings.warn("Image was not the expected size") 

307 

308 index = self.ico.getentryindex(self.size) 

309 sizes = list(self.info["sizes"]) 

310 sizes[index] = im.size 

311 self.info["sizes"] = set(sizes) 

312 

313 self.size = im.size
