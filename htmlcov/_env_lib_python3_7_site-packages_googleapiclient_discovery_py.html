127def fix_method_name(name): 

128 """Fix method names to avoid '$' characters and reserved word conflicts. 

129

132 

133 Returns: 

134 The name with '_' appended if the name is a reserved word and '$'

144def key2param(key): 

145 """Converts key names into parameter names. 

146 

147 For example, converting "max-results" -> "max_results" 

148 

149 Args: 

150 key: string, the method key name. 

151 

152 Returns: 

153 A safe method name based on the key name. 

154 """ 

155 result = [] 

156 key = list(key) 

157 if not key[0].isalpha(): 

158 result.append('x') 

159 for c in key: 

160 if c.isalnum(): 

161 result.append(c) 

162 else: 

163 result.append('_') 

164 

165 return ''.join(result)

168@positional(2) 

169def build(serviceName, 

170 version, 

171 http=None, 

172 discoveryServiceUrl=DISCOVERY_URI, 

173 developerKey=None, 

174 model=None, 

175 requestBuilder=HttpRequest, 

176 credentials=None, 

177 cache_discovery=True, 

178 cache=None): 

179 """Construct a Resource for interacting with an API. 

180 

181 Construct a Resource object for interacting with an API. The serviceName and 

182 version are the names from the Discovery service. 

183 

184 Args: 

185 serviceName: string, name of the service. 

186 version: string, the version of the service. 

187 http: httplib2.Http, An instance of httplib2.Http or something that acts 

188 like it that HTTP requests will be made through. 

189 discoveryServiceUrl: string, a URI Template that points to the location of 

190 the discovery service. It should have two parameters {api} and 

191 {apiVersion} that when filled in produce an absolute URI to the discovery 

192 document for that service. 

193 developerKey: string, key obtained from 

194 https://code.google.com/apis/console. 

195 model: googleapiclient.Model, converts to and from the wire format. 

196 requestBuilder: googleapiclient.http.HttpRequest, encapsulator for an HTTP 

197 request. 

198 credentials: oauth2client.Credentials or 

199 google.auth.credentials.Credentials, credentials to be used for 

200 authentication. 

201 cache_discovery: Boolean, whether or not to cache the discovery doc. 

202 cache: googleapiclient.discovery_cache.base.CacheBase, an optional 

203 cache object for the discovery documents. 

204 

205 Returns: 

206 A Resource object with methods for interacting with the service. 

207 """ 

208 params = { 

209 'api': serviceName, 

210 'apiVersion': version 

211 } 

212 

213 if http is None: 

214 discovery_http = build_http() 

215 else: 

216 discovery_http = http 

217 

218 for discovery_url in (discoveryServiceUrl, V2_DISCOVERY_URI,): 

219 requested_url = uritemplate.expand(discovery_url, params) 

220 

221 try: 

222 content = _retrieve_discovery_doc( 

223 requested_url, discovery_http, cache_discovery, cache, developerKey) 

224 return build_from_document(content, base=discovery_url, http=http, 

225 developerKey=developerKey, model=model, requestBuilder=requestBuilder, 

226 credentials=credentials) 

227 except HttpError as e: 

228 if e.resp.status == http_client.NOT_FOUND: 

229 continue 

230 else: 

231 raise e 

232 

233 raise UnknownApiNameOrVersion( 

234 "name: %s version: %s" % (serviceName, version))

237def _retrieve_discovery_doc(url, http, cache_discovery, cache=None, 

238 developerKey=None): 

239 """Retrieves the discovery_doc from cache or the internet. 

240 

241 Args: 

242 url: string, the URL of the discovery document. 

243 http: httplib2.Http, An instance of httplib2.Http or something that acts 

244 like it through which HTTP requests will be made. 

245 cache_discovery: Boolean, whether or not to cache the discovery doc. 

246 cache: googleapiclient.discovery_cache.base.Cache, an optional cache 

247 object for the discovery documents. 

248 

249 Returns: 

250 A unicode string representation of the discovery document. 

251 """ 

252 if cache_discovery: 

253 from . import discovery_cache 

254 from .discovery_cache import base 

255 if cache is None: 

256 cache = discovery_cache.autodetect() 

257 if cache: 

258 content = cache.get(url) 

259 if content: 

260 return content 

261 

262 actual_url = url 

263 # REMOTE_ADDR is defined by the CGI spec [RFC3875] as the environment 

264 # variable that contains the network address of the client sending the 

265 # request. If it exists then add that to the request for the discovery 

266 # document to avoid exceeding the quota on discovery requests. 

267 if 'REMOTE_ADDR' in os.environ: 

268 actual_url = _add_query_parameter(url, 'userIp', os.environ['REMOTE_ADDR']) 

269 if developerKey: 

270 actual_url = _add_query_parameter(url, 'key', developerKey) 

271 logger.info('URL being requested: GET %s', actual_url) 

272 

273 resp, content = http.request(actual_url) 

274 

275 if resp.status >= 400: 

276 raise HttpError(resp, content, uri=actual_url) 

277 

278 try: 

279 content = content.decode('utf-8') 

280 except AttributeError: 

281 pass 

282 

283 try: 

284 service = json.loads(content) 

285 except ValueError as e: 

286 logger.error('Failed to parse as JSON: ' + content) 

287 raise InvalidJsonError() 

288 if cache_discovery and cache: 

289 cache.set(url, content) 

290 return content

293@positional(1) 

294def build_from_document( 

295 service, 

296 base=None, 

297 future=None, 

298 http=None, 

299 developerKey=None, 

300 model=None, 

301 requestBuilder=HttpRequest, 

302 credentials=None): 

303 """Create a Resource for interacting with an API. 

304 

305 Same as `build()`, but constructs the Resource object from a discovery 

306 document that is it given, as opposed to retrieving one over HTTP. 

307 

308 Args: 

309 service: string or object, the JSON discovery document describing the API. 

310 The value passed in may either be the JSON string or the deserialized 

311 JSON. 

312 base: string, base URI for all HTTP requests, usually the discovery URI. 

313 This parameter is no longer used as rootUrl and servicePath are included 

314 within the discovery document. (deprecated) 

315 future: string, discovery document with future capabilities (deprecated). 

316 http: httplib2.Http, An instance of httplib2.Http or something that acts 

317 like it that HTTP requests will be made through. 

318 developerKey: string, Key for controlling API usage, generated 

319 from the API Console. 

320 model: Model class instance that serializes and de-serializes requests and 

321 responses. 

322 requestBuilder: Takes an http request and packages it up to be executed. 

323 credentials: oauth2client.Credentials or 

324 google.auth.credentials.Credentials, credentials to be used for 

325 authentication. 

326 

327 Returns: 

328 A Resource object with methods for interacting with the service. 

329 """ 

330 

331 if http is not None and credentials is not None: 

332 raise ValueError('Arguments http and credentials are mutually exclusive.') 

333 

334 if isinstance(service, six.string_types): 

335 service = json.loads(service) 

336 

337 if 'rootUrl' not in service and (isinstance(http, (HttpMock, 

338 HttpMockSequence))): 

339 logger.error("You are using HttpMock or HttpMockSequence without" + 

340 "having the service discovery doc in cache. Try calling " + 

341 "build() without mocking once first to populate the " + 

342 "cache.") 

343 raise InvalidJsonError() 

344 

345 base = urljoin(service['rootUrl'], service['servicePath']) 

346 schema = Schemas(service) 

347 

348 # If the http client is not specified, then we must construct an http client 

349 # to make requests. If the service has scopes, then we also need to setup 

350 # authentication. 

351 if http is None: 

352 # Does the service require scopes? 

353 scopes = list( 

354 service.get('auth', {}).get('oauth2', {}).get('scopes', {}).keys()) 

355 

356 # If so, then the we need to setup authentication if no developerKey is 

357 # specified. 

358 if scopes and not developerKey: 

359 # If the user didn't pass in credentials, attempt to acquire application 

360 # default credentials. 

361 if credentials is None: 

362 credentials = _auth.default_credentials() 

363 

364 # The credentials need to be scoped. 

365 credentials = _auth.with_scopes(credentials, scopes) 

366 

367 # If credentials are provided, create an authorized http instance; 

368 # otherwise, skip authentication. 

369 if credentials: 

370 http = _auth.authorized_http(credentials) 

371 

372 # If the service doesn't require scopes then there is no need for 

373 # authentication. 

374 else: 

375 http = build_http() 

376 

377 if model is None: 

378 features = service.get('features', []) 

379 model = JsonModel('dataWrapper' in features) 

380 

381 return Resource(http=http, baseUrl=base, model=model, 

382 developerKey=developerKey, requestBuilder=requestBuilder, 

383 resourceDesc=service, rootDesc=service, schema=schema)

386def _cast(value, schema_type): 

387 """Convert value to a string based on JSON Schema type. 

388 

389 See http://tools.ietf.org/html/draft-zyp-json-schema-03 for more details on 

390 JSON Schema. 

391 

392 Args: 

393 value: any, the value to convert 

394 schema_type: string, the type that value should be interpreted as 

395 

396 Returns: 

397 A string representation of 'value' based on the schema_type. 

398 """ 

399 if schema_type == 'string': 

400 if type(value) == type('') or type(value) == type(u''): 

401 return value 

402 else: 

403 return str(value) 

404 elif schema_type == 'integer': 

405 return str(int(value)) 

406 elif schema_type == 'number': 

407 return str(float(value)) 

408 elif schema_type == 'boolean': 

409 return str(bool(value)).lower() 

410 else: 

411 if type(value) == type('') or type(value) == type(u''): 

412 return value 

413 else: 

414 return str(value)

417def _media_size_to_long(maxSize): 

418 """Convert a string media size, such as 10GB or 3TB into an integer. 

419 

420 Args: 

421 maxSize: string, size as a string, such as 2MB or 7GB. 

422 

423 Returns: 

424 The size as an integer value. 

425 """ 

426 if len(maxSize) < 2: 

427 return 0 

428 units = maxSize[-2:].upper() 

429 bit_shift = _MEDIA_SIZE_BIT_SHIFTS.get(units) 

430 if bit_shift is not None: 

431 return int(maxSize[:-2]) << bit_shift 

432 else: 

433 return int(maxSize)

457def _fix_up_parameters(method_desc, root_desc, http_method, schema): 

458 """Updates parameters of an API method with values specific to this library. 

459 

460 Specifically, adds whatever global parameters are specified by the API to the 

461 parameters for the individual method. Also adds parameters which don't 

462 appear in the discovery document, but are available to all discovery based 

463 APIs (these are listed in STACK_QUERY_PARAMETERS). 

464 

465 SIDE EFFECTS: This updates the parameters dictionary object in the method 

466 description. 

467 

468 Args: 

469 method_desc: Dictionary with metadata describing an API method. Value comes 

470 from the dictionary of methods stored in the 'methods' key in the 

471 deserialized discovery document. 

472 root_desc: Dictionary; the entire original deserialized discovery document. 

473 http_method: String; the HTTP method used to call the API method described 

474 in method_desc. 

475 schema: Object, mapping of schema names to schema descriptions. 

476 

477 Returns: 

478 The updated Dictionary stored in the 'parameters' key of the method 

479 description dictionary. 

480 """ 

481 parameters = method_desc.setdefault('parameters', {}) 

482 

483 # Add in the parameters common to all methods. 

484 for name, description in six.iteritems(root_desc.get('parameters', {})): 

485 parameters[name] = description 

486 

487 # Add in undocumented query parameters. 

488 for name in STACK_QUERY_PARAMETERS: 

489 parameters[name] = STACK_QUERY_PARAMETER_DEFAULT_VALUE.copy() 

490 

491 # Add 'body' (our own reserved word) to parameters if the method supports 

492 # a request payload. 

493 if http_method in HTTP_PAYLOAD_METHODS and 'request' in method_desc: 

494 body = BODY_PARAMETER_DEFAULT_VALUE.copy() 

495 body.update(method_desc['request']) 

496 parameters['body'] = body 

497 

498 return parameters

502 """Adds 'media_body' and 'media_mime_type' parameters if supported by method.

588def _urljoin(base, url): 

589 """Custom urljoin replacement supporting : before / in url.""" 

590 # In general, it's unsafe to simply join base and url. However, for 

591 # the case of discovery documents, we know: 

592 # * base will never contain params, query, or fragment 

593 # * url will never contain a scheme or net_loc. 

594 # In general, this means we can safely join on /; we just need to 

595 # ensure we end up with precisely one / joining base and url. The 

596 # exception here is the case of media uploads, where url will be an 

597 # absolute url. 

598 if url.startswith('http://') or url.startswith('https://'): 

599 return urljoin(base, url) 

600 new_base = base if base.endswith('/') else base + '/' 

601 new_url = url[1:] if url.startswith('/') else url 

602 return new_base + new_url

654 def set_parameters(self, method_desc): 

655 """Populates maps and lists based on method description. 

656 

657 Iterates through each parameter for the method and parses the values from 

658 the parameter dictionary. 

659 

660 Args: 

661 method_desc: Dictionary with metadata describing an API method. Value 

662 comes from the dictionary of methods stored in the 'methods' key in 

663 the deserialized discovery document. 

664 """ 

665 for arg, desc in six.iteritems(method_desc.get('parameters', {})): 

666 param = key2param(arg) 

667 self.argmap[param] = arg 

668 

669 if desc.get('pattern'): 

670 self.pattern_params[param] = desc['pattern'] 

671 if desc.get('enum'): 

672 self.enum_params[param] = desc['enum'] 

673 if desc.get('required'): 

674 self.required_params.append(param) 

675 if desc.get('repeated'): 

676 self.repeated_params.append(param) 

677 if desc.get('location') == 'query': 

678 self.query_params.append(param) 

679 if desc.get('location') == 'path': 

680 self.path_params.add(param) 

681 self.param_types[param] = desc.get('type', 'string') 

682 

683 # TODO(dhermes): Determine if this is still necessary. Discovery based APIs 

684 # should have all path parameters already marked with 

685 # 'location: path'. 

686 for match in URITEMPLATE.finditer(method_desc['path']): 

687 for namematch in VARNAME.finditer(match.group(0)): 

688 name = key2param(namematch.group(0)) 

689 self.path_params.add(name) 

690 if name in self.query_params: 

691 self.query_params.remove(name)

694def createMethod(methodName, methodDesc, rootDesc, schema): 

695 """Creates a method for attaching to a Resource. 

696 

697 Args: 

698 methodName: string, name of the method to use. 

699 methodDesc: object, fragment of deserialized discovery document that 

700 describes the method. 

701 rootDesc: object, the entire deserialized discovery document. 

702 schema: object, mapping of schema names to schema descriptions. 

703 """ 

704 methodName = fix_method_name(methodName) 

705 (pathUrl, httpMethod, methodId, accept, 

706 maxSize, mediaPathUrl) = _fix_up_method_description(methodDesc, rootDesc, schema) 

707 

708 parameters = ResourceMethodParameters(methodDesc) 

709 

710 def method(self, **kwargs): 

711 # Don't bother with doc string, it will be over-written by createMethod. 

712 

713 for name in six.iterkeys(kwargs): 

714 if name not in parameters.argmap: 

715 raise TypeError('Got an unexpected keyword argument "%s"' % name) 

716 

717 # Remove args that have a value of None. 

718 keys = list(kwargs.keys()) 

719 for name in keys: 

720 if kwargs[name] is None: 

721 del kwargs[name] 

722 

723 for name in parameters.required_params: 

724 if name not in kwargs: 

725 # temporary workaround for non-paging methods incorrectly requiring 

726 # page token parameter (cf. drive.changes.watch vs. drive.changes.list) 

727 if name not in _PAGE_TOKEN_NAMES or _findPageTokenName( 

728 _methodProperties(methodDesc, schema, 'response')): 

729 raise TypeError('Missing required parameter "%s"' % name) 

730 

731 for name, regex in six.iteritems(parameters.pattern_params): 

732 if name in kwargs: 

733 if isinstance(kwargs[name], six.string_types): 

734 pvalues = [kwargs[name]] 

735 else: 

736 pvalues = kwargs[name] 

737 for pvalue in pvalues: 

738 if re.match(regex, pvalue) is None: 

739 raise TypeError( 

740 'Parameter "%s" value "%s" does not match the pattern "%s"' % 

741 (name, pvalue, regex)) 

742 

743 for name, enums in six.iteritems(parameters.enum_params): 

744 if name in kwargs: 

745 # We need to handle the case of a repeated enum 

746 # name differently, since we want to handle both 

747 # arg='value' and arg=['value1', 'value2'] 

748 if (name in parameters.repeated_params and 

749 not isinstance(kwargs[name], six.string_types)): 

750 values = kwargs[name] 

751 else: 

752 values = [kwargs[name]] 

753 for value in values: 

754 if value not in enums: 

755 raise TypeError( 

756 'Parameter "%s" value "%s" is not an allowed value in "%s"' % 

757 (name, value, str(enums))) 

758 

759 actual_query_params = {} 

760 actual_path_params = {} 

761 for key, value in six.iteritems(kwargs): 

762 to_type = parameters.param_types.get(key, 'string') 

763 # For repeated parameters we cast each member of the list. 

764 if key in parameters.repeated_params and type(value) == type([]): 

765 cast_value = [_cast(x, to_type) for x in value] 

766 else: 

767 cast_value = _cast(value, to_type) 

768 if key in parameters.query_params: 

769 actual_query_params[parameters.argmap[key]] = cast_value 

770 if key in parameters.path_params: 

771 actual_path_params[parameters.argmap[key]] = cast_value 

772 body_value = kwargs.get('body', None) 

773 media_filename = kwargs.get('media_body', None) 

774 media_mime_type = kwargs.get('media_mime_type', None) 

775 

776 if self._developerKey: 

777 actual_query_params['key'] = self._developerKey 

778 

779 model = self._model 

780 if methodName.endswith('_media'): 

781 model = MediaModel() 

782 elif 'response' not in methodDesc: 

783 model = RawModel() 

784 

785 headers = {} 

786 headers, params, query, body = model.request(headers, 

787 actual_path_params, actual_query_params, body_value) 

788 

789 expanded_url = uritemplate.expand(pathUrl, params) 

790 url = _urljoin(self._baseUrl, expanded_url + query) 

791 

792 resumable = None 

793 multipart_boundary = '' 

794 

795 if media_filename: 

796 # Ensure we end up with a valid MediaUpload object. 

797 if isinstance(media_filename, six.string_types): 

798 if media_mime_type is None: 

799 logger.warning( 

800 'media_mime_type argument not specified: trying to auto-detect for %s', 

801 media_filename) 

802 media_mime_type, _ = mimetypes.guess_type(media_filename) 

803 if media_mime_type is None: 

804 raise UnknownFileType(media_filename) 

805 if not mimeparse.best_match([media_mime_type], ','.join(accept)): 

806 raise UnacceptableMimeTypeError(media_mime_type) 

807 media_upload = MediaFileUpload(media_filename, 

808 mimetype=media_mime_type) 

809 elif isinstance(media_filename, MediaUpload): 

810 media_upload = media_filename 

811 else: 

812 raise TypeError('media_filename must be str or MediaUpload.') 

813 

814 # Check the maxSize 

815 if media_upload.size() is not None and media_upload.size() > maxSize > 0: 

816 raise MediaUploadSizeError("Media larger than: %s" % maxSize) 

817 

818 # Use the media path uri for media uploads 

819 expanded_url = uritemplate.expand(mediaPathUrl, params) 

820 url = _urljoin(self._baseUrl, expanded_url + query) 

821 if media_upload.resumable(): 

822 url = _add_query_parameter(url, 'uploadType', 'resumable') 

823 

824 if media_upload.resumable(): 

825 # This is all we need to do for resumable, if the body exists it gets 

826 # sent in the first request, otherwise an empty body is sent. 

827 resumable = media_upload 

828 else: 

829 # A non-resumable upload 

830 if body is None: 

831 # This is a simple media upload 

832 headers['content-type'] = media_upload.mimetype() 

833 body = media_upload.getbytes(0, media_upload.size()) 

834 url = _add_query_parameter(url, 'uploadType', 'media') 

835 else: 

836 # This is a multipart/related upload. 

837 msgRoot = MIMEMultipart('related') 

838 # msgRoot should not write out it's own headers 

839 setattr(msgRoot, '_write_headers', lambda self: None) 

840 

841 # attach the body as one part 

842 msg = MIMENonMultipart(*headers['content-type'].split('/')) 

843 msg.set_payload(body) 

844 msgRoot.attach(msg) 

845 

846 # attach the media as the second part 

847 msg = MIMENonMultipart(*media_upload.mimetype().split('/')) 

848 msg['Content-Transfer-Encoding'] = 'binary' 

849 

850 payload = media_upload.getbytes(0, media_upload.size()) 

851 msg.set_payload(payload) 

852 msgRoot.attach(msg) 

853 # encode the body: note that we can't use `as_string`, because 

854 # it plays games with `From ` lines. 

855 fp = BytesIO() 

856 g = _BytesGenerator(fp, mangle_from_=False) 

857 g.flatten(msgRoot, unixfrom=False) 

858 body = fp.getvalue() 

859 

860 multipart_boundary = msgRoot.get_boundary() 

861 headers['content-type'] = ('multipart/related; ' 

862 'boundary="%s"') % multipart_boundary 

863 url = _add_query_parameter(url, 'uploadType', 'multipart') 

864 

865 logger.info('URL being requested: %s %s' % (httpMethod,url)) 

866 return self._requestBuilder(self._http, 

867 model.response, 

868 url, 

869 method=httpMethod, 

870 body=body, 

871 headers=headers, 

872 methodId=methodId, 

873 resumable=resumable) 

874 

875 docs = [methodDesc.get('description', DEFAULT_METHOD_DOC), '\n\n'] 

876 if len(parameters.argmap) > 0: 

877 docs.append('Args:\n') 

878 

879 # Skip undocumented params and params common to all methods. 

880 skip_parameters = list(rootDesc.get('parameters', {}).keys()) 

881 skip_parameters.extend(STACK_QUERY_PARAMETERS) 

882 

883 all_args = list(parameters.argmap.keys()) 

884 args_ordered = [key2param(s) for s in methodDesc.get('parameterOrder', [])] 

885 

886 # Move body to the front of the line. 

887 if 'body' in all_args: 

888 args_ordered.append('body') 

889 

890 for name in all_args: 

891 if name not in args_ordered: 

892 args_ordered.append(name) 

893 

894 for arg in args_ordered: 

895 if arg in skip_parameters: 

896 continue 

897 

898 repeated = '' 

899 if arg in parameters.repeated_params: 

900 repeated = ' (repeated)' 

901 required = '' 

902 if arg in parameters.required_params: 

903 required = ' (required)' 

904 paramdesc = methodDesc['parameters'][parameters.argmap[arg]] 

905 paramdoc = paramdesc.get('description', 'A parameter') 

906 if '$ref' in paramdesc: 

907 docs.append( 

908 (' %s: object, %s%s%s\n The object takes the' 

909 ' form of:\n\n%s\n\n') % (arg, paramdoc, required, repeated, 

910 schema.prettyPrintByName(paramdesc['$ref']))) 

911 else: 

912 paramtype = paramdesc.get('type', 'string') 

913 docs.append(' %s: %s, %s%s%s\n' % (arg, paramtype, paramdoc, required, 

914 repeated)) 

915 enum = paramdesc.get('enum', []) 

916 enumDesc = paramdesc.get('enumDescriptions', []) 

917 if enum and enumDesc: 

918 docs.append(' Allowed values\n') 

919 for (name, desc) in zip(enum, enumDesc): 

920 docs.append(' %s - %s\n' % (name, desc)) 

921 if 'response' in methodDesc: 

922 if methodName.endswith('_media'): 

923 docs.append('\nReturns:\n The media object as a string.\n\n ') 

924 else: 

925 docs.append('\nReturns:\n An object of the form:\n\n ') 

926 docs.append(schema.prettyPrintSchema(methodDesc['response'])) 

927 

928 setattr(method, '__doc__', ''.join(docs)) 

929 return (methodName, method)

954 previous_response: The response from the request for the previous page. (required)
