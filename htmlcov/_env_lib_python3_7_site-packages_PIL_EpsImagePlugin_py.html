64def Ghostscript(tile, size, fp, scale=1): 

65 """Render an image using Ghostscript""" 

66 

67 # Unpack decoder tile 

68 decoder, tile, offset, data = tile[0] 

69 length, bbox = data 

70 

71 # Hack to support hi-res rendering 

72 scale = int(scale) or 1 

73 # orig_size = size 

74 # orig_bbox = bbox 

75 size = (size[0] * scale, size[1] * scale) 

76 # resolution is dependent on bbox and size 

77 res = ( 

78 72.0 * size[0] / (bbox[2] - bbox[0]), 

79 72.0 * size[1] / (bbox[3] - bbox[1]), 

80 ) 

81 

82 out_fd, outfile = tempfile.mkstemp() 

83 os.close(out_fd) 

84 

85 infile_temp = None 

86 if hasattr(fp, "name") and os.path.exists(fp.name): 

87 infile = fp.name 

88 else: 

89 in_fd, infile_temp = tempfile.mkstemp() 

90 os.close(in_fd) 

91 infile = infile_temp 

92 

93 # Ignore length and offset! 

94 # Ghostscript can read it 

95 # Copy whole file to read in Ghostscript 

96 with open(infile_temp, "wb") as f: 

97 # fetch length of fp 

98 fp.seek(0, io.SEEK_END) 

99 fsize = fp.tell() 

100 # ensure start position 

101 # go back 

102 fp.seek(0) 

103 lengthfile = fsize 

104 while lengthfile > 0: 

105 s = fp.read(min(lengthfile, 100 * 1024)) 

106 if not s: 

107 break 

108 lengthfile -= len(s) 

109 f.write(s) 

110 

111 # Build Ghostscript command 

112 command = [ 

113 "gs", 

114 "-q", # quiet mode 

115 "-g%dx%d" % size, # set output geometry (pixels) 

116 "-r%fx%f" % res, # set input DPI (dots per inch) 

117 "-dBATCH", # exit after processing 

118 "-dNOPAUSE", # don't pause between pages 

119 "-dSAFER", # safe mode 

120 "-sDEVICE=ppmraw", # ppm driver 

121 f"-sOutputFile={outfile}", # output file 

122 # adjust for image origin 

123 "-c", 

124 f"{-bbox[0]} {-bbox[1]} translate", 

125 "-f", 

126 infile, # input file 

127 # showpage (see https://bugs.ghostscript.com/show_bug.cgi?id=698272) 

128 "-c", 

129 "showpage", 

130 ] 

131 

132 if gs_windows_binary is not None: 

133 if not gs_windows_binary: 

134 raise OSError("Unable to locate Ghostscript on paths") 

135 command[0] = gs_windows_binary 

136 

137 # push data through Ghostscript 

138 try: 

139 startupinfo = None 

140 if sys.platform.startswith("win"): 

141 startupinfo = subprocess.STARTUPINFO() 

142 startupinfo.dwFlags |= subprocess.STARTF_USESHOWWINDOW 

143 subprocess.check_call(command, startupinfo=startupinfo) 

144 out_im = Image.open(outfile) 

145 out_im.load() 

146 finally: 

147 try: 

148 os.unlink(outfile) 

149 if infile_temp: 

150 os.unlink(infile_temp) 

151 except OSError: 

152 pass 

153 

154 im = out_im.im.copy() 

155 out_im.close() 

156 return im

159class PSFile: 

160 """ 

161 Wrapper for bytesio object that treats either CR or LF as end of line. 

162 """ 

163 

164 def __init__(self, fp): 

165 self.fp = fp 

166 self.char = None 

167 

168 def seek(self, offset, whence=io.SEEK_SET): 

169 self.char = None 

170 self.fp.seek(offset, whence) 

171 

172 def readline(self): 

173 s = [self.char or b""] 

174 self.char = None 

175 

176 c = self.fp.read(1) 

177 while (c not in b"\r\n") and len(c): 

178 s.append(c) 

179 c = self.fp.read(1) 

180 

181 self.char = self.fp.read(1) 

182 # line endings can be 1 or 2 of \r \n, in either order 

183 if self.char in b"\r\n": 

184 self.char = None 

185 

186 return b"".join(s).decode("latin-1")

198class EpsImageFile(ImageFile.ImageFile): 

199 """EPS File Parser for the Python Imaging Library""" 

200 

201 format = "EPS" 

202 format_description = "Encapsulated Postscript" 

203 

204 mode_map = {1: "L", 2: "LAB", 3: "RGB", 4: "CMYK"} 

205 

206 def _open(self): 

207 (length, offset) = self._find_offset(self.fp) 

208 

209 # Rewrap the open file pointer in something that will 

210 # convert line endings and decode to latin-1. 

211 fp = PSFile(self.fp) 

212 

213 # go to offset - start of "%!PS" 

214 fp.seek(offset) 

215 

216 box = None 

217 

218 self.mode = "RGB" 

219 self._size = 1, 1 # FIXME: huh? 

220 

221 # 

222 # Load EPS header 

223 

224 s_raw = fp.readline() 

225 s = s_raw.strip("\r\n") 

226 

227 while s_raw: 

228 if s: 

229 if len(s) > 255: 

230 raise SyntaxError("not an EPS file") 

231 

232 try: 

233 m = split.match(s) 

234 except re.error as e: 

235 raise SyntaxError("not an EPS file") from e 

236 

237 if m: 

238 k, v = m.group(1, 2) 

239 self.info[k] = v 

240 if k == "BoundingBox": 

241 try: 

242 # Note: The DSC spec says that BoundingBox 

243 # fields should be integers, but some drivers 

244 # put floating point values there anyway. 

245 box = [int(float(i)) for i in v.split()] 

246 self._size = box[2] - box[0], box[3] - box[1] 

247 self.tile = [ 

248 ("eps", (0, 0) + self.size, offset, (length, box)) 

249 ] 

250 except Exception: 

251 pass 

252 

253 else: 

254 m = field.match(s) 

255 if m: 

256 k = m.group(1) 

257 

258 if k == "EndComments": 

259 break 

260 if k[:8] == "PS-Adobe": 

261 self.info[k[:8]] = k[9:] 

262 else: 

263 self.info[k] = "" 

264 elif s[0] == "%": 

265 # handle non-DSC PostScript comments that some 

266 # tools mistakenly put in the Comments section 

267 pass 

268 else: 

269 raise OSError("bad EPS header") 

270 

271 s_raw = fp.readline() 

272 s = s_raw.strip("\r\n") 

273 

274 if s and s[:1] != "%": 

275 break 

276 

277 # 

278 # Scan for an "ImageData" descriptor 

279 

280 while s[:1] == "%": 

281 

282 if len(s) > 255: 

283 raise SyntaxError("not an EPS file") 

284 

285 if s[:11] == "%ImageData:": 

286 # Encoded bitmapped image. 

287 x, y, bi, mo = s[11:].split(None, 7)[:4] 

288 

289 if int(bi) != 8: 

290 break 

291 try: 

292 self.mode = self.mode_map[int(mo)] 

293 except ValueError: 

294 break 

295 

296 self._size = int(x), int(y) 

297 return 

298 

299 s = fp.readline().strip("\r\n") 

300 if not s: 

301 break 

302 

303 if not box: 

304 raise OSError("cannot determine EPS bounding box")

306 def _find_offset(self, fp): 

307 

308 s = fp.read(160) 

309 

310 if s[:4] == b"%!PS": 

311 # for HEAD without binary preview 

312 fp.seek(0, io.SEEK_END) 

313 length = fp.tell() 

314 offset = 0 

315 elif i32(s, 0) == 0xC6D3D0C5: 

316 # FIX for: Some EPS file not handled correctly / issue #302 

317 # EPS can contain binary data 

318 # or start directly with latin coding 

319 # more info see: 

320 # https://web.archive.org/web/20160528181353/http://partners.adobe.com/public/developer/en/ps/5002.EPSF_Spec.pdf 

321 offset = i32(s, 4) 

322 length = i32(s, 8) 

323 else: 

324 raise SyntaxError("not an EPS file") 

325 

326 return (length, offset)

347def _save(im, fp, filename, eps=1): 

348 """EPS Writer for the Python Imaging Library.""" 

349 

350 # 

351 # make sure image data is available 

352 im.load() 

353 

354 # 

355 # determine PostScript image mode 

356 if im.mode == "L": 

357 operator = (8, 1, "image") 

358 elif im.mode == "RGB": 

359 operator = (8, 3, "false 3 colorimage") 

360 elif im.mode == "CMYK": 

361 operator = (8, 4, "false 4 colorimage") 

362 else: 

363 raise ValueError("image mode is not supported") 

364 

365 base_fp = fp 

366 wrapped_fp = False 

367 if fp != sys.stdout: 

368 fp = io.TextIOWrapper(fp, encoding="latin-1") 

369 wrapped_fp = True 

370 

371 try: 

372 if eps: 

373 # 

374 # write EPS header 

375 fp.write("%!PS-Adobe-3.0 EPSF-3.0\n") 

376 fp.write("%%Creator: PIL 0.1 EpsEncode\n") 

377 # fp.write("%%CreationDate: %s"...) 

378 fp.write("%%%%BoundingBox: 0 0 %d %d\n" % im.size) 

379 fp.write("%%Pages: 1\n") 

380 fp.write("%%EndComments\n") 

381 fp.write("%%Page: 1 1\n") 

382 fp.write("%%ImageData: %d %d " % im.size) 

383 fp.write('%d %d 0 1 1 "%s"\n' % operator) 

384 

385 # 

386 # image header 

387 fp.write("gsave\n") 

388 fp.write("10 dict begin\n") 

389 fp.write(f"/buf {im.size[0] * operator[1]} string def\n") 

390 fp.write("%d %d scale\n" % im.size) 

391 fp.write("%d %d 8\n" % im.size) # <= bits 

392 fp.write(f"[{im.size[0]} 0 0 -{im.size[1]} 0 {im.size[1]}]\n") 

393 fp.write("{ currentfile buf readhexstring pop } bind\n") 

394 fp.write(operator[2] + "\n") 

395 if hasattr(fp, "flush"): 

396 fp.flush() 

397 

398 ImageFile._save(im, base_fp, [("eps", (0, 0) + im.size, 0, None)]) 

399 

400 fp.write("\n%%%%EndBinary\n") 

401 fp.write("grestore end\n") 

402 if hasattr(fp, "flush"): 

403 fp.flush() 

404 finally: 

405 if wrapped_fp: 

406 fp.detach()
