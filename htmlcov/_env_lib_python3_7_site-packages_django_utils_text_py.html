29def wrap(text, width): 

30 """ 

31 A word-wrap function that preserves existing line breaks. Expects that 

32 existing line breaks are posix newlines. 

33 

34 Preserve all white space except added line breaks consume the space on 

35 which they break the line. 

36 

37 Don't wrap long words, thus the output text may have lines longer than 

38 ``width``. 

39 """ 

40 def _generator(): 

41 for line in text.splitlines(True): # True keeps trailing linebreaks 

42 max_width = min((line.endswith('\n') and width + 1 or width), width) 

43 while len(line) > max_width: 

44 space = line[:max_width + 1].rfind(' ') + 1 

45 if space == 0: 

46 space = line.find(' ') + 1 

47 if space == 0: 

48 yield line 

49 line = '' 

50 break 

51 yield '%s\n' % line[:space - 1] 

52 line = line[space:] 

53 max_width = min((line.endswith('\n') and width + 1 or width), width) 

54 if line: 

55 yield line 

56 return ''.join(_generator())

81 def chars(self, num, truncate=None, html=False): 

82 """ 

83 Return the text truncated to be no longer than the specified number 

84 of characters. 

85 

86 `truncate` specifies what should be used to notify that the string has 

87 been truncated, defaulting to a translatable string of an ellipsis 

88 (...). 

89 """ 

90 self._setup() 

91 length = int(num) 

92 text = unicodedata.normalize('NFC', self._wrapped) 

93 

94 # Calculate the length to truncate to (max length - end_text length) 

95 truncate_len = length 

96 for char in self.add_truncation_text('', truncate): 

97 if not unicodedata.combining(char): 

98 truncate_len -= 1 

99 if truncate_len == 0: 

100 break 

101 if html: 

102 return self._truncate_html(length, truncate, text, truncate_len, False) 

103 return self._text_chars(length, truncate, text, truncate_len)

105 def _text_chars(self, length, truncate, text, truncate_len): 

106 """Truncate a string after a certain number of chars.""" 

107 s_len = 0 

108 end_index = None 

109 for i, char in enumerate(text): 

110 if unicodedata.combining(char): 

111 # Don't consider combining characters 

112 # as adding to the string length 

113 continue 

114 s_len += 1 

115 if end_index is None and s_len > truncate_len: 

116 end_index = i 

117 if s_len > length: 

118 # Return the truncated string 

119 return self.add_truncation_text(text[:end_index or 0], 

120 truncate) 

121 

122 # Return the original string since no truncation was necessary 

123 return text

149 def _truncate_html(self, length, truncate, text, truncate_len, words): 

150 """ 

151 Truncate HTML to a certain number of chars (not counting tags and 

152 comments), or, if words is True, then to a certain number of words. 

153 Close opened tags if they were correctly closed in the given HTML. 

154 

155 Preserve newlines in the HTML. 

156 """ 

157 if words and length <= 0: 

158 return '' 

159 

160 html4_singlets = ( 

161 'br', 'col', 'link', 'base', 'img', 

162 'param', 'area', 'hr', 'input' 

163 ) 

164 

165 # Count non-HTML chars/words and keep note of open tags 

166 pos = 0 

167 end_text_pos = 0 

168 current_len = 0 

169 open_tags = [] 

170 

171 regex = re_words if words else re_chars 

172 

173 while current_len <= length: 

174 m = regex.search(text, pos) 

175 if not m: 

176 # Checked through whole string 

177 break 

178 pos = m.end(0) 

179 if m.group(1): 

180 # It's an actual non-HTML word or char 

181 current_len += 1 

182 if current_len == truncate_len: 

183 end_text_pos = pos 

184 continue 

185 # Check for tag 

186 tag = re_tag.match(m.group(0)) 

187 if not tag or current_len >= truncate_len: 

188 # Don't worry about non tags or tags after our truncate point 

189 continue 

190 closing_tag, tagname, self_closing = tag.groups() 

191 # Element names are always case-insensitive 

192 tagname = tagname.lower() 

193 if self_closing or tagname in html4_singlets: 

194 pass 

195 elif closing_tag: 

196 # Check for match in open tags list 

197 try: 

198 i = open_tags.index(tagname) 

199 except ValueError: 

200 pass 

201 else: 

202 # SGML: An end tag closes, back to the matching start tag, 

203 # all unclosed intervening start tags with omitted end tags 

204 open_tags = open_tags[i + 1:] 

205 else: 

206 # Add it to the start of the open tags list 

207 open_tags.insert(0, tagname) 

208 

209 if current_len <= length: 

210 return text 

211 out = text[:end_text_pos] 

212 truncate_text = self.add_truncation_text('', truncate) 

213 if truncate_text: 

214 out += truncate_text 

215 # Close any tags still open 

216 for tag in open_tags: 

217 out += '</%s>' % tag 

218 # Return string 

219 return out

335def smart_split(text): 

336 r""" 

337 Generator that splits a string by spaces, leaving quoted phrases together. 

338 Supports both single and double quotes, and supports escaping quotes with 

339 backslashes. In the output, strings will keep their initial and trailing 

340 quote marks and escaped quotes will remain escaped (the results can then 

341 be further processed with unescape_string_literal()). 

342 

343 >>> list(smart_split(r'This is "a person\'s" test.')) 

344 ['This', 'is', '"a person\\\'s"', 'test.'] 

345 >>> list(smart_split(r"Another 'person\'s' test.")) 

346 ['Another', "'person\\'s'", 'test.'] 

347 >>> list(smart_split(r'A "\"funky\" style" test.')) 

348 ['A', '"\\"funky\\" style"', 'test.'] 

349 """ 

350 for bit in smart_split_re.finditer(str(text)): 

351 yield bit.group(0)
