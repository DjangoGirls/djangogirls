118def url_params_from_lookup_dict(lookups): 

119 """ 

120 Convert the type of lookups specified in a ForeignKey limit_choices_to 

121 attribute to a dictionary of query parameters 

122 """ 

123 params = {} 

124 if lookups and hasattr(lookups, 'items'): 

125 for k, v in lookups.items(): 

126 if callable(v): 

127 v = v() 

128 if isinstance(v, (tuple, list)): 

129 v = ','.join(str(x) for x in v) 

130 elif isinstance(v, bool): 

131 v = ('0', '1')[v] 

132 else: 

133 v = str(v) 

134 params[k] = v 

135 return params

151 def get_context(self, name, value, attrs): 

152 context = super().get_context(name, value, attrs) 

153 rel_to = self.rel.model 

154 if rel_to in self.admin_site._registry: 

155 # The related object is registered with the same AdminSite 

156 related_url = reverse( 

157 'admin:%s_%s_changelist' % ( 

158 rel_to._meta.app_label, 

159 rel_to._meta.model_name, 

160 ), 

161 current_app=self.admin_site.name, 

162 ) 

163 

164 params = self.url_parameters() 

165 if params: 

166 related_url += '?' + '&amp;'.join('%s=%s' % (k, v) for k, v in params.items()) 

167 context['related_url'] = mark_safe(related_url) 

168 context['link_title'] = _('Lookup') 

169 # The JavaScript code looks for this class. 

170 context['widget']['attrs'].setdefault('class', 'vForeignKeyRawIdAdminField') 

171 if context['widget']['value']: 

172 context['link_label'], context['link_url'] = self.label_and_url_for_value(value) 

173 return context

187 def label_and_url_for_value(self, value): 

188 key = self.rel.get_related_field().name 

189 try: 

190 obj = self.rel.model._default_manager.using(self.db).get(**{key: value}) 

191 except (ValueError, self.rel.model.DoesNotExist, ValidationError): 

192 return '', '' 

193 

194 try: 

195 url = reverse( 

196 '%s:%s_%s_change' % ( 

197 self.admin_site.name, 

198 obj._meta.app_label, 

199 obj._meta.object_name.lower(), 

200 ), 

201 args=(obj.pk,) 

202 ) 

203 except NoReverseMatch: 

204 url = '' # Admin not registered for target model.

206 return Truncator(obj).words(14, truncate='...'), url

421 def build_attrs(self, base_attrs, extra_attrs=None): 

422 """ 

423 Set select2's AJAX attributes. 

424 

425 Attributes can be set using the html5 data attribute. 

426 Nested attributes require a double dash as per 

427 https://select2.org/configuration/data-attributes#nested-subkey-options 

428 """ 

429 attrs = super().build_attrs(base_attrs, extra_attrs=extra_attrs) 

430 attrs.setdefault('class', '') 

431 attrs.update({ 

432 'data-ajax--cache': 'true', 

433 'data-ajax--type': 'GET', 

434 'data-ajax--url': self.get_url(), 

435 'data-theme': 'admin-autocomplete', 

436 'data-allow-clear': json.dumps(not self.is_required), 

437 'data-placeholder': '', # Allows clearing of the input. 

438 'class': attrs['class'] + (' ' if attrs['class'] else '') + 'admin-autocomplete', 

439 }) 

440 return attrs

442 def optgroups(self, name, value, attr=None): 

443 """Return selected options based on the ModelChoiceIterator.""" 

444 default = (None, [], 0) 

445 groups = [default] 

446 has_selected = False 

447 selected_choices = { 

448 str(v) for v in value 

449 if str(v) not in self.choices.field.empty_values 

450 } 

451 if not self.is_required and not self.allow_multiple_selected: 

452 default[1].append(self.create_option(name, '', '', False, 0)) 

453 choices = ( 

454 (obj.pk, self.choices.field.label_from_instance(obj)) 

455 for obj in self.choices.queryset.using(self.db).filter(pk__in=selected_choices) 

456 ) 

457 for option_value, option_label in choices: 

458 selected = ( 

459 str(option_value) in value and 

460 (has_selected is False or self.allow_multiple_selected) 

461 ) 

462 if selected is True and has_selected is False: 

463 has_selected = True 

464 index = len(default[1]) 

465 subgroup = default[1] 

466 subgroup.append(self.create_option(name, option_value, option_label, selected_choices, index)) 

467 return groups
