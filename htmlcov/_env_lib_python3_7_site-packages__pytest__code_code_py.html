702 def _truncate_recursive_traceback(self, traceback): 

703 """ 

704 Truncate the given recursive traceback trying to find the starting point 

705 of the recursion. 

706 

707 The detection is done by going through each traceback entry and finding the 

708 point in which the locals of the frame are equal to the locals of a previous frame (see ``recursionindex()``. 

709 

710 Handle the situation where the recursion process might raise an exception (for example 

711 comparing numpy arrays using equality raises a TypeError), in which case we do our best to 

712 warn the user of the error and show a limited traceback. 

713 """ 

714 try: 

715 recursionindex = traceback.recursionindex() 

716 except Exception as e: 

717 max_frames = 10 

718 extraline = ( 

719 "!!! Recursion error detected, but an error occurred locating the origin of recursion.\n" 

720 " The following exception happened when comparing locals in the stack frame:\n" 

721 " {exc_type}: {exc_msg}\n" 

722 " Displaying first and last {max_frames} stack frames out of {total}." 

723 ).format( 

724 exc_type=type(e).__name__, 

725 exc_msg=safe_str(e), 

726 max_frames=max_frames, 

727 total=len(traceback), 

728 ) 

729 traceback = traceback[:max_frames] + traceback[-max_frames:] 

730 else: 

731 if recursionindex is not None: 

732 extraline = "!!! Recursion detected (same locals & position)" 

733 traceback = traceback[: recursionindex + 1] 

734 else: 

735 extraline = None 

736 

737 return traceback, extraline
