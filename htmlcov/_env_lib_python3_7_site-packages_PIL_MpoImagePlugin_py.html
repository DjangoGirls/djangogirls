44 def _open(self): 

45 self.fp.seek(0) # prep the fp in order to pass the JPEG test 

46 JpegImagePlugin.JpegImageFile._open(self) 

47 self._after_jpeg_open()

49 def _after_jpeg_open(self, mpheader=None): 

50 self.mpinfo = mpheader if mpheader is not None else self._getmp() 

51 self.n_frames = self.mpinfo[0xB001] 

52 self.__mpoffsets = [ 

53 mpent["DataOffset"] + self.info["mpoffset"] for mpent in self.mpinfo[0xB002] 

54 ] 

55 self.__mpoffsets[0] = 0 

56 # Note that the following assertion will only be invalid if something 

57 # gets broken within JpegImagePlugin. 

58 assert self.n_frames == len(self.__mpoffsets) 

59 del self.info["mpoffset"] # no longer needed 

60 self.is_animated = self.n_frames > 1 

61 self.__fp = self.fp # FIXME: hack 

62 self.__fp.seek(self.__mpoffsets[0]) # get ready to read first frame 

63 self.__frame = 0 

64 self.offset = 0 

65 # for now we can only handle reading and individual frame extraction 

66 self.readonly = 1 

67 

68 def load_seek(self, pos): 

69 self.__fp.seek(pos) 

70 

71 def seek(self, frame): 

72 if not self._seek_check(frame): 

73 return 

74 self.fp = self.__fp 

75 self.offset = self.__mpoffsets[frame] 

76 

77 self.fp.seek(self.offset + 2) # skip SOI marker 

78 segment = self.fp.read(2) 

79 if not segment: 

80 raise ValueError("No data found for frame") 

81 if i16(segment) == 0xFFE1: # APP1 

82 n = i16(self.fp.read(2)) - 2 

83 self.info["exif"] = ImageFile._safe_read(self.fp, n) 

84 

85 mptype = self.mpinfo[0xB002][frame]["Attribute"]["MPType"] 

86 if mptype.startswith("Large Thumbnail"): 

87 exif = self.getexif().get_ifd(0x8769) 

88 if 40962 in exif and 40963 in exif: 

89 self._size = (exif[40962], exif[40963]) 

90 elif "exif" in self.info: 

91 del self.info["exif"] 

92 

93 self.tile = [("jpeg", (0, 0) + self.size, self.offset, (self.mode, ""))] 

94 self.__frame = frame 

95 

96 def tell(self): 

97 return self.__frame 

98 

99 def _close__fp(self): 

100 try: 

101 if self.__fp != self.fp: 

102 self.__fp.close() 

103 except AttributeError: 

104 pass 

105 finally: 

106 self.__fp = None

109 def adopt(jpeg_instance, mpheader=None): 

110 """ 

111 Transform the instance of JpegImageFile into 

112 an instance of MpoImageFile. 

113 After the call, the JpegImageFile is extended 

114 to be an MpoImageFile. 

115 

116 This is essentially useful when opening a JPEG 

117 file that reveals itself as an MPO, to avoid 

118 double call to _open. 

119 """ 

120 jpeg_instance.__class__ = MpoImageFile 

121 jpeg_instance._after_jpeg_open(mpheader) 

122 return jpeg_instance
