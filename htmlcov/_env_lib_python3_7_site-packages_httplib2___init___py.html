217 """Return a filename suitable for the cache. 

218 Strips dangerous and common characters to create a filename we 

219 can use to store the cache in. 

220 """ 

221 if isinstance(filename, bytes): 

222 filename_bytes = filename 

223 filename = filename.decode("utf-8") 

224 else: 

225 filename_bytes = filename.encode("utf-8")

277def _entry_disposition(response_headers, request_headers): 

278 """Determine freshness from the Date, Expires and Cache-Control headers. 

279 

280 We don't handle the following: 

281 

282 1. Cache-Control: max-stale 

283 2. Age: headers are not used in the calculations. 

284 

285 Not that this algorithm is simpler than you might think 

286 because we are operating as a private (non-shared) cache. 

287 This lets us ignore 's-maxage'. We can also ignore 

288 'proxy-invalidate' since we aren't a proxy. 

289 We will never return a stale document as 

290 fresh as a design decision, and thus the non-implementation 

291 of 'max-stale'. This also lets us safely ignore 'must-revalidate' 

292 since we operate as if every server has sent 'must-revalidate'. 

293 Since we are private we get to ignore both 'public' and 

294 'private' parameters. We also ignore 'no-transform' since 

295 we don't do any transformations. 

296 The 'no-store' parameter is handled at a higher level. 

297 So the only Cache-Control parameters we look at are: 

298 

299 no-cache 

300 only-if-cached 

301 max-age 

302 min-fresh 

303 """ 

304 

305 retval = "STALE" 

306 cc = _parse_cache_control(request_headers) 

307 cc_response = _parse_cache_control(response_headers) 

308 

309 if "pragma" in request_headers and request_headers["pragma"].lower().find("no-cache") != -1: 

310 retval = "TRANSPARENT" 

311 if "cache-control" not in request_headers: 

312 request_headers["cache-control"] = "no-cache" 

313 elif "no-cache" in cc: 

314 retval = "TRANSPARENT" 

315 elif "no-cache" in cc_response: 

316 retval = "STALE" 

317 elif "only-if-cached" in cc: 

318 retval = "FRESH" 

319 elif "date" in response_headers: 

320 date = calendar.timegm(email.utils.parsedate_tz(response_headers["date"])) 

321 now = time.time() 

322 current_age = max(0, now - date) 

323 if "max-age" in cc_response: 

324 try: 

325 freshness_lifetime = int(cc_response["max-age"]) 

326 except ValueError: 

327 freshness_lifetime = 0 

328 elif "expires" in response_headers: 

329 expires = email.utils.parsedate_tz(response_headers["expires"]) 

330 if None == expires: 

331 freshness_lifetime = 0 

332 else: 

333 freshness_lifetime = max(0, calendar.timegm(expires) - date) 

334 else: 

335 freshness_lifetime = 0 

336 if "max-age" in cc: 

337 try: 

338 freshness_lifetime = int(cc["max-age"]) 

339 except ValueError: 

340 freshness_lifetime = 0 

341 if "min-fresh" in cc: 

342 try: 

343 min_fresh = int(cc["min-fresh"]) 

344 except ValueError: 

345 min_fresh = 0 

346 current_age += min_fresh 

347 if freshness_lifetime > current_age: 

348 retval = "FRESH" 

349 return retval
