56def sentence(): 

57 """ 

58 Return a randomly generated sentence of lorem ipsum text. 

59 

60 The first word is capitalized, and the sentence ends in either a period or 

61 question mark. Commas are added at random. 

62 """ 

63 # Determine the number of comma-separated sections and number of words in 

64 # each section for this sentence. 

65 sections = [' '.join(random.sample(WORDS, random.randint(3, 12))) for i in range(random.randint(1, 5))] 

66 s = ', '.join(sections) 

67 # Convert to sentence case and add end punctuation. 

68 return '%s%s%s' % (s[0].upper(), s[1:], random.choice('?.'))

71def paragraph(): 

72 """ 

73 Return a randomly generated paragraph of lorem ipsum text. 

74 

75 The paragraph consists of between 1 and 4 sentences, inclusive. 

76 """ 

77 return ' '.join(sentence() for i in range(random.randint(1, 4)))

80def paragraphs(count, common=True): 

81 """ 

82 Return a list of paragraphs as returned by paragraph(). 

83 

84 If `common` is True, then the first paragraph will be the standard 

85 'lorem ipsum' paragraph. Otherwise, the first paragraph will be random 

86 Latin text. Either way, subsequent paragraphs will be random Latin text. 

87 """ 

88 paras = [] 

89 for i in range(count): 

90 if common and i == 0: 

91 paras.append(COMMON_P) 

92 else: 

93 paras.append(paragraph()) 

94 return paras

97def words(count, common=True): 

98 """ 

99 Return a string of `count` lorem ipsum words separated by a single space. 

100 

101 If `common` is True, then the first 19 words will be the standard 

102 'lorem ipsum' words. Otherwise, all words will be selected randomly. 

103 """ 

104 word_list = list(COMMON_WORDS) if common else [] 

105 c = len(word_list) 

106 if count > c: 

107 count -= c 

108 while count > 0: 

109 c = min(count, len(WORDS)) 

110 count -= c 

111 word_list += random.sample(WORDS, c) 

112 else: 

113 word_list = word_list[:count] 

114 return ' '.join(word_list)
