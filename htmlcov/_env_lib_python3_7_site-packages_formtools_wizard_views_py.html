22def normalize_name(name):

25

28 >>> normalize_name('FourFiveSix') 

29 'four_five_six' 

30 

31 """ 

32 new = re.sub('(((?<=[a-z])[A-Z])|([A-Z](?![A-Z]|$)))', '_\\1', name) 

33 return new.lower().strip('_')
