677 def validate_unique(self): 

678 # Collect unique_checks and date_checks to run from all the forms. 

679 all_unique_checks = set() 

680 all_date_checks = set() 

681 forms_to_delete = self.deleted_forms 

682 valid_forms = [form for form in self.forms if form.is_valid() and form not in forms_to_delete] 

683 for form in valid_forms: 

684 exclude = form._get_validation_exclusions() 

685 unique_checks, date_checks = form.instance._get_unique_checks(exclude=exclude) 

686 all_unique_checks.update(unique_checks) 

687 all_date_checks.update(date_checks) 

688 

689 errors = [] 

690 # Do each of the unique checks (unique and unique_together) 

691 for uclass, unique_check in all_unique_checks: 

692 seen_data = set() 

693 for form in valid_forms: 

694 # Get the data for the set of fields that must be unique among the forms. 

695 row_data = ( 

696 field if field in self.unique_fields else form.cleaned_data[field] 

697 for field in unique_check if field in form.cleaned_data 

698 ) 

699 # Reduce Model instances to their primary key values 

700 row_data = tuple(d._get_pk_val() if hasattr(d, '_get_pk_val') else d 

701 for d in row_data) 

702 if row_data and None not in row_data: 

703 # if we've already seen it then we have a uniqueness failure 

704 if row_data in seen_data: 

705 # poke error messages into the right places and mark 

706 # the form as invalid 

707 errors.append(self.get_unique_error_message(unique_check)) 

708 form._errors[NON_FIELD_ERRORS] = self.error_class([self.get_form_error()]) 

709 # remove the data from the cleaned_data dict since it was invalid 

710 for field in unique_check: 

711 if field in form.cleaned_data: 

712 del form.cleaned_data[field] 

713 # mark the data as seen 

714 seen_data.add(row_data) 

715 # iterate over each of the date checks now 

716 for date_check in all_date_checks: 

717 seen_data = set() 

718 uclass, lookup, field, unique_for = date_check 

719 for form in valid_forms: 

720 # see if we have data for both fields 

721 if (form.cleaned_data and form.cleaned_data[field] is not None and 

722 form.cleaned_data[unique_for] is not None): 

723 # if it's a date lookup we need to get the data for all the fields 

724 if lookup == 'date': 

725 date = form.cleaned_data[unique_for] 

726 date_data = (date.year, date.month, date.day) 

727 # otherwise it's just the attribute on the date/datetime 

728 # object 

729 else: 

730 date_data = (getattr(form.cleaned_data[unique_for], lookup),) 

731 data = (form.cleaned_data[field],) + date_data 

732 # if we've already seen it then we have a uniqueness failure 

733 if data in seen_data: 

734 # poke error messages into the right places and mark 

735 # the form as invalid 

736 errors.append(self.get_date_error_message(date_check)) 

737 form._errors[NON_FIELD_ERRORS] = self.error_class([self.get_form_error()]) 

738 # remove the data from the cleaned_data dict since it was invalid 

739 del form.cleaned_data[field] 

740 # mark the data as seen 

741 seen_data.add(data) 

742 

743 if errors: 

744 raise ValidationError(errors)

829 # as it could be an auto-generated default which isn't actually 

830 # in the database.

832 else: 

833 try: 

834 if index is not None: 

835 pk_value = self.get_queryset()[index].pk 

836 else: 

837 pk_value = None 

838 except IndexError: 

839 pk_value = None

988def _get_foreign_key(parent_model, model, fk_name=None, can_fail=False): 

989 """ 

990 Find and return the ForeignKey from model to parent if there is one 

991 (return None if can_fail is True and no such field exists). If fk_name is 

992 provided, assume it is the name of the ForeignKey field. Unless can_fail is 

993 True, raise an exception if there isn't a ForeignKey from model to 

994 parent_model. 

995 """ 

996 # avoid circular import 

997 from django.db.models import ForeignKey 

998 opts = model._meta 

999 if fk_name: 

1000 fks_to_parent = [f for f in opts.fields if f.name == fk_name] 

1001 if len(fks_to_parent) == 1: 

1002 fk = fks_to_parent[0] 

1003 if not isinstance(fk, ForeignKey) or \ 

1004 (fk.remote_field.model != parent_model and 

1005 fk.remote_field.model not in parent_model._meta.get_parent_list()): 

1006 raise ValueError( 

1007 "fk_name '%s' is not a ForeignKey to '%s'." % (fk_name, parent_model._meta.label) 

1008 ) 

1009 elif len(fks_to_parent) == 0: 

1010 raise ValueError( 

1011 "'%s' has no field named '%s'." % (model._meta.label, fk_name) 

1012 ) 

1013 else: 

1014 # Try to discover what the ForeignKey from model to parent_model is 

1015 fks_to_parent = [ 

1016 f for f in opts.fields 

1017 if isinstance(f, ForeignKey) and ( 

1018 f.remote_field.model == parent_model or 

1019 f.remote_field.model in parent_model._meta.get_parent_list() 

1020 ) 

1021 ] 

1022 if len(fks_to_parent) == 1: 

1023 fk = fks_to_parent[0] 

1024 elif len(fks_to_parent) == 0: 

1025 if can_fail: 

1026 return 

1027 raise ValueError( 

1028 "'%s' has no ForeignKey to '%s'." % ( 

1029 model._meta.label, 

1030 parent_model._meta.label, 

1031 ) 

1032 ) 

1033 else: 

1034 raise ValueError( 

1035 "'%s' has more than one ForeignKey to '%s'." % ( 

1036 model._meta.label, 

1037 parent_model._meta.label, 

1038 ) 

1039 ) 

1040 return fk
