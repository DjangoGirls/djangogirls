8class PasswordResetTokenGenerator: 

9 """ 

10 Strategy object used to generate and check tokens for the password 

11 reset mechanism. 

12 """ 

13 key_salt = "django.contrib.auth.tokens.PasswordResetTokenGenerator" 

14 secret = settings.SECRET_KEY 

15 

16 def make_token(self, user): 

17 """ 

18 Return a token that can be used once to do a password reset 

19 for the given user. 

20 """ 

21 return self._make_token_with_timestamp(user, self._num_days(self._today())) 

22 

23 def check_token(self, user, token): 

24 """ 

25 Check that a password reset token is correct for a given user. 

26 """ 

27 if not (user and token): 

28 return False 

29 # Parse the token 

30 try: 

31 ts_b36, hash = token.split("-") 

32 except ValueError: 

33 return False 

34 

35 try: 

36 ts = base36_to_int(ts_b36) 

37 except ValueError: 

38 return False 

39 

40 # Check that the timestamp/uid has not been tampered with 

41 if not constant_time_compare(self._make_token_with_timestamp(user, ts), token): 

42 return False 

43 

44 # Check the timestamp is within limit 

45 if (self._num_days(self._today()) - ts) > settings.PASSWORD_RESET_TIMEOUT_DAYS: 

46 return False 

47 

48 return True 

49 

50 def _make_token_with_timestamp(self, user, timestamp): 

51 # timestamp is number of days since 2001-1-1. Converted to 

52 # base 36, this gives us a 3 digit string until about 2121 

53 ts_b36 = int_to_base36(timestamp) 

54 

55 # By hashing on the internal state of the user and using state 

56 # that is sure to change (the password salt will change as soon as 

57 # the password is set, at least for current Django auth, and 

58 # last_login will also change), we produce a hash that will be 

59 # invalid as soon as it is used. 

60 # We limit the hash to 20 chars to keep URL short 

61 

62 hash = salted_hmac( 

63 self.key_salt, 

64 self._make_hash_value(user, timestamp), 

65 secret=self.secret, 

66 ).hexdigest()[::2] 

67 return "%s-%s" % (ts_b36, hash)

69 def _make_hash_value(self, user, timestamp): 

70 # Ensure results are consistent across DB backends 

71 login_timestamp = '' if user.last_login is None else user.last_login.replace(microsecond=0, tzinfo=None) 

72 return str(user.pk) + user.password + str(login_timestamp) + str(timestamp)
