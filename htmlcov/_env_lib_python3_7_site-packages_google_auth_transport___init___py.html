15"""Transport - HTTP client library support. 

16 

17:mod:`google.auth` is designed to work with various HTTP client libraries such 

18as urllib3 and requests. In order to work across these libraries with different 

19interfaces some abstraction is needed. 

20 

21This module provides two interfaces that are implemented by transport adapters 

22to support HTTP libraries. :class:`Request` defines the interface expected by 

23:mod:`google.auth` to make requests. :class:`Response` defines the interface 

24for the return value of :class:`Request`. 

25""" 

26 

27import abc 

28import http.client 

29 

30DEFAULT_REFRESH_STATUS_CODES = (http.client.UNAUTHORIZED,) 

31"""Sequence[int]: Which HTTP status code indicate that credentials should be 

32refreshed and a request should be retried. 

33""" 

34 

35DEFAULT_MAX_REFRESH_ATTEMPTS = 2 

36"""int: How many times to refresh the credentials and retry a request.""" 

37 

38 

39class Response(object, metaclass=abc.ABCMeta): 

40 """HTTP Response data.""" 

41 

42 @abc.abstractproperty 

43 def status(self): 

44 """int: The HTTP status code.""" 

45 raise NotImplementedError("status must be implemented.") 

46 

47 @abc.abstractproperty 

48 def headers(self): 

49 """Mapping[str, str]: The HTTP response headers.""" 

50 raise NotImplementedError("headers must be implemented.") 

51 

52 @abc.abstractproperty 

53 def data(self): 

54 """bytes: The response body.""" 

55 raise NotImplementedError("data must be implemented.") 

56 

57 

58class Request(object, metaclass=abc.ABCMeta): 

59 """Interface for a callable that makes HTTP requests. 

60 

61 Specific transport implementations should provide an implementation of 

62 this that adapts their specific request / response API. 

63 

64 .. automethod:: __call__ 

65 """ 

66 

67 @abc.abstractmethod 

68 def __call__( 

69 self, url, method="GET", body=None, headers=None, timeout=None, **kwargs 

70 ): 

71 """Make an HTTP request. 

72 

73 Args: 

74 url (str): The URI to be requested. 

75 method (str): The HTTP method to use for the request. Defaults 

76 to 'GET'. 

77 body (bytes): The payload / body in HTTP request. 

78 headers (Mapping[str, str]): Request headers. 

79 timeout (Optional[int]): The number of seconds to wait for a 

80 response from the server. If not specified or if None, the 

81 transport-specific default timeout will be used. 

82 kwargs: Additionally arguments passed on to the transport's 

83 request method. 

84 

85 Returns: 

86 Response: The HTTP response. 

87 

88 Raises: 

89 google.auth.exceptions.TransportError: If any exception occurred. 

90 """ 

91 # pylint: disable=redundant-returns-doc, missing-raises-doc 

92 # (pylint doesn't play well with abstract docstrings.) 

93 raise NotImplementedError("__call__ must be implemented.")
