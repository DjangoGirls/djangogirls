41 def mktemp(self, basename, numbered=True): 

42 """makes a temporary directory managed by the factory""" 

43 if not numbered: 

44 p = self.getbasetemp().joinpath(basename) 

45 p.mkdir() 

46 else: 

47 p = make_numbered_dir(root=self.getbasetemp(), prefix=basename) 

48 self._trace("mktemp", p) 

49 return p 

50 

51 def getbasetemp(self): 

52 """ return base temporary directory. """ 

53 if self._basetemp is None: 

54 if self._given_basetemp is not None: 

55 basetemp = Path(self._given_basetemp) 

56 ensure_reset_dir(basetemp) 

57 else: 

58 from_env = os.environ.get("PYTEST_DEBUG_TEMPROOT") 

59 temproot = Path(from_env or tempfile.gettempdir()) 

60 user = get_user() or "unknown" 

61 # use a sub-directory in the temproot to speed-up 

62 # make_numbered_dir() call 

63 rootdir = temproot.joinpath("pytest-of-{}".format(user)) 

64 rootdir.mkdir(exist_ok=True) 

65 basetemp = make_numbered_dir_with_cleanup( 

66 prefix="pytest-", root=rootdir, keep=3, lock_timeout=LOCK_TIMEOUT 

67 ) 

68 assert basetemp is not None 

69 self._basetemp = t = basetemp 

70 self._trace("new basetemp", t) 

71 return t 

72 else: 

73 return self._basetemp

122def pytest_configure(config): 

123 """Create a TempdirFactory and attach it to the config object. 

124 

125 This is to comply with existing plugins which expect the handler to be 

126 available at pytest_configure time, but ideally should be moved entirely 

127 to the tmpdir_factory session fixture. 

128 """ 

129 mp = MonkeyPatch() 

130 tmppath_handler = TempPathFactory.from_config(config) 

131 t = TempdirFactory(tmppath_handler) 

132 config._cleanup.append(mp.undo) 

133 mp.setattr(config, "_tmp_path_factory", tmppath_handler, raising=False) 

134 mp.setattr(config, "_tmpdirhandler", t, raising=False) 

135 mp.setattr(pytest, "ensuretemp", t.ensuretemp, raising=False)
