77class Concat(Func): 

78 """ 

79 Concatenate text fields together. Backends that result in an entire 

80 null expression when any arguments are null will wrap each argument in 

81 coalesce functions to ensure a non-null result. 

82 """ 

83 function = None 

84 template = "%(expressions)s" 

85 

86 def __init__(self, *expressions, **extra): 

87 if len(expressions) < 2: 

88 raise ValueError('Concat must take at least two expressions') 

89 paired = self._paired(expressions) 

90 super().__init__(paired, **extra) 

91 

92 def _paired(self, expressions): 

93 # wrap pairs of expressions in successive concat functions 

94 # exp = [a, b, c, d] 

95 # -> ConcatPair(a, ConcatPair(b, ConcatPair(c, d)))) 

96 if len(expressions) == 2: 

97 return ConcatPair(*expressions) 

98 return ConcatPair(expressions[0], self._paired(expressions[1:]))
