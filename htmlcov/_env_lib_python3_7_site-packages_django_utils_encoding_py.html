122def iri_to_uri(iri): 

123 """ 

124 Convert an Internationalized Resource Identifier (IRI) portion to a URI 

125 portion that is suitable for inclusion in a URL. 

126 

127 This is the algorithm from section 3.1 of RFC 3987, slightly simplified 

128 since the input is assumed to be a string rather than an arbitrary byte 

129 stream.

132 b'/I \xe2\x99\xa5 Django/') and return a string containing the encoded 

133 result with ASCII chars only (e.g. '/I%20%E2%99%A5%20Django/'). 

134 """

136 # "unreserved" characters specified in sections 2.2 and 2.3 of RFC 3986:

143 # but the ~ safe.

145 # end of section 3.1 of RFC 3987 specifically mentions that % must not be

172def uri_to_iri(uri): 

173 """ 

174 Convert a Uniform Resource Identifier(URI) into an Internationalized 

175 Resource Identifier(IRI). 

176 

177 This is the algorithm from section 3.2 of RFC 3987, excluding step 4. 

178 

179 Take an URI in ASCII bytes (e.g. '/I%20%E2%99%A5%20Django/') and return 

180 a string containing the encoded result (e.g. '/I%20♥%20Django/'). 

181 """

183 return uri

186 # second block, decode the first 2 bytes if they represent a hex code to 

187 # decode. The rest of the block is the part after '%AB', not containing 

188 # any '%'. Add that to the output without further processing.

190 if len(bits) == 1: 

191 iri = uri 

192 else: 

193 parts = [bits[0]] 

194 append = parts.append 

195 hextobyte = _hextobyte

208def escape_uri_path(path): 

209 """ 

210 Escape the unsafe characters from the path portion of a Uniform Resource 

211 Identifier (URI). 

212 """ 

213 # These are the "reserved" and "unreserved" characters specified in 

214 # sections 2.2 and 2.3 of RFC 2396: 

215 # reserved = ";" | "/" | "?" | ":" | "@" | "&" | "=" | "+" | "$" | "," 

216 # unreserved = alphanum | mark 

217 # mark = "-" | "_" | "." | "!" | "~" | "*" | "'" | "(" | ")" 

218 # The list of safe characters here is constructed subtracting ";", "=", 

219 # and "?" according to section 3.3 of RFC 2396. 

220 # The reason for not subtracting and escaping "/" is that we are escaping 

221 # the entire path, not a path segment. 

222 return quote(path, safe="/:@&+$,-_.!~*'()")
