797class HstoreAdapter(object): 

798 """Adapt a Python dict to the hstore syntax.""" 

799 def __init__(self, wrapped): 

800 self.wrapped = wrapped 

801 

802 def prepare(self, conn): 

803 self.conn = conn 

804

810 """Use the operators available in PG pre-9.0.""" 

811 if not self.wrapped: 

812 return b"''::hstore" 

813 

814 adapt = _ext.adapt 

815 rv = [] 

816 for k, v in self.wrapped.items(): 

817 k = adapt(k) 

818 k.prepare(self.conn) 

819 k = k.getquoted() 

820

823 v.prepare(self.conn) 

824 v = v.getquoted() 

825 else:

829 rv.append(b"(" + k + b" => " + v + b")")

833 def _getquoted_9(self): 

834 """Use the hstore(text[], text[]) function.""" 

835 if not self.wrapped: 

836 return b"''::hstore"

839 k.prepare(self.conn) 

840 v = _ext.adapt(list(self.wrapped.values())) 

841 v.prepare(self.conn)

844 getquoted = _getquoted_9

846 _re_hstore = _re.compile(r""" 

847 # hstore key: 

848 # a string of normal or escaped chars 

849 "((?: [^"\\] | \\. )*)" 

850 \s*=>\s* # hstore value 

851 (?: 

852 NULL # the value can be null - not catched 

853 # or a quoted string like the key 

854 | "((?: [^"\\] | \\. )*)" 

855 ) 

856 (?:\s*,\s*|$) # pairs separated by comma or end of string. 

857 """, _re.VERBOSE)

859 @classmethod 

860 def parse(self, s, cur, _bsdec=_re.compile(r"\\(.)")): 

861 """Parse an hstore representation in a Python string. 

862 

863 The hstore is represented as something like:: 

864 

865 "a"=>"1", "b"=>"2" 

866 

867 with backslash-escaped strings. 

868 """ 

869 if s is None: 

870 return None 

871 

872 rv = {} 

873 start = 0 

874 for m in self._re_hstore.finditer(s): 

875 if m is None or m.start() != start: 

876 raise psycopg2.InterfaceError( 

877 "error parsing hstore pair at char %d" % start) 

878 k = _bsdec.sub(r'\1', m.group(1)) 

879 v = m.group(2) 

880 if v is not None: 

881 v = _bsdec.sub(r'\1', v) 

882 

883 rv[k] = v 

884 start = m.end() 

885 

886 if start < len(s): 

887 raise psycopg2.InterfaceError( 

888 "error parsing hstore: unparsed data after char %d" % start) 

889 

890 return rv

999class CompositeCaster(object): 

1000 """Helps conversion of a PostgreSQL composite type into a Python object. 

1001 

1002 The class is usually created by the `register_composite()` function. 

1003 You may want to create and register manually instances of the class if 

1004 querying the database at registration time is not desirable (such as when 

1005 using an :ref:`asynchronous connections <async-support>`). 

1006 

1007 """ 

1008 def __init__(self, name, oid, attrs, array_oid=None, schema=None): 

1009 self.name = name 

1010 self.schema = schema 

1011 self.oid = oid 

1012 self.array_oid = array_oid 

1013 

1014 self.attnames = [a[0] for a in attrs] 

1015 self.atttypes = [a[1] for a in attrs] 

1016 self._create_type(name, self.attnames) 

1017 self.typecaster = _ext.new_type((oid,), name, self.parse) 

1018 if array_oid: 

1019 self.array_typecaster = _ext.new_array_type( 

1020 (array_oid,), "%sARRAY" % name, self.typecaster) 

1021 else: 

1022 self.array_typecaster = None 

1023 

1024 def parse(self, s, curs): 

1025 if s is None: 

1026 return None 

1027 

1028 tokens = self.tokenize(s) 

1029 if len(tokens) != len(self.atttypes): 

1030 raise psycopg2.DataError( 

1031 "expecting %d components for the type %s, %d found instead" % 

1032 (len(self.atttypes), self.name, len(tokens))) 

1033 

1034 values = [curs.cast(oid, token) 

1035 for oid, token in zip(self.atttypes, tokens)] 

1036 

1037 return self.make(values) 

1038 

1039 def make(self, values): 

1040 """Return a new Python object representing the data being casted. 

1041 

1042 *values* is the list of attributes, already casted into their Python 

1043 representation. 

1044 

1045 You can subclass this method to :ref:`customize the composite cast 

1046 <custom-composite>`. 

1047 """ 

1048 

1049 return self._ctor(values) 

1050 

1051 _re_tokenize = _re.compile(r""" 

1052 \(? ([,)]) # an empty token, representing NULL 

1053| \(? " ((?: [^"] | "")*) " [,)] # or a quoted string 

1054| \(? ([^",)]+) [,)] # or an unquoted string 

1055 """, _re.VERBOSE) 

1056 

1057 _re_undouble = _re.compile(r'(["\\])\1') 

1058 

1059 @classmethod 

1060 def tokenize(self, s): 

1061 rv = [] 

1062 for m in self._re_tokenize.finditer(s): 

1063 if m is None: 

1064 raise psycopg2.InterfaceError("can't parse type: %r" % s) 

1065 if m.group(1) is not None: 

1066 rv.append(None) 

1067 elif m.group(2) is not None: 

1068 rv.append(self._re_undouble.sub(r"\1", m.group(2))) 

1069 else: 

1070 rv.append(m.group(3)) 

1071 

1072 return rv

1075 self.type = namedtuple(name, attnames) 

1076 self._ctor = self.type._make

1087 conn_status = conn.status

1118 raise psycopg2.ProgrammingError(

1120 

1121 type_oid = recs[0][0]

1159

1201def execute_values(cur, sql, argslist, template=None, page_size=100, fetch=False): 

1202 '''Execute a statement using :sql:`VALUES` with a sequence of parameters. 

1203 

1204 :param cur: the cursor to use to execute the query. 

1205 

1206 :param sql: the query to execute. It must contain a single ``%s`` 

1207 placeholder, which will be replaced by a `VALUES list`__. 

1208 Example: ``"INSERT INTO mytable (id, f1, f2) VALUES %s"``. 

1209 

1210 :param argslist: sequence of sequences or dictionaries with the arguments 

1211 to send to the query. The type and content must be consistent with 

1212 *template*. 

1213 

1214 :param template: the snippet to merge to every item in *argslist* to 

1215 compose the query. 

1216 

1217 - If the *argslist* items are sequences it should contain positional 

1218 placeholders (e.g. ``"(%s, %s, %s)"``, or ``"(%s, %s, 42)``" if there 

1219 are constants value...). 

1220 

1221 - If the *argslist* items are mappings it should contain named 

1222 placeholders (e.g. ``"(%(id)s, %(f1)s, 42)"``). 

1223 

1224 If not specified, assume the arguments are sequence and use a simple 

1225 positional template (i.e. ``(%s, %s, ...)``), with the number of 

1226 placeholders sniffed by the first element in *argslist*. 

1227 

1228 :param page_size: maximum number of *argslist* items to include in every 

1229 statement. If there are more items the function will execute more than 

1230 one statement. 

1231 

1232 :param fetch: if `!True` return the query results into a list (like in a 

1233 `~cursor.fetchall()`). Useful for queries with :sql:`RETURNING` 

1234 clause. 

1235 

1236 .. __: https://www.postgresql.org/docs/current/static/queries-values.html 

1237 

1238 After the execution of the function the `cursor.rowcount` property will 

1239 **not** contain a total result. 

1240 

1241 While :sql:`INSERT` is an obvious candidate for this function it is 

1242 possible to use it with other statements, for example:: 

1243 

1244 >>> cur.execute( 

1245 ... "create table test (id int primary key, v1 int, v2 int)") 

1246 

1247 >>> execute_values(cur, 

1248 ... "INSERT INTO test (id, v1, v2) VALUES %s", 

1249 ... [(1, 2, 3), (4, 5, 6), (7, 8, 9)]) 

1250 

1251 >>> execute_values(cur, 

1252 ... """UPDATE test SET v1 = data.v1 FROM (VALUES %s) AS data (id, v1) 

1253 ... WHERE test.id = data.id""", 

1254 ... [(1, 20), (4, 50)]) 

1255 

1256 >>> cur.execute("select * from test order by id") 

1257 >>> cur.fetchall() 

1258 [(1, 20, 3), (4, 50, 6), (7, 8, 9)]) 

1259 

1260 ''' 

1261 from psycopg2.sql import Composable 

1262 if isinstance(sql, Composable): 

1263 sql = sql.as_string(cur) 

1264 

1265 # we can't just use sql % vals because vals is bytes: if sql is bytes 

1266 # there will be some decoding error because of stupid codec used, and Py3 

1267 # doesn't implement % on bytes. 

1268 if not isinstance(sql, bytes): 

1269 sql = sql.encode(_ext.encodings[cur.connection.encoding]) 

1270 pre, post = _split_sql(sql) 

1271 

1272 result = [] if fetch else None 

1273 for page in _paginate(argslist, page_size=page_size): 

1274 if template is None: 

1275 template = b'(' + b','.join([b'%s'] * len(page[0])) + b')' 

1276 parts = pre[:] 

1277 for args in page: 

1278 parts.append(cur.mogrify(template, args)) 

1279 parts.append(b',') 

1280 parts[-1:] = post 

1281 cur.execute(b''.join(parts)) 

1282 if fetch: 

1283 result.extend(cur.fetchall()) 

1284 

1285 return result

1288def _split_sql(sql): 

1289 """Split *sql* on a single ``%s`` placeholder. 

1290 

1291 Split on the %s, perform %% replacement and return pre, post lists of 

1292 snippets. 

1293 """ 

1294 curr = pre = [] 

1295 post = [] 

1296 tokens = _re.split(br'(%.)', sql) 

1297 for token in tokens: 

1298 if len(token) != 2 or token[:1] != b'%': 

1299 curr.append(token) 

1300 continue 

1301 

1302 if token[1:] == b's': 

1303 if curr is pre: 

1304 curr = post 

1305 else: 

1306 raise ValueError( 

1307 "the query contains more than one '%s' placeholder") 

1308 elif token[1:] == b'%': 

1309 curr.append(b'%') 

1310 else: 

1311 raise ValueError("unsupported format character: '%s'" 

1312 % token[1:].decode('ascii', 'replace')) 

1313 

1314 if curr is pre: 

1315 raise ValueError("the query doesn't contain any '%s' placeholder") 

1316 

1317 return pre, post
