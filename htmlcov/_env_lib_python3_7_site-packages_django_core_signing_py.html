1""" 

2Functions for creating and restoring url-safe signed JSON objects. 

3 

4The format used looks like this: 

5 

6>>> signing.dumps("hello") 

7'ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk' 

8 

9There are two components here, separated by a ':'. The first component is a 

10URLsafe base64 encoded JSON of the object passed to dumps(). The second 

11component is a base64 encoded hmac/SHA1 hash of "$first_component:$secret" 

12 

13signing.loads(s) checks the signature and returns the deserialized object. 

14If the signature fails, a BadSignature exception is raised. 

15 

16>>> signing.loads("ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk") 

17'hello' 

18>>> signing.loads("ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk-modified") 

19... 

20BadSignature: Signature failed: ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk-modified 

21 

22You can optionally compress the JSON prior to base64 encoding it to save 

23space, using the compress=True argument. This checks if compression actually 

24helps and only applies compression if the result is a shorter string: 

25 

26>>> signing.dumps(range(1, 20), compress=True) 

27'.eJwFwcERACAIwLCF-rCiILN47r-GyZVJsNgkxaFxoDgxcOHGxMKD_T7vhAml:1QaUaL:BA0thEZrp4FQVXIXuOvYJtLJSrQ' 

28 

29The fact that the string is compressed is signalled by the prefixed '.' at the 

30start of the base64 JSON. 

31 

32There are 65 url-safe characters: the 64 used by url-safe base64 and the ':'. 

33These functions make use of all of them. 

34"""

63 return base64.urlsafe_b64encode(s).strip(b'=')

66def b64_decode(s):

93def dumps(obj, key=None, salt='django.core.signing', serializer=JSONSerializer, compress=False): 

94 """ 

95 Return URL-safe, hmac/SHA1 signed base64 compressed JSON string. If key is 

96 None, use settings.SECRET_KEY instead. 

97 

98 If compress is True (not the default), check if compressing using zlib can 

99 save some space. Prepend a '.' to signify compression. This is included 

100 in the signature, to protect against zip bombs. 

101 

102 Salt can be used to namespace the hash, so that a signed string is 

103 only valid for a given namespace. Leaving this at the default 

104 value or re-using a salt value across different parts of your 

105 application without good cause is a security risk. 

106 

107 The serializer is expected to return a bytestring. 

108 """ 

109 data = serializer().dumps(obj) 

110 

111 # Flag for if it's been compressed or not 

112 is_compressed = False 

113 

114 if compress: 

115 # Avoid zlib dependency unless compress is being used 

116 compressed = zlib.compress(data) 

117 if len(compressed) < (len(data) - 1): 

118 data = compressed 

119 is_compressed = True 

120 base64d = b64_encode(data).decode() 

121 if is_compressed: 

122 base64d = '.' + base64d 

123 return TimestampSigner(key, salt=salt).sign(base64d)

126def loads(s, key=None, salt='django.core.signing', serializer=JSONSerializer, max_age=None): 

127 """ 

128 Reverse of dumps(), raise BadSignature if signature fails. 

129 

130 The serializer is expected to accept a bytestring. 

131 """ 

132 # TimestampSigner.unsign() returns str but base64 and zlib compression 

133 # operate on bytes. 

134 base64d = force_bytes(TimestampSigner(key, salt=salt).unsign(s, max_age=max_age)) 

135 decompress = False 

136 if base64d[:1] == b'.': 

137 # It's compressed; uncompress it first 

138 base64d = base64d[1:] 

139 decompress = True 

140 data = b64_decode(base64d) 

141 if decompress: 

142 data = zlib.decompress(data) 

143 return serializer().loads(data)

146class Signer: 

147 

148 def __init__(self, key=None, sep=':', salt=None): 

149 # Use of native strings in all versions of Python 

150 self.key = key or settings.SECRET_KEY 

151 self.sep = sep 

152 if _SEP_UNSAFE.match(self.sep): 

153 raise ValueError( 

154 'Unsafe Signer separator: %r (cannot be empty or consist of ' 

155 'only A-z0-9-_=)' % sep, 

156 ) 

157 self.salt = salt or '%s.%s' % (self.__class__.__module__, self.__class__.__name__) 

158 

159 def signature(self, value): 

160 return base64_hmac(self.salt + 'signer', value, self.key) 

161 

162 def sign(self, value): 

163 return '%s%s%s' % (value, self.sep, self.signature(value)) 

164 

165 def unsign(self, signed_value): 

166 if self.sep not in signed_value: 

167 raise BadSignature('No "%s" found in value' % self.sep) 

168 value, sig = signed_value.rsplit(self.sep, 1) 

169 if constant_time_compare(sig, self.signature(value)): 

170 return value 

171 raise BadSignature('Signature "%s" does not match' % sig)

174class TimestampSigner(Signer): 

175 

176 def timestamp(self): 

177 return baseconv.base62.encode(int(time.time())) 

178 

179 def sign(self, value): 

180 value = '%s%s%s' % (value, self.sep, self.timestamp()) 

181 return super().sign(value) 

182 

183 def unsign(self, value, max_age=None): 

184 """ 

185 Retrieve original value and check it wasn't signed more 

186 than max_age seconds ago. 

187 """ 

188 result = super().unsign(value) 

189 value, timestamp = result.rsplit(self.sep, 1) 

190 timestamp = baseconv.base62.decode(timestamp) 

191 if max_age is not None: 

192 if isinstance(max_age, datetime.timedelta): 

193 max_age = max_age.total_seconds() 

194 # Check timestamp is not older than max_age 

195 age = time.time() - timestamp 

196 if age > max_age: 

197 raise SignatureExpired( 

198 'Signature age %s > %s seconds' % (age, max_age)) 

199 return value
