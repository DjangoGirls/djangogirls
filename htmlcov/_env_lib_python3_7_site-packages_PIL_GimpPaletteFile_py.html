35

39 

40 # skip fields and comment lines

42 continue 

43 if len(s) > 100: 

44 raise SyntaxError("bad palette file") 

45

49 

50 self.palette[i] = o8(v[0]) + o8(v[1]) + o8(v[2]) 

51
