102 def _add_view(self, request, form_url='', extra_context=None): 

103 # It's an error for a user to have add permission but NOT change 

104 # permission for users. If we allowed such users to add users, they 

105 # could create superusers, which would mean they would essentially have 

106 # the permission to change users. To avoid the problem entirely, we 

107 # disallow users from adding users if they don't have change 

108 # permission. 

109 if not self.has_change_permission(request): 

110 if self.has_add_permission(request) and settings.DEBUG: 

111 # Raise Http404 in debug mode so that the user gets a helpful 

112 # error message. 

113 raise Http404( 

114 'Your user does not have the "Change user" permission. In ' 

115 'order to add users, Django requires that your user ' 

116 'account have both the "Add user" and "Change user" ' 

117 'permissions set.') 

118 raise PermissionDenied 

119 if extra_context is None: 

120 extra_context = {} 

121 username_field = self.model._meta.get_field(self.model.USERNAME_FIELD) 

122 defaults = { 

123 'auto_populated_fields': (), 

124 'username_help_text': username_field.help_text, 

125 } 

126 extra_context.update(defaults) 

127 return super().add_view(request, form_url, extra_context)

132 raise PermissionDenied 

133 user = self.get_object(request, unquote(id)) 

134 if user is None: 

135 raise Http404(_('%(name)s object with primary key %(key)r does not exist.') % { 

136 'name': self.model._meta.verbose_name, 

137 'key': escape(id),

140 form = self.change_password_form(user, request.POST)

192 def response_add(self, request, obj, post_url_continue=None): 

193 """ 

194 Determine the HttpResponse for the add_view stage. It mostly defers to 

195 its superclass implementation but is customized because the User model 

196 has a slightly different workflow. 

197 """ 

198 # We should allow further modification of the user just added i.e. the 

199 # 'Save' button should behave like the 'Save and continue editing' 

200 # button except in two scenarios: 

201 # * The user has pressed the 'Save and add another' button 

202 # * We are adding a user in a popup 

203 if '_addanother' not in request.POST and IS_POPUP_VAR not in request.POST: 

204 request.POST = request.POST.copy() 

205 request.POST['_continue'] = 1 

206 return super().response_add(request, obj, post_url_continue)
