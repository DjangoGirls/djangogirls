23def set_language(request): 

24 """ 

25 Redirect to a given URL while setting the chosen language in the session or 

26 cookie. The URL and the language code need to be specified in the request 

27 parameters. 

28 

29 Since this view changes how the user will see the rest of the site, it must 

30 only be accessed as a POST request. If called as a GET request, it will 

31 redirect to the page in the request (the 'next' parameter) without changing 

32 any state. 

33 """ 

34 next = request.POST.get('next', request.GET.get('next')) 

35 if ((next or not request.is_ajax()) and 

36 not is_safe_url(url=next, allowed_hosts={request.get_host()}, require_https=request.is_secure())): 

37 next = request.META.get('HTTP_REFERER') 

38 if next: 

39 next = unquote(next) # HTTP_REFERER may be encoded. 

40 if not is_safe_url(url=next, allowed_hosts={request.get_host()}, require_https=request.is_secure()): 

41 next = '/' 

42 response = HttpResponseRedirect(next) if next else HttpResponse(status=204) 

43 if request.method == 'POST': 

44 lang_code = request.POST.get(LANGUAGE_QUERY_PARAMETER) 

45 if lang_code and check_for_language(lang_code): 

46 if next: 

47 next_trans = translate_url(next, lang_code) 

48 if next_trans != next: 

49 response = HttpResponseRedirect(next_trans) 

50 if hasattr(request, 'session'): 

51 request.session[LANGUAGE_SESSION_KEY] = lang_code 

52 else: 

53 response.set_cookie( 

54 settings.LANGUAGE_COOKIE_NAME, lang_code, 

55 max_age=settings.LANGUAGE_COOKIE_AGE, 

56 path=settings.LANGUAGE_COOKIE_PATH, 

57 domain=settings.LANGUAGE_COOKIE_DOMAIN, 

58 ) 

59 return response

205class JavaScriptCatalog(View): 

206 """ 

207 Return the selected language catalog as a JavaScript library. 

208 

209 Receive the list of packages to check for translations in the `packages` 

210 kwarg either from the extra dictionary passed to the url() function or as a 

211 plus-sign delimited string from the request. Default is 'django.conf'. 

212 

213 You can override the gettext domain for this view, but usually you don't 

214 want to do that as JavaScript messages go to the djangojs domain. This 

215 might be needed if you deliver your JavaScript source from Django templates. 

216 """ 

217 domain = 'djangojs' 

218 packages = None 

219 

220 def get(self, request, *args, **kwargs): 

221 locale = get_language() 

222 domain = kwargs.get('domain', self.domain) 

223 # If packages are not provided, default to all installed packages, as 

224 # DjangoTranslation without localedirs harvests them all. 

225 packages = kwargs.get('packages', '') 

226 packages = packages.split('+') if packages else self.packages 

227 paths = self.get_paths(packages) if packages else None 

228 self.translation = DjangoTranslation(locale, domain=domain, localedirs=paths) 

229 context = self.get_context_data(**kwargs) 

230 return self.render_to_response(context) 

231 

232 def get_paths(self, packages): 

233 allowable_packages = {app_config.name: app_config for app_config in apps.get_app_configs()} 

234 app_configs = [allowable_packages[p] for p in packages if p in allowable_packages] 

235 if len(app_configs) < len(packages): 

236 excluded = [p for p in packages if p not in allowable_packages] 

237 raise ValueError( 

238 'Invalid package(s) provided to JavaScriptCatalog: %s' % ','.join(excluded) 

239 ) 

240 # paths of requested packages 

241 return [os.path.join(app.path, 'locale') for app in app_configs]

244 def _num_plurals(self): 

245 """ 

246 Return the number of plurals for this catalog language, or 2 if no 

247 plural string is available. 

248 """ 

249 match = re.search(r'nplurals=\s*(\d+)', self._plural_string or '') 

250 if match: 

251 return int(match.groups()[0]) 

252 return 2

254 @property 

255 def _plural_string(self): 

256 """ 

257 Return the plural string (including nplurals) for this catalog language, 

258 or None if no plural string is available. 

259 """ 

260 if '' in self.translation._catalog: 

261 for line in self.translation._catalog[''].split('\n'): 

262 if line.startswith('Plural-Forms:'): 

263 return line.split(':', 1)[1].strip() 

264 return None 

265 

266 def get_plural(self): 

267 plural = self._plural_string 

268 if plural is not None: 

269 # This should be a compiled function of a typical plural-form: 

270 # Plural-Forms: nplurals=3; plural=n%10==1 && n%100!=11 ? 0 : 

271 # n%10>=2 && n%10<=4 && (n%100<10 || n%100>=20) ? 1 : 2; 

272 plural = [el.strip() for el in plural.split(';') if el.strip().startswith('plural=')][0].split('=', 1)[1] 

273 return plural

275 def get_catalog(self): 

276 pdict = {} 

277 num_plurals = self._num_plurals 

278 catalog = {} 

279 trans_cat = self.translation._catalog 

280 trans_fallback_cat = self.translation._fallback._catalog if self.translation._fallback else {} 

281 seen_keys = set() 

282 for key, value in itertools.chain(trans_cat.items(), trans_fallback_cat.items()): 

283 if key == '' or key in seen_keys: 

284 continue 

285 if isinstance(key, str): 

286 catalog[key] = value 

287 elif isinstance(key, tuple): 

288 msgid, cnt = key 

289 pdict.setdefault(msgid, {})[cnt] = value 

290 else: 

291 raise TypeError(key) 

292 seen_keys.add(key) 

293 for k, v in pdict.items(): 

294 catalog[k] = [v.get(i, '') for i in range(num_plurals)] 

295 return catalog
