19class MigrationAutodetector: 

20 """ 

21 Take a pair of ProjectStates and compare them to see what the first would 

22 need doing to make it match the second (the second usually being the 

23 project's current state). 

24 

25 Note that this naturally operates on entire projects at a time, 

26 as it's likely that changes interact (for example, you can't 

27 add a ForeignKey without having a migration to add the table it 

28 depends on first). A user interface may offer single-app usage 

29 if it wishes, with the caveat that it may not always be possible. 

30 """

35 self.questioner = questioner or MigrationQuestioner() 

36 self.existing_apps = {app for app, model in from_state.models}

50 def deep_deconstruct(self, obj): 

51 """ 

52 Recursive deconstruction for a field and its arguments. 

53 Used for full comparison for rename/alter; sometimes a single-level 

54 deconstruction will not compare correctly. 

55 """ 

56 if isinstance(obj, list): 

57 return [self.deep_deconstruct(value) for value in obj] 

58 elif isinstance(obj, tuple): 

59 return tuple(self.deep_deconstruct(value) for value in obj) 

60 elif isinstance(obj, dict): 

61 return { 

62 key: self.deep_deconstruct(value) 

63 for key, value in obj.items() 

64 } 

65 elif isinstance(obj, functools.partial): 

66 return (obj.func, self.deep_deconstruct(obj.args), self.deep_deconstruct(obj.keywords)) 

67 elif isinstance(obj, COMPILED_REGEX_TYPE): 

68 return RegexObject(obj) 

69 elif isinstance(obj, type): 

70 # If this is a type that implements 'deconstruct' as an instance method, 

71 # avoid treating this as being deconstructible itself - see #22951 

72 return obj 

73 elif hasattr(obj, 'deconstruct'): 

74 deconstructed = obj.deconstruct() 

75 if isinstance(obj, models.Field): 

76 # we have a field which also returns a name 

77 deconstructed = deconstructed[1:] 

78 path, args, kwargs = deconstructed 

79 return ( 

80 path, 

81 [self.deep_deconstruct(value) for value in args], 

82 { 

83 key: self.deep_deconstruct(value) 

84 for key, value in kwargs.items() 

85 }, 

86 ) 

87 else: 

88 return obj

90 def only_relation_agnostic_fields(self, fields): 

91 """ 

92 Return a definition of the fields that ignores field names and 

93 what related fields actually relate to. Used for detecting renames (as, 

94 of course, the related fields change during renames). 

95 """ 

96 fields_def = [] 

97 for name, field in sorted(fields): 

98 deconstruction = self.deep_deconstruct(field) 

99 if field.remote_field and field.remote_field.model: 

100 del deconstruction[2]['to'] 

101 fields_def.append(deconstruction) 

102 return fields_def

104 def _detect_changes(self, convert_apps=None, graph=None): 

105 """ 

106 Return a dict of migration plans which will achieve the 

107 change from from_state to to_state. The dict has app labels 

108 as keys and a list of migrations as values. 

109 

110 The resulting migrations aren't specially named, but the names 

111 do matter for dependencies inside the set. 

112 

113 convert_apps is the list of apps to convert to use migrations 

114 (i.e. to make initial migrations for, in the usual case) 

115 

116 graph is an optional argument that, if provided, can help improve 

117 dependency generation and avoid potential circular dependencies. 

118 """ 

119 # The first phase is generating all the operations for each app 

120 # and gathering them into a big per-app list. 

121 # Then go through that list, order it, and split into migrations to 

122 # resolve dependencies caused by M2Ms and FKs. 

123 self.generated_operations = {} 

124 self.altered_indexes = {} 

125 

126 # Prepare some old/new state and model lists, separating 

127 # proxy models and ignoring unmigrated apps. 

128 self.old_apps = self.from_state.concrete_apps 

129 self.new_apps = self.to_state.apps 

130 self.old_model_keys = set() 

131 self.old_proxy_keys = set() 

132 self.old_unmanaged_keys = set() 

133 self.new_model_keys = set() 

134 self.new_proxy_keys = set() 

135 self.new_unmanaged_keys = set() 

136 for al, mn in self.from_state.models: 

137 model = self.old_apps.get_model(al, mn) 

138 if not model._meta.managed: 

139 self.old_unmanaged_keys.add((al, mn)) 

140 elif al not in self.from_state.real_apps: 

141 if model._meta.proxy: 

142 self.old_proxy_keys.add((al, mn)) 

143 else: 

144 self.old_model_keys.add((al, mn)) 

145 

146 for al, mn in self.to_state.models: 

147 model = self.new_apps.get_model(al, mn) 

148 if not model._meta.managed: 

149 self.new_unmanaged_keys.add((al, mn)) 

150 elif ( 

151 al not in self.from_state.real_apps or 

152 (convert_apps and al in convert_apps) 

153 ): 

154 if model._meta.proxy: 

155 self.new_proxy_keys.add((al, mn)) 

156 else: 

157 self.new_model_keys.add((al, mn)) 

158 

159 # Renames have to come first 

160 self.generate_renamed_models() 

161 

162 # Prepare lists of fields and generate through model map 

163 self._prepare_field_lists() 

164 self._generate_through_model_map() 

165 

166 # Generate non-rename model operations 

167 self.generate_deleted_models() 

168 self.generate_created_models() 

169 self.generate_deleted_proxies() 

170 self.generate_created_proxies() 

171 self.generate_altered_options() 

172 self.generate_altered_managers() 

173 

174 # Create the altered indexes and store them in self.altered_indexes. 

175 # This avoids the same computation in generate_removed_indexes() 

176 # and generate_added_indexes(). 

177 self.create_altered_indexes() 

178 # Generate index removal operations before field is removed 

179 self.generate_removed_indexes() 

180 # Generate field operations 

181 self.generate_renamed_fields() 

182 self.generate_removed_fields() 

183 self.generate_added_fields() 

184 self.generate_altered_fields() 

185 self.generate_altered_unique_together() 

186 self.generate_altered_index_together() 

187 self.generate_added_indexes() 

188 self.generate_altered_db_table() 

189 self.generate_altered_order_with_respect_to() 

190 

191 self._sort_migrations() 

192 self._build_migration_list(graph) 

193 self._optimize_migrations() 

194 

195 return self.migrations

236 def _build_migration_list(self, graph=None): 

237 """ 

238 Chop the lists of operations up into migrations with dependencies on 

239 each other. Do this by going through an app's list of operations until 

240 one is found that has an outgoing dependency that isn't in another 

241 app's migration yet (hasn't been chopped off its list). Then chop off 

242 the operations before it into a migration and move onto the next app. 

243 If the loops completes without doing anything, there's a circular 

244 dependency (which _should_ be impossible as the operations are 

245 all split at this point so they can't depend and be depended on). 

246 """ 

247 self.migrations = {} 

248 num_ops = sum(len(x) for x in self.generated_operations.values()) 

249 chop_mode = False 

250 while num_ops: 

251 # On every iteration, we step through all the apps and see if there 

252 # is a completed set of operations. 

253 # If we find that a subset of the operations are complete we can 

254 # try to chop it off from the rest and continue, but we only 

255 # do this if we've already been through the list once before 

256 # without any chopping and nothing has changed. 

257 for app_label in sorted(self.generated_operations): 

258 chopped = [] 

259 dependencies = set() 

260 for operation in list(self.generated_operations[app_label]): 

261 deps_satisfied = True 

262 operation_dependencies = set() 

263 for dep in operation._auto_deps: 

264 is_swappable_dep = False 

265 if dep[0] == "__setting__": 

266 # We need to temporarily resolve the swappable dependency to prevent 

267 # circular references. While keeping the dependency checks on the 

268 # resolved model we still add the swappable dependencies. 

269 # See #23322 

270 resolved_app_label, resolved_object_name = getattr(settings, dep[1]).split('.') 

271 original_dep = dep 

272 dep = (resolved_app_label, resolved_object_name.lower(), dep[2], dep[3]) 

273 is_swappable_dep = True 

274 if dep[0] != app_label and dep[0] != "__setting__": 

275 # External app dependency. See if it's not yet 

276 # satisfied. 

277 for other_operation in self.generated_operations.get(dep[0], []): 

278 if self.check_dependency(other_operation, dep): 

279 deps_satisfied = False 

280 break 

281 if not deps_satisfied: 

282 break 

283 else: 

284 if is_swappable_dep: 

285 operation_dependencies.add((original_dep[0], original_dep[1])) 

286 elif dep[0] in self.migrations: 

287 operation_dependencies.add((dep[0], self.migrations[dep[0]][-1].name)) 

288 else: 

289 # If we can't find the other app, we add a first/last dependency, 

290 # but only if we've already been through once and checked everything 

291 if chop_mode: 

292 # If the app already exists, we add a dependency on the last migration, 

293 # as we don't know which migration contains the target field. 

294 # If it's not yet migrated or has no migrations, we use __first__ 

295 if graph and graph.leaf_nodes(dep[0]): 

296 operation_dependencies.add(graph.leaf_nodes(dep[0])[0]) 

297 else: 

298 operation_dependencies.add((dep[0], "__first__")) 

299 else: 

300 deps_satisfied = False 

301 if deps_satisfied: 

302 chopped.append(operation) 

303 dependencies.update(operation_dependencies) 

304 self.generated_operations[app_label] = self.generated_operations[app_label][1:] 

305 else: 

306 break 

307 # Make a migration! Well, only if there's stuff to put in it 

308 if dependencies or chopped: 

309 if not self.generated_operations[app_label] or chop_mode: 

310 subclass = type("Migration", (Migration,), {"operations": [], "dependencies": []}) 

311 instance = subclass("auto_%i" % (len(self.migrations.get(app_label, [])) + 1), app_label) 

312 instance.dependencies = list(dependencies) 

313 instance.operations = chopped 

314 instance.initial = app_label not in self.existing_apps 

315 self.migrations.setdefault(app_label, []).append(instance) 

316 chop_mode = False 

317 else: 

318 self.generated_operations[app_label] = chopped + self.generated_operations[app_label] 

319 new_num_ops = sum(len(x) for x in self.generated_operations.values()) 

320 if new_num_ops == num_ops: 

321 if not chop_mode: 

322 chop_mode = True 

323 else: 

324 raise ValueError("Cannot resolve operation dependencies: %r" % self.generated_operations) 

325 num_ops = new_num_ops

327 def _sort_migrations(self): 

328 """ 

329 Reorder to make things possible. Reordering may be needed so FKs work 

330 nicely inside the same app. 

331 """ 

332 for app_label, ops in sorted(self.generated_operations.items()): 

333 # construct a dependency graph for intra-app dependencies 

334 dependency_graph = {op: set() for op in ops} 

335 for op in ops: 

336 for dep in op._auto_deps: 

337 if dep[0] == app_label: 

338 for op2 in ops: 

339 if self.check_dependency(op2, dep): 

340 dependency_graph[op].add(op2) 

341 

342 # we use a stable sort for deterministic tests & general behavior 

343 self.generated_operations[app_label] = stable_topological_sort(ops, dependency_graph)

345 def _optimize_migrations(self): 

346 # Add in internal dependencies among the migrations 

347 for app_label, migrations in self.migrations.items(): 

348 for m1, m2 in zip(migrations, migrations[1:]): 

349 m2.dependencies.append((app_label, m1.name)) 

350 

351 # De-dupe dependencies 

352 for app_label, migrations in self.migrations.items(): 

353 for migration in migrations: 

354 migration.dependencies = list(set(migration.dependencies)) 

355 

356 # Optimize migrations 

357 for app_label, migrations in self.migrations.items(): 

358 for migration in migrations: 

359 migration.operations = MigrationOptimizer().optimize(migration.operations, app_label=app_label)

361 def check_dependency(self, operation, dependency): 

362 """ 

363 Return True if the given operation depends on the given dependency, 

364 False otherwise. 

365 """ 

366 # Created model 

367 if dependency[2] is None and dependency[3] is True: 

368 return ( 

369 isinstance(operation, operations.CreateModel) and 

370 operation.name_lower == dependency[1].lower() 

371 ) 

372 # Created field 

373 elif dependency[2] is not None and dependency[3] is True: 

374 return ( 

375 ( 

376 isinstance(operation, operations.CreateModel) and 

377 operation.name_lower == dependency[1].lower() and 

378 any(dependency[2] == x for x, y in operation.fields) 

379 ) or 

380 ( 

381 isinstance(operation, operations.AddField) and 

382 operation.model_name_lower == dependency[1].lower() and 

383 operation.name_lower == dependency[2].lower() 

384 ) 

385 ) 

386 # Removed field 

387 elif dependency[2] is not None and dependency[3] is False: 

388 return ( 

389 isinstance(operation, operations.RemoveField) and 

390 operation.model_name_lower == dependency[1].lower() and 

391 operation.name_lower == dependency[2].lower() 

392 ) 

393 # Removed model 

394 elif dependency[2] is None and dependency[3] is False: 

395 return ( 

396 isinstance(operation, operations.DeleteModel) and 

397 operation.name_lower == dependency[1].lower() 

398 ) 

399 # Field being altered 

400 elif dependency[2] is not None and dependency[3] == "alter": 

401 return ( 

402 isinstance(operation, operations.AlterField) and 

403 operation.model_name_lower == dependency[1].lower() and 

404 operation.name_lower == dependency[2].lower() 

405 ) 

406 # order_with_respect_to being unset for a field 

407 elif dependency[2] is not None and dependency[3] == "order_wrt_unset": 

408 return ( 

409 isinstance(operation, operations.AlterOrderWithRespectTo) and 

410 operation.name_lower == dependency[1].lower() and 

411 (operation.order_with_respect_to or "").lower() != dependency[2].lower() 

412 ) 

413 # Field is removed and part of an index/unique_together 

414 elif dependency[2] is not None and dependency[3] == "foo_together_change": 

415 return ( 

416 isinstance(operation, (operations.AlterUniqueTogether, 

417 operations.AlterIndexTogether)) and 

418 operation.name_lower == dependency[1].lower() 

419 ) 

420 # Unknown dependency. Raise an error. 

421 else: 

422 raise ValueError("Can't handle dependency %r" % (dependency, ))

432 def swappable_first_key(self, item): 

433 """ 

434 Place potential swappable models first in lists of created models (only 

435 real way to solve #22783). 

436 """

438 model = self.new_apps.get_model(item[0], item[1])

452 def generate_renamed_models(self): 

453 """ 

454 Find any renamed models, generate the operations for them, and remove 

455 the old entry from the model lists. Must be run before other 

456 model-level generation. 

457 """ 

458 self.renamed_models = {} 

459 self.renamed_models_rel = {} 

460 added_models = self.new_model_keys - self.old_model_keys 

461 for app_label, model_name in sorted(added_models): 

462 model_state = self.to_state.models[app_label, model_name] 

463 model_fields_def = self.only_relation_agnostic_fields(model_state.fields) 

464 

465 removed_models = self.old_model_keys - self.new_model_keys 

466 for rem_app_label, rem_model_name in removed_models: 

467 if rem_app_label == app_label: 

468 rem_model_state = self.from_state.models[rem_app_label, rem_model_name] 

469 rem_model_fields_def = self.only_relation_agnostic_fields(rem_model_state.fields) 

470 if model_fields_def == rem_model_fields_def: 

471 if self.questioner.ask_rename_model(rem_model_state, model_state): 

472 model_opts = self.new_apps.get_model(app_label, model_name)._meta 

473 dependencies = [] 

474 for field in model_opts.get_fields(): 

475 if field.is_relation: 

476 dependencies.extend(self._get_dependencies_for_foreign_key(field)) 

477 self.add_operation( 

478 app_label, 

479 operations.RenameModel( 

480 old_name=rem_model_state.name, 

481 new_name=model_state.name, 

482 ), 

483 dependencies=dependencies, 

484 ) 

485 self.renamed_models[app_label, model_name] = rem_model_name 

486 renamed_models_rel_key = '%s.%s' % (rem_model_state.app_label, rem_model_state.name) 

487 self.renamed_models_rel[renamed_models_rel_key] = '%s.%s' % ( 

488 model_state.app_label, 

489 model_state.name, 

490 ) 

491 self.old_model_keys.remove((rem_app_label, rem_model_name)) 

492 self.old_model_keys.add((app_label, model_name)) 

493 break

495 def generate_created_models(self): 

496 """ 

497 Find all new models (both managed and unmanaged) and make create 

498 operations for them as well as separate operations to create any 

499 foreign key or M2M relationships (these are optimized later, if 

500 possible). 

501 

502 Defer any model options that refer to collections of fields that might 

503 be deferred (e.g. unique_together, index_together). 

504 """ 

505 old_keys = self.old_model_keys | self.old_unmanaged_keys 

506 added_models = self.new_model_keys - old_keys 

507 added_unmanaged_models = self.new_unmanaged_keys - old_keys 

508 all_added_models = chain( 

509 sorted(added_models, key=self.swappable_first_key, reverse=True), 

510 sorted(added_unmanaged_models, key=self.swappable_first_key, reverse=True) 

511 ) 

512 for app_label, model_name in all_added_models: 

513 model_state = self.to_state.models[app_label, model_name] 

514 model_opts = self.new_apps.get_model(app_label, model_name)._meta 

515 # Gather related fields 

516 related_fields = {} 

517 primary_key_rel = None 

518 for field in model_opts.local_fields: 

519 if field.remote_field: 

520 if field.remote_field.model: 

521 if field.primary_key: 

522 primary_key_rel = field.remote_field.model 

523 elif not field.remote_field.parent_link: 

524 related_fields[field.name] = field 

525 # through will be none on M2Ms on swapped-out models; 

526 # we can treat lack of through as auto_created=True, though. 

527 if (getattr(field.remote_field, "through", None) and 

528 not field.remote_field.through._meta.auto_created): 

529 related_fields[field.name] = field 

530 for field in model_opts.local_many_to_many: 

531 if field.remote_field.model: 

532 related_fields[field.name] = field 

533 if getattr(field.remote_field, "through", None) and not field.remote_field.through._meta.auto_created: 

534 related_fields[field.name] = field 

535 # Are there indexes/unique|index_together to defer? 

536 indexes = model_state.options.pop('indexes') 

537 unique_together = model_state.options.pop('unique_together', None) 

538 index_together = model_state.options.pop('index_together', None) 

539 order_with_respect_to = model_state.options.pop('order_with_respect_to', None) 

540 # Depend on the deletion of any possible proxy version of us 

541 dependencies = [ 

542 (app_label, model_name, None, False), 

543 ] 

544 # Depend on all bases 

545 for base in model_state.bases: 

546 if isinstance(base, str) and "." in base: 

547 base_app_label, base_name = base.split(".", 1) 

548 dependencies.append((base_app_label, base_name, None, True)) 

549 # Depend on the other end of the primary key if it's a relation 

550 if primary_key_rel: 

551 dependencies.append(( 

552 primary_key_rel._meta.app_label, 

553 primary_key_rel._meta.object_name, 

554 None, 

555 True 

556 )) 

557 # Generate creation operation 

558 self.add_operation( 

559 app_label, 

560 operations.CreateModel( 

561 name=model_state.name, 

562 fields=[d for d in model_state.fields if d[0] not in related_fields], 

563 options=model_state.options, 

564 bases=model_state.bases, 

565 managers=model_state.managers, 

566 ), 

567 dependencies=dependencies, 

568 beginning=True, 

569 ) 

570 

571 # Don't add operations which modify the database for unmanaged models 

572 if not model_opts.managed: 

573 continue 

574 

575 # Generate operations for each related field 

576 for name, field in sorted(related_fields.items()): 

577 dependencies = self._get_dependencies_for_foreign_key(field) 

578 # Depend on our own model being created

684 def generate_deleted_models(self): 

685 """ 

686 Find all deleted models (managed and unmanaged) and make delete 

687 operations for them as well as separate operations to delete any 

688 foreign key or M2M relationships (these are optimized later, if 

689 possible). 

690 

691 Also bring forward removal of any model options that refer to 

692 collections of fields - the inverse of generate_created_models(). 

693 """ 

694 new_keys = self.new_model_keys | self.new_unmanaged_keys 

695 deleted_models = self.old_model_keys - new_keys 

696 deleted_unmanaged_models = self.old_unmanaged_keys - new_keys 

697 all_deleted_models = chain(sorted(deleted_models), sorted(deleted_unmanaged_models)) 

698 for app_label, model_name in all_deleted_models: 

699 model_state = self.from_state.models[app_label, model_name] 

700 model = self.old_apps.get_model(app_label, model_name) 

701 if not model._meta.managed: 

702 # Skip here, no need to handle fields for unmanaged models 

703 continue 

704 

705 # Gather related fields 

706 related_fields = {} 

707 for field in model._meta.local_fields: 

708 if field.remote_field: 

709 if field.remote_field.model: 

710 related_fields[field.name] = field 

711 # through will be none on M2Ms on swapped-out models; 

712 # we can treat lack of through as auto_created=True, though. 

713 if (getattr(field.remote_field, "through", None) and 

714 not field.remote_field.through._meta.auto_created): 

715 related_fields[field.name] = field 

716 for field in model._meta.local_many_to_many: 

717 if field.remote_field.model: 

718 related_fields[field.name] = field 

719 if getattr(field.remote_field, "through", None) and not field.remote_field.through._meta.auto_created: 

720 related_fields[field.name] = field 

721 # Generate option removal first 

722 unique_together = model_state.options.pop('unique_together', None) 

723 index_together = model_state.options.pop('index_together', None) 

724 if unique_together: 

725 self.add_operation( 

726 app_label, 

727 operations.AlterUniqueTogether( 

728 name=model_name, 

729 unique_together=None, 

730 ) 

731 ) 

732 if index_together: 

733 self.add_operation( 

734 app_label, 

735 operations.AlterIndexTogether( 

736 name=model_name, 

737 index_together=None, 

738 ) 

739 ) 

740 # Then remove each related field 

741 for name, field in sorted(related_fields.items()): 

742 self.add_operation( 

743 app_label, 

744 operations.RemoveField( 

745 model_name=model_name, 

746 name=name, 

747 ) 

748 ) 

749 # Finally, remove the model. 

750 # This depends on both the removal/alteration of all incoming fields 

751 # and the removal of all its own related fields, and if it's 

752 # a through model the field that references it. 

753 dependencies = [] 

754 for related_object in model._meta.related_objects: 

755 related_object_app_label = related_object.related_model._meta.app_label 

756 object_name = related_object.related_model._meta.object_name 

757 field_name = related_object.field.name 

758 dependencies.append((related_object_app_label, object_name, field_name, False)) 

759 if not related_object.many_to_many: 

760 dependencies.append((related_object_app_label, object_name, field_name, "alter")) 

761 

762 for name, field in sorted(related_fields.items()): 

763 dependencies.append((app_label, model_name, name, False)) 

764 # We're referenced in another field's through= 

765 through_user = self.through_users.get((app_label, model_state.name_lower)) 

766 if through_user: 

767 dependencies.append((through_user[0], through_user[1], through_user[2], False)) 

768 # Finally, make the operation, deduping any dependencies 

769 self.add_operation( 

770 app_label, 

771 operations.DeleteModel( 

772 name=model_state.name, 

773 ), 

774 dependencies=list(set(dependencies)), 

775 )

790 def generate_renamed_fields(self): 

791 """Work out renamed fields.""" 

792 self.renamed_fields = {} 

793 for app_label, model_name, field_name in sorted(self.new_field_keys - self.old_field_keys): 

794 old_model_name = self.renamed_models.get((app_label, model_name), model_name) 

795 old_model_state = self.from_state.models[app_label, old_model_name] 

796 field = self.new_apps.get_model(app_label, model_name)._meta.get_field(field_name) 

797 # Scan to see if this is actually a rename! 

798 field_dec = self.deep_deconstruct(field) 

799 for rem_app_label, rem_model_name, rem_field_name in sorted(self.old_field_keys - self.new_field_keys): 

800 if rem_app_label == app_label and rem_model_name == model_name: 

801 old_field_dec = self.deep_deconstruct(old_model_state.get_field_by_name(rem_field_name)) 

802 if field.remote_field and field.remote_field.model and 'to' in old_field_dec[2]: 

803 old_rel_to = old_field_dec[2]['to'] 

804 if old_rel_to in self.renamed_models_rel: 

805 old_field_dec[2]['to'] = self.renamed_models_rel[old_rel_to] 

806 if old_field_dec == field_dec: 

807 if self.questioner.ask_rename(model_name, rem_field_name, field_name, field): 

808 self.add_operation( 

809 app_label, 

810 operations.RenameField( 

811 model_name=model_name, 

812 old_name=rem_field_name, 

813 new_name=field_name, 

814 ) 

815 ) 

816 self.old_field_keys.remove((rem_app_label, rem_model_name, rem_field_name)) 

817 self.old_field_keys.add((app_label, model_name, field_name)) 

818 self.renamed_fields[app_label, model_name, field_name] = rem_field_name 

819 break

878 def generate_altered_fields(self): 

879 """ 

880 Make AlterField operations, or possibly RemovedField/AddField if alter 

881 isn's possible. 

882 """ 

883 for app_label, model_name, field_name in sorted(self.old_field_keys & self.new_field_keys): 

884 # Did the field change? 

885 old_model_name = self.renamed_models.get((app_label, model_name), model_name) 

886 old_field_name = self.renamed_fields.get((app_label, model_name, field_name), field_name) 

887 old_field = self.old_apps.get_model(app_label, old_model_name)._meta.get_field(old_field_name) 

888 new_field = self.new_apps.get_model(app_label, model_name)._meta.get_field(field_name) 

889 # Implement any model renames on relations; these are handled by RenameModel 

890 # so we need to exclude them from the comparison 

891 if hasattr(new_field, "remote_field") and getattr(new_field.remote_field, "model", None): 

892 rename_key = ( 

893 new_field.remote_field.model._meta.app_label, 

894 new_field.remote_field.model._meta.model_name, 

895 ) 

896 if rename_key in self.renamed_models: 

897 new_field.remote_field.model = old_field.remote_field.model 

898 if hasattr(new_field, "remote_field") and getattr(new_field.remote_field, "through", None): 

899 rename_key = ( 

900 new_field.remote_field.through._meta.app_label, 

901 new_field.remote_field.through._meta.model_name, 

902 ) 

903 if rename_key in self.renamed_models: 

904 new_field.remote_field.through = old_field.remote_field.through 

905 old_field_dec = self.deep_deconstruct(old_field) 

906 new_field_dec = self.deep_deconstruct(new_field) 

907 if old_field_dec != new_field_dec: 

908 both_m2m = old_field.many_to_many and new_field.many_to_many 

909 neither_m2m = not old_field.many_to_many and not new_field.many_to_many 

910 if both_m2m or neither_m2m: 

911 # Either both fields are m2m or neither is 

912 preserve_default = True 

913 if (old_field.null and not new_field.null and not new_field.has_default() and 

914 not new_field.many_to_many): 

915 field = new_field.clone() 

916 new_default = self.questioner.ask_not_null_alteration(field_name, model_name) 

917 if new_default is not models.NOT_PROVIDED: 

918 field.default = new_default 

919 preserve_default = False 

920 else: 

921 field = new_field 

922 self.add_operation( 

923 app_label, 

924 operations.AlterField( 

925 model_name=model_name, 

926 name=field_name, 

927 field=field, 

928 preserve_default=preserve_default, 

929 ) 

930 ) 

931 else: 

932 # We cannot alter between m2m and concrete fields 

933 self._generate_removed_field(app_label, model_name, field_name) 

934 self._generate_added_field(app_label, model_name, field_name)

1133 """ 

1134 Take a result from changes() and a MigrationGraph, and fix the names 

1135 and dependencies of the changes so they extend the graph from the leaf 

1136 nodes for each app. 

1137 """ 

1138 leaves = graph.leaf_nodes() 

1139 name_map = {} 

1140 for app_label, migrations in list(changes.items()): 

1141 if not migrations: 

1142 continue 

1143 # Find the app label's current leaf node 

1144 app_leaf = None 

1145 for leaf in leaves: 

1146 if leaf[0] == app_label: 

1147 app_leaf = leaf 

1148 break 

1149 # Do they want an initial migration for this app? 

1150 if app_leaf is None and not self.questioner.ask_initial(app_label): 

1151 # They don't. 

1152 for migration in migrations: 

1153 name_map[(app_label, migration.name)] = (app_label, "__first__") 

1154 del changes[app_label] 

1155 continue 

1156 # Work out the next number in the sequence 

1157 if app_leaf is None: 

1158 next_number = 1 

1159 else: 

1160 next_number = (self.parse_number(app_leaf[1]) or 0) + 1 

1161 # Name each migration 

1162 for i, migration in enumerate(migrations): 

1163 if i == 0 and app_leaf: 

1164 migration.dependencies.append(app_leaf) 

1165 if i == 0 and not app_leaf: 

1166 new_name = "0001_%s" % migration_name if migration_name else "0001_initial" 

1167 else: 

1168 new_name = "%04i_%s" % ( 

1169 next_number, 

1170 migration_name or self.suggest_name(migration.operations)[:100], 

1171 ) 

1172 name_map[(app_label, migration.name)] = (app_label, new_name) 

1173 next_number += 1 

1174 migration.name = new_name 

1175 # Now fix dependencies 

1176 for app_label, migrations in changes.items(): 

1177 for migration in migrations: 

1178 migration.dependencies = [name_map.get(d, d) for d in migration.dependencies] 

1179 return changes

1181 def _trim_to_apps(self, changes, app_labels): 

1182 """ 

1183 Take changes from arrange_for_graph() and set of app labels, and return 

1184 a modified set of changes which trims out as many migrations that are 

1185 not in app_labels as possible. Note that some other migrations may 

1186 still be present as they may be required dependencies. 

1187 """ 

1188 # Gather other app dependencies in a first pass 

1189 app_dependencies = {} 

1190 for app_label, migrations in changes.items(): 

1191 for migration in migrations: 

1192 for dep_app_label, name in migration.dependencies: 

1193 app_dependencies.setdefault(app_label, set()).add(dep_app_label) 

1194 required_apps = set(app_labels) 

1195 # Keep resolving till there's no change 

1196 old_required_apps = None 

1197 while old_required_apps != required_apps: 

1198 old_required_apps = set(required_apps) 

1199 required_apps.update(*[app_dependencies.get(app_label, ()) for app_label in required_apps]) 

1200 # Remove all migrations that aren't needed 

1201 for app_label in list(changes): 

1202 if app_label not in required_apps: 

1203 del changes[app_label] 

1204 return changes

1206 @classmethod 

1207 def suggest_name(cls, ops): 

1208 """ 

1209 Given a set of operations, suggest a name for the migration they might 

1210 represent. Names are not guaranteed to be unique, but put some effort 

1211 into the fallback name to avoid VCS conflicts if possible. 

1212 """ 

1213 if len(ops) == 1: 

1214 if isinstance(ops[0], operations.CreateModel): 

1215 return ops[0].name_lower 

1216 elif isinstance(ops[0], operations.DeleteModel): 

1217 return "delete_%s" % ops[0].name_lower 

1218 elif isinstance(ops[0], operations.AddField): 

1219 return "%s_%s" % (ops[0].model_name_lower, ops[0].name_lower) 

1220 elif isinstance(ops[0], operations.RemoveField): 

1221 return "remove_%s_%s" % (ops[0].model_name_lower, ops[0].name_lower) 

1222 elif len(ops) > 1: 

1223 if all(isinstance(o, operations.CreateModel) for o in ops): 

1224 return "_".join(sorted(o.name_lower for o in ops)) 

1225 return "auto_%s" % get_migration_name_timestamp()

1227 @classmethod 

1228 def parse_number(cls, name): 

1229 """ 

1230 Given a migration name, try to extract a number from the beginning of 

1231 it. If no number is found, return None. 

1232 """ 

1233 match = re.match(r'^\d+', name) 

1234 if match: 

1235 return int(match.group()) 

1236 return None
