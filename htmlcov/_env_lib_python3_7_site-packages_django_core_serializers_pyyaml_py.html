44 def handle_field(self, obj, field): 

45 # A nasty special case: base YAML doesn't support serialization of time 

46 # types (as opposed to dates or datetimes, which it does support). Since 

47 # we want to use the "safe" serializer for better interoperability, we 

48 # need to do something with those pesky times. Converting 'em to strings 

49 # isn't perfect, but it's better than a "!!python/time" type which would 

50 # halt deserialization under any other language. 

51 if isinstance(field, models.TimeField) and getattr(obj, field.name) is not None: 

52 self._current[field.name] = str(getattr(obj, field.name)) 

53 else: 

54 super().handle_field(obj, field)

64def Deserializer(stream_or_string, **options): 

65 """Deserialize a stream or string of YAML data.""" 

66 if isinstance(stream_or_string, bytes): 

67 stream_or_string = stream_or_string.decode() 

68 if isinstance(stream_or_string, str): 

69 stream = StringIO(stream_or_string) 

70 else: 

71 stream = stream_or_string 

72 try: 

73 yield from PythonDeserializer(yaml.load(stream, Loader=SafeLoader), **options) 

74 except (GeneratorExit, DeserializationError): 

75 raise
