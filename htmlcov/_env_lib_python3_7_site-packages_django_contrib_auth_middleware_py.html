27class RemoteUserMiddleware(MiddlewareMixin): 

28 """ 

29 Middleware for utilizing Web-server-provided authentication. 

30 

31 If request.user is not authenticated, then this middleware attempts to 

32 authenticate the username passed in the ``REMOTE_USER`` request header. 

33 If authentication is successful, the user is automatically logged in to 

34 persist the user in the session. 

35 

36 The header used is configurable and defaults to ``REMOTE_USER``. Subclass 

37 this class and change the ``header`` attribute if you need to use a 

38 different header. 

39 """ 

40 

41 # Name of request header to grab username from. This will be the key as 

42 # used in the request.META dictionary, i.e. the normalization of headers to 

43 # all uppercase and the addition of "HTTP_" prefix apply. 

44 header = "REMOTE_USER" 

45 force_logout_if_no_header = True 

46 

47 def process_request(self, request): 

48 # AuthenticationMiddleware is required so that request.user exists. 

49 if not hasattr(request, 'user'): 

50 raise ImproperlyConfigured( 

51 "The Django remote user auth middleware requires the" 

52 " authentication middleware to be installed. Edit your" 

53 " MIDDLEWARE setting to insert" 

54 " 'django.contrib.auth.middleware.AuthenticationMiddleware'" 

55 " before the RemoteUserMiddleware class.") 

56 try: 

57 username = request.META[self.header] 

58 except KeyError: 

59 # If specified header doesn't exist then remove any existing 

60 # authenticated remote-user, or return (leaving request.user set to 

61 # AnonymousUser by the AuthenticationMiddleware). 

62 if self.force_logout_if_no_header and request.user.is_authenticated: 

63 self._remove_invalid_user(request) 

64 return 

65 # If the user is already authenticated and that user is the user we are 

66 # getting passed in the headers, then the correct user is already 

67 # persisted in the session and we don't need to continue. 

68 if request.user.is_authenticated: 

69 if request.user.get_username() == self.clean_username(username, request): 

70 return 

71 else: 

72 # An authenticated user is associated with the request, but 

73 # it does not match the authorized user in the header. 

74 self._remove_invalid_user(request) 

75 

76 # We are seeing this user for the first time in this session, attempt 

77 # to authenticate the user. 

78 user = auth.authenticate(request, remote_user=username) 

79 if user: 

80 # User is valid. Set request.user and persist user in the session 

81 # by logging the user in. 

82 request.user = user 

83 auth.login(request, user)

85 def clean_username(self, username, request): 

86 """ 

87 Allow the backend to clean the username, if the backend defines a 

88 clean_username method. 

89 """ 

90 backend_str = request.session[auth.BACKEND_SESSION_KEY] 

91 backend = auth.load_backend(backend_str) 

92 try: 

93 username = backend.clean_username(username) 

94 except AttributeError: # Backend has no clean_username method. 

95 pass 

96 return username

99 """ 

100 Remove the current authenticated user in the request which is invalid

103 try:
