876 def _create_index_name(self, table_name, column_names, suffix=""): 

877 """ 

878 Generate a unique name for an index/unique constraint. 

879 

880 The name is divided into 3 parts: the table name, the column names, 

881 and a unique digest and suffix. 

882 """ 

883 _, table_name = split_identifier(table_name) 

884 hash_data = [table_name] + list(column_names) 

885 hash_suffix_part = '%s%s' % (self._digest(*hash_data), suffix) 

886 max_length = self.connection.ops.max_name_length() or 200 

887 # If everything fits into max_length, use that name. 

888 index_name = '%s_%s_%s' % (table_name, '_'.join(column_names), hash_suffix_part) 

889 if len(index_name) <= max_length: 

890 return index_name 

891 # Shorten a long suffix. 

892 if len(hash_suffix_part) > max_length / 3:
