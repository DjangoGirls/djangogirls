547def cycle(parser, token): 

548 """ 

549 Cycle among the given strings each time this tag is encountered. 

550 

551 Within a loop, cycles among the given strings each time through 

552 the loop:: 

553 

554 {% for o in some_list %} 

555 <tr class="{% cycle 'row1' 'row2' %}"> 

556 ... 

557 </tr> 

558 {% endfor %} 

559 

560 Outside of a loop, give the values a unique name the first time you call 

561 it, then use that name each successive time through:: 

562 

563 <tr class="{% cycle 'row1' 'row2' 'row3' as rowcolors %}">...</tr> 

564 <tr class="{% cycle rowcolors %}">...</tr> 

565 <tr class="{% cycle rowcolors %}">...</tr> 

566 

567 You can use any number of values, separated by spaces. Commas can also 

568 be used to separate values; if a comma is used, the cycle values are 

569 interpreted as literal strings. 

570 

571 The optional flag "silent" can be used to prevent the cycle declaration 

572 from returning any value:: 

573 

574 {% for o in some_list %} 

575 {% cycle 'row1' 'row2' as rowcolors silent %} 

576 <tr class="{{ rowcolors }}">{% include "subtemplate.html " %}</tr> 

577 {% endfor %} 

578 """ 

579 # Note: This returns the exact same node on each {% cycle name %} call; 

580 # that is, the node object returned from {% cycle a b c as name %} and the 

581 # one returned from {% cycle name %} are the exact same object. This 

582 # shouldn't cause problems (heh), but if it does, now you know. 

583 # 

584 # Ugly hack warning: This stuffs the named template dict into parser so 

585 # that names are only unique within each template (as opposed to using 

586 # a global variable, which would make cycle names have to be unique across 

587 # *all* templates. 

588 # 

589 # It keeps the last node in the parser to be able to reset it with 

590 # {% resetcycle %}. 

591 

592 args = token.split_contents() 

593 

594 if len(args) < 2: 

595 raise TemplateSyntaxError("'cycle' tag requires at least two arguments") 

596 

597 if len(args) == 2: 

598 # {% cycle foo %} case. 

599 name = args[1] 

600 if not hasattr(parser, '_named_cycle_nodes'): 

601 raise TemplateSyntaxError("No named cycles in template. '%s' is not defined" % name) 

602 if name not in parser._named_cycle_nodes: 

603 raise TemplateSyntaxError("Named cycle '%s' does not exist" % name) 

604 return parser._named_cycle_nodes[name] 

605 

606 as_form = False 

607 

608 if len(args) > 4: 

609 # {% cycle ... as foo [silent] %} case. 

610 if args[-3] == "as": 

611 if args[-1] != "silent": 

612 raise TemplateSyntaxError("Only 'silent' flag is allowed after cycle's name, not '%s'." % args[-1]) 

613 as_form = True

615 args = args[:-1]

617 as_form = True 

618 silent = False
