151 @property 

152 def unicode_flag(self): 

153 """ 

154 Generate a unicode flag for the given country. 

155 

156 The logic for how these are determined can be found at: 

157 

158 https://en.wikipedia.org/wiki/Regional_Indicator_Symbol 

159 

160 Currently, these glyphs appear to only be supported on OS X and iOS. 

161 """ 

162 if not self.code: 

163 return "" 

164 

165 # Don't really like magic numbers, but this is the code point for [A] 

166 # (Regional Indicator A), minus the code point for ASCII A. By adding 

167 # this to the uppercase characters making up the ISO 3166-1 alpha-2 

168 # codes we can get the flag. 

169 OFFSET = 127397 

170 points = [ord(x) + OFFSET for x in self.code.upper()] 

171 

172 try: 

173 # Python 3 is simple: we can just chr() the unicode points. 

174 return chr(points[0]) + chr(points[1]) 

175 except ValueError: 

176 # Python 2 requires us to be a bit more creative. We could use 

177 # unichr(), but that only works if the python has been compiled 

178 # with wide unicode support. This method should always work. 

179 return ("\\U%08x\\U%08x" % tuple(points)).decode("unicode-escape")
