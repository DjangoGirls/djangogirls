53 # recursion depth for each recursion. 

54 def ancestors(self): 

55 # Use self.key instead of self to speed up the frequent hashing 

56 # when constructing an OrderedSet. 

57 if '_ancestors' not in self.__dict__: 

58 ancestors = [] 

59 for parent in sorted(self.parents, reverse=True): 

60 ancestors += parent.ancestors() 

61 ancestors.append(self.key) 

62 self.__dict__['_ancestors'] = list(OrderedSet(ancestors)) 

63 return self.__dict__['_ancestors']

66 # recursion depth for each recursion.

73 descendants += child.descendants() 

74 descendants.append(self.key)

99class MigrationGraph: 

100 """ 

101 Represent the digraph of all migrations in a project. 

102 

103 Each migration is a node, and each dependency is an edge. There are 

104 no implicit dependencies between numbered migrations - the numbering is 

105 merely a convention to aid file listing. Every new numbered migration 

106 has a declared dependency to the previous number, meaning that VCS 

107 branch merges can be detected and resolved. 

108 

109 Migrations files can be marked as replacing another set of migrations - 

110 this is to support the "squash" feature. The graph handler isn't responsible 

111 for these; instead, the code to load them in here should examine the 

112 migration files and if the replaced migrations are all either unapplied 

113 or not present, it should ignore the replaced ones, load in just the 

114 replacing migration, and repoint any dependencies that pointed to the 

115 replaced migrations to point to the replacing one. 

116 

117 A node should be a tuple: (app_path, migration_name). The tree special-cases 

118 things within an app - namely, root nodes and leaf nodes ignore dependencies 

119 to other apps. 

120 """ 

121 

122 def __init__(self): 

123 self.node_map = {} 

124 self.nodes = {} 

125 self.cached = False 

126 

127 def add_node(self, key, migration): 

128 # If the key already exists, then it must be a dummy node. 

129 dummy_node = self.node_map.get(key) 

130 if dummy_node: 

131 # Promote DummyNode to Node. 

132 dummy_node.promote() 

133 else: 

134 node = Node(key) 

135 self.node_map[key] = node 

136 self.nodes[key] = migration 

137 self.clear_cache() 

138 

139 def add_dummy_node(self, key, origin, error_message): 

140 node = DummyNode(key, origin, error_message) 

141 self.node_map[key] = node 

142 self.nodes[key] = None

144 def add_dependency(self, migration, child, parent, skip_validation=False): 

145 """ 

146 This may create dummy nodes if they don't yet exist. If 

147 `skip_validation=True`, validate_consistency() should be called 

148 afterwards. 

149 """ 

150 if child not in self.nodes: 

151 error_message = ( 

152 "Migration %s dependencies reference nonexistent" 

153 " child node %r" % (migration, child) 

154 ) 

155 self.add_dummy_node(child, migration, error_message) 

156 if parent not in self.nodes: 

157 error_message = ( 

158 "Migration %s dependencies reference nonexistent" 

159 " parent node %r" % (migration, parent) 

160 ) 

161 self.add_dummy_node(parent, migration, error_message) 

162 self.node_map[child].add_parent(self.node_map[parent]) 

163 self.node_map[parent].add_child(self.node_map[child]) 

164 if not skip_validation: 

165 self.validate_consistency() 

166 self.clear_cache()

168 def remove_replaced_nodes(self, replacement, replaced): 

169 """ 

170 Remove each of the `replaced` nodes (when they exist). Any 

171 dependencies that were referencing them are changed to reference the 

172 `replacement` node instead. 

173 """ 

174 # Cast list of replaced keys to set to speed up lookup later. 

175 replaced = set(replaced) 

176 try: 

177 replacement_node = self.node_map[replacement] 

178 except KeyError as err: 

179 raise NodeNotFoundError( 

180 "Unable to find replacement node %r. It was either never added" 

181 " to the migration graph, or has been removed." % (replacement, ), 

182 replacement 

183 ) from err 

184 for replaced_key in replaced: 

185 self.nodes.pop(replaced_key, None) 

186 replaced_node = self.node_map.pop(replaced_key, None) 

187 if replaced_node: 

188 for child in replaced_node.children: 

189 child.parents.remove(replaced_node) 

190 # We don't want to create dependencies between the replaced 

191 # node and the replacement node as this would lead to 

192 # self-referencing on the replacement node at a later iteration. 

193 if child.key not in replaced: 

194 replacement_node.add_child(child) 

195 child.add_parent(replacement_node) 

196 for parent in replaced_node.parents: 

197 parent.children.remove(replaced_node) 

198 # Again, to avoid self-referencing. 

199 if parent.key not in replaced: 

200 replacement_node.add_parent(parent) 

201 parent.add_child(replacement_node) 

202 self.clear_cache()

204 def remove_replacement_node(self, replacement, replaced): 

205 """ 

206 The inverse operation to `remove_replaced_nodes`. Almost. Remove the 

207 replacement node `replacement` and remap its child nodes to `replaced` 

208 - the list of nodes it would have replaced. Don't remap its parent 

209 nodes as they are expected to be correct already. 

210 """ 

211 self.nodes.pop(replacement, None) 

212 try: 

213 replacement_node = self.node_map.pop(replacement) 

214 except KeyError as err: 

215 raise NodeNotFoundError( 

216 "Unable to remove replacement node %r. It was either never added" 

217 " to the migration graph, or has been removed already." % (replacement, ), 

218 replacement 

219 ) from err 

220 replaced_nodes = set() 

221 replaced_nodes_parents = set() 

222 for key in replaced: 

223 replaced_node = self.node_map.get(key) 

224 if replaced_node: 

225 replaced_nodes.add(replaced_node) 

226 replaced_nodes_parents |= replaced_node.parents 

227 # We're only interested in the latest replaced node, so filter out 

228 # replaced nodes that are parents of other replaced nodes. 

229 replaced_nodes -= replaced_nodes_parents 

230 for child in replacement_node.children: 

231 child.parents.remove(replacement_node) 

232 for replaced_node in replaced_nodes: 

233 replaced_node.add_child(child) 

234 child.add_parent(replaced_node) 

235 for parent in replacement_node.parents: 

236 parent.children.remove(replacement_node) 

237 # NOTE: There is no need to remap parent dependencies as we can 

238 # assume the replaced nodes already have the correct ancestry. 

239 self.clear_cache()

252 def forwards_plan(self, target): 

253 """ 

254 Given a node, return a list of which previous nodes (dependencies) must 

255 be applied, ending with the node itself. This is the list you would 

256 follow if applying the migrations to a database. 

257 """ 

258 if target not in self.nodes: 

259 raise NodeNotFoundError("Node %r not a valid node" % (target, ), target) 

260 # Use parent.key instead of parent to speed up the frequent hashing in ensure_not_cyclic 

261 self.ensure_not_cyclic(target, lambda x: (parent.key for parent in self.node_map[x].parents)) 

262 self.cached = True

266 except RuntimeError:

272 """ 

273 Given a node, return a list of which dependent nodes (dependencies) 

274 must be unapplied, ending with the node itself. This is the list you 

275 would follow if removing the migrations from a database. 

276 """

278 raise NodeNotFoundError("Node %r not a valid node" % (target, ), target) 

279 # Use child.key instead of child to speed up the frequent hashing in ensure_not_cyclic

284 return node.descendants()

290 def iterative_dfs(self, start, forwards=True): 

291 """Iterative depth-first search for finding dependencies.""" 

292 visited = [] 

293 stack = [start] 

294 while stack: 

295 node = stack.pop() 

296 visited.append(node) 

297 stack += sorted(node.parents if forwards else node.children) 

298 return list(OrderedSet(reversed(visited)))

300 def root_nodes(self, app=None): 

301 """ 

302 Return all root nodes - that is, nodes with no dependencies inside 

303 their app. These are the starting point for an app. 

304 """ 

305 roots = set() 

306 for node in self.nodes: 

307 if not any(key[0] == node[0] for key in self.node_map[node].parents) and (not app or app == node[0]): 

308 roots.add(node) 

309 return sorted(roots)

325 def ensure_not_cyclic(self, start, get_children): 

326 # Algo from GvR: 

327 # http://neopythonic.blogspot.co.uk/2009/01/detecting-cycles-in-directed-graph.html 

328 todo = set(self.nodes) 

329 while todo: 

330 node = todo.pop() 

331 stack = [node] 

332 while stack: 

333 top = stack[-1] 

334 for node in get_children(top): 

335 if node in stack: 

336 cycle = stack[stack.index(node):] 

337 raise CircularDependencyError(", ".join("%s.%s" % n for n in cycle)) 

338 if node in todo: 

339 stack.append(node) 

340 todo.remove(node) 

341 break 

342 else: 

343 node = stack.pop()

355 def make_state(self, nodes=None, at_end=True, real_apps=None): 

356 """ 

357 Given a migration node or nodes, return a complete ProjectState for it. 

358 If at_end is False, return the state before the migration has run. 

359 If nodes is not provided, return the overall most current project state. 

360 """ 

361 if nodes is None: 

362 nodes = list(self.leaf_nodes()) 

363 if len(nodes) == 0: 

364 return ProjectState() 

365 if not isinstance(nodes[0], tuple): 

366 nodes = [nodes] 

367 plan = [] 

368 for node in nodes: 

369 for migration in self.forwards_plan(node): 

370 if migration not in plan: 

371 if not at_end and migration in nodes: 

372 continue 

373 plan.append(migration) 

374 project_state = ProjectState(real_apps=real_apps) 

375 for node in plan: 

376 project_state = self.nodes[node].mutate_state(project_state, preserve=False) 

377 return project_state
