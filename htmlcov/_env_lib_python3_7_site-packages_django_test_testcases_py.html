617 def assertFieldOutput(self, fieldclass, valid, invalid, field_args=None, 

618 field_kwargs=None, empty_value=''): 

619 """ 

620 Assert that a form field behaves correctly with various inputs. 

621 

622 Args: 

623 fieldclass: the class of the field to be tested. 

624 valid: a dictionary mapping valid inputs to their expected 

625 cleaned values. 

626 invalid: a dictionary mapping invalid inputs to one or more 

627 raised error messages. 

628 field_args: the args passed to instantiate the field 

629 field_kwargs: the kwargs passed to instantiate the field 

630 empty_value: the expected clean output for inputs in empty_values 

631 """ 

632 if field_args is None: 

633 field_args = [] 

634 if field_kwargs is None: 

635 field_kwargs = {} 

636 required = fieldclass(*field_args, **field_kwargs) 

637 optional = fieldclass(*field_args, **dict(field_kwargs, required=False)) 

638 # test valid inputs 

639 for input, output in valid.items(): 

640 self.assertEqual(required.clean(input), output) 

641 self.assertEqual(optional.clean(input), output)

1053 """Descriptor class for deferred condition checking.""" 

1054 def __init__(self, *conditions):

1061 # Trigger access for all bases. 

1062 if any(getattr(base, '__unittest_skip__', False) for base in cls.__bases__): 

1063 return True

1080 raise unittest.SkipTest(reason)

1083 else:

1085 test_item = test_func 

1086 # Retrieve the possibly existing value from the class's dict to 

1087 # avoid triggering the descriptor.

1089 if isinstance(skip, CheckCondition): 

1090 test_item.__unittest_skip__ = skip.add_condition(condition, reason)

1255 framework, such as Selenium for example, instead of the built-in dummy 

1256 client. 

1257 It inherits from TransactionTestCase instead of TestCase because the 

1258 threads don't share the same transactions (unless if using in-memory sqlite)

1260 thread can see the changes. 

1261 """

1263 port = 0

1265 static_handler = _StaticFilesHandler

1270 

1271 @classmethod 

1272 def setUpClass(cls): 

1273 super().setUpClass()

1276 # If using in-memory sqlite databases, pass the connections to 

1277 # the server thread. 

1278 if conn.vendor == 'sqlite' and conn.is_in_memory_db(): 

1279 # Explicitly enable thread-shareability for this connection 

1280 conn.allow_thread_sharing = True 

1281 connections_override[conn.alias] = conn 

1282 

1283 cls._live_server_modified_settings = modify_settings( 

1284 ALLOWED_HOSTS={'append': cls.host}, 

1285 )

1328class SerializeMixin:

1331

1333 file must exist on the filesystem. 

1334

1336 """ 

1337 lockfile = None 

1338

1344 "in the base class.".format(cls.__name__))

1347 super().setUpClass() 

1348

1350 def tearDownClass(cls):

1352 cls._lockfile.close()
