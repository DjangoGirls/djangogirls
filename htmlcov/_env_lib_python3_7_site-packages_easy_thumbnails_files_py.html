20def get_thumbnailer(obj, relative_name=None): 

21 """ 

22 Get a :class:`Thumbnailer` for a source file. 

23 

24 The ``obj`` argument is usually either one of the following: 

25 

26 * ``FieldFile`` instance (i.e. a model instance file/image field 

27 property). 

28 

29 * A string, which will be used as the relative name (the source will be 

30 set to the default storage). 

31 

32 * ``Storage`` instance - the ``relative_name`` argument must also be 

33 provided. 

34 

35 Or it could be: 

36 

37 * A file-like instance - the ``relative_name`` argument must also be 

38 provided. 

39 

40 In this case, the thumbnailer won't use or create a cached reference 

41 to the thumbnail (i.e. a new thumbnail will be created for every 

42 :meth:`Thumbnailer.get_thumbnail` call). 

43 

44 If ``obj`` is a ``Thumbnailer`` instance, it will just be returned. If it's 

45 an object with an ``easy_thumbnails_thumbnailer`` then the attribute is 

46 simply returned under the assumption it is a Thumbnailer instance) 

47 """ 

48 if hasattr(obj, 'easy_thumbnails_thumbnailer'): 

49 return obj.easy_thumbnails_thumbnailer 

50 if isinstance(obj, Thumbnailer): 

51 return obj 

52 elif isinstance(obj, FieldFile): 

53 if not relative_name: 

54 relative_name = obj.name 

55 return ThumbnailerFieldFile(obj.instance, obj.field, relative_name) 

56 

57 source_storage = None 

58 

59 if isinstance(obj, six.string_types): 

60 relative_name = obj 

61 obj = None 

62 

63 if not relative_name: 

64 raise ValueError( 

65 "If object is not a FieldFile or Thumbnailer instance, the " 

66 "relative name must be provided") 

67 

68 if isinstance(obj, File): 

69 obj = obj.file 

70 if isinstance(obj, Storage) or obj == default_storage: 

71 source_storage = obj 

72 obj = None 

73 

74 return Thumbnailer( 

75 file=obj, name=relative_name, source_storage=source_storage, 

76 remote_source=obj is not None)

359 def generate_thumbnail(self, thumbnail_options, high_resolution=False, 

360 silent_template_exception=False): 

361 """ 

362 Return an unsaved ``ThumbnailFile`` containing a thumbnail image. 

363 

364 The thumbnail image is generated using the ``thumbnail_options`` 

365 dictionary. 

366 """ 

367 thumbnail_options = self.get_options(thumbnail_options) 

368 orig_size = thumbnail_options['size'] # remember original size 

369 # Size sanity check. 

370 min_dim, max_dim = 0, 0 

371 for dim in orig_size: 

372 try: 

373 dim = int(dim) 

374 except (TypeError, ValueError): 

375 continue 

376 min_dim, max_dim = min(min_dim, dim), max(max_dim, dim) 

377 if max_dim == 0 or min_dim < 0: 

378 raise exceptions.EasyThumbnailsError( 

379 "The source image is an invalid size (%sx%s)" % orig_size) 

380 

381 if high_resolution: 

382 thumbnail_options['size'] = (orig_size[0] * 2, orig_size[1] * 2) 

383 image = engine.generate_source_image( 

384 self, thumbnail_options, self.source_generators, 

385 fail_silently=silent_template_exception) 

386 if image is None: 

387 raise exceptions.InvalidImageFormatError( 

388 "The source file does not appear to be an image") 

389 

390 thumbnail_image = engine.process_image(image, thumbnail_options, 

391 self.thumbnail_processors) 

392 if high_resolution: 

393 thumbnail_options['size'] = orig_size # restore original size 

394 

395 filename = self.get_thumbnail_name( 

396 thumbnail_options, 

397 transparent=utils.is_transparent(thumbnail_image), 

398 high_resolution=high_resolution) 

399 quality = thumbnail_options['quality'] 

400 subsampling = thumbnail_options['subsampling'] 

401 

402 img = engine.save_image( 

403 thumbnail_image, filename=filename, quality=quality, 

404 subsampling=subsampling) 

405 data = img.read() 

406 

407 thumbnail = ThumbnailFile( 

408 filename, file=ContentFile(data), storage=self.thumbnail_storage, 

409 thumbnail_options=thumbnail_options) 

410 thumbnail.image = thumbnail_image 

411 thumbnail._committed = False 

412 

413 return thumbnail

415 def get_thumbnail_name(self, thumbnail_options, transparent=False, 

416 high_resolution=False): 

417 """ 

418 Return a thumbnail filename for the given ``thumbnail_options`` 

419 dictionary and ``source_name`` (which defaults to the File's ``name`` 

420 if not provided). 

421 """ 

422 thumbnail_options = self.get_options(thumbnail_options) 

423 path, source_filename = os.path.split(self.name) 

424 source_extension = os.path.splitext(source_filename)[1][1:] 

425 preserve_extensions = self.thumbnail_preserve_extensions 

426 if preserve_extensions and ( 

427 preserve_extensions is True or 

428 source_extension.lower() in preserve_extensions): 

429 extension = source_extension 

430 elif transparent: 

431 extension = self.thumbnail_transparency_extension 

432 else: 

433 extension = self.thumbnail_extension 

434 extension = extension or 'jpg' 

435 

436 prepared_opts = thumbnail_options.prepared_options() 

437 opts_text = '_'.join(prepared_opts) 

438 

439 data = {'opts': opts_text} 

440 basedir = self.thumbnail_basedir % data 

441 subdir = self.thumbnail_subdir % data 

442

444 namer_func = utils.dynamic_import(self.thumbnail_namer)

448 thumbnailer=self,

452 prepared_options=prepared_opts, 

453 )

490 def get_thumbnail(self, thumbnail_options, save=True, generate=None, 

491 silent_template_exception=False): 

492 """ 

493 Return a ``ThumbnailFile`` containing a thumbnail. 

494 

495 If a matching thumbnail already exists, it will simply be returned. 

496 

497 By default (unless the ``Thumbnailer`` was instanciated with 

498 ``generate=False``), thumbnails that don't exist are generated. 

499 Otherwise ``None`` is returned. 

500 

501 Force the generation behaviour by setting the ``generate`` param to 

502 either ``True`` or ``False`` as required. 

503 

504 The new thumbnail image is generated using the ``thumbnail_options`` 

505 dictionary. If the ``save`` argument is ``True`` (default), the 

506 generated thumbnail will be saved too. 

507 """ 

508 thumbnail_options = self.get_options(thumbnail_options) 

509 if generate is None: 

510 generate = self.generate 

511 

512 thumbnail = self.get_existing_thumbnail(thumbnail_options) 

513 if not thumbnail: 

514 if generate: 

515 thumbnail = self.generate_thumbnail( 

516 thumbnail_options, 

517 silent_template_exception=silent_template_exception) 

518 if save: 

519 self.save_thumbnail(thumbnail) 

520 else: 

521 signals.thumbnail_missed.send( 

522 sender=self, options=thumbnail_options, 

523 high_resolution=False) 

524 

525 if 'HIGH_RESOLUTION' in thumbnail_options: 

526 generate_high_resolution = thumbnail_options.get('HIGH_RESOLUTION') 

527 else: 

528 generate_high_resolution = self.thumbnail_high_resolution 

529 if generate_high_resolution: 

530 thumbnail.high_resolution = self.get_existing_thumbnail( 

531 thumbnail_options, high_resolution=True) 

532 if not thumbnail.high_resolution: 

533 if generate: 

534 thumbnail.high_resolution = self.generate_thumbnail( 

535 thumbnail_options, high_resolution=True, 

536 silent_template_exception=silent_template_exception) 

537 if save: 

538 self.save_thumbnail(thumbnail.high_resolution) 

539 else: 

540 signals.thumbnail_missed.send( 

541 sender=self, options=thumbnail_options, 

542 high_resolution=False) 

543 

544 return thumbnail

577 def thumbnail_exists(self, thumbnail_name): 

578 """ 

579 Calculate whether the thumbnail already exists and that the source is 

580 not newer than the thumbnail. 

581 

582 If the source and thumbnail file storages are local, their file 

583 modification times are used. Otherwise the database cached modification 

584 times are used. 

585 """ 

586 if self.remote_source: 

587 return False 

588 

589 if utils.is_storage_local(self.source_storage): 

590 source_modtime = utils.get_modified_time( 

591 self.source_storage, self.name) 

592 else: 

593 source = self.get_source_cache() 

594 if not source: 

595 return False 

596 source_modtime = source.modified 

597 

598 if not source_modtime: 

599 return False 

600 

601 local_thumbnails = utils.is_storage_local(self.thumbnail_storage) 

602 if local_thumbnails: 

603 thumbnail_modtime = utils.get_modified_time( 

604 self.thumbnail_storage, thumbnail_name) 

605 if not thumbnail_modtime: 

606 return False 

607 return source_modtime <= thumbnail_modtime 

608 

609 thumbnail = self.get_thumbnail_cache(thumbnail_name) 

610 if not thumbnail: 

611 return False 

612 thumbnail_modtime = thumbnail.modified 

613 

614 if thumbnail.modified and source_modtime <= thumbnail.modified: 

615 return thumbnail 

616 return False

688 def delete_thumbnails(self, source_cache=None): 

689 """ 

690 Delete any thumbnails generated from the source image. 

691 

692 :arg source_cache: An optional argument only used for optimisation 

693 where the source cache instance is already known. 

694 :returns: The number of files deleted. 

695 """ 

696 source_cache = self.get_source_cache() 

697 deleted = 0 

698 if source_cache: 

699 thumbnail_storage_hash = utils.get_storage_hash( 

700 self.thumbnail_storage) 

701 for thumbnail_cache in source_cache.thumbnails.all(): 

702 # Only attempt to delete the file if it was stored using the 

703 # same storage as is currently used. 

704 if thumbnail_cache.storage_hash == thumbnail_storage_hash: 

705 self.thumbnail_storage.delete(thumbnail_cache.name) 

706 # Delete the cache thumbnail instance too. 

707 thumbnail_cache.delete() 

708 deleted += 1 

709 return deleted

714 """

716 """ 

717 # First, delete any related thumbnails. 

718 source_cache = self.get_source_cache() 

719 if source_cache: 

720 thumbnail_storage_hash = utils.get_storage_hash( 

721 self.thumbnail_storage) 

722 for thumbnail_cache in source_cache.thumbnails.all(): 

723 # Only iterate files which are stored using the current 

724 # thumbnail storage. 

725 if thumbnail_cache.storage_hash == thumbnail_storage_hash: 

726 yield ThumbnailFile(name=thumbnail_cache.name,
