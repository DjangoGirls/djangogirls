2"""This module contains the parser/generators (or coders/encoders if you 

3prefer) for the classes/datatypes that are used in iCalendar: 

4 

5########################################################################### 

6# This module defines these property value data types and property parameters 

7 

84.2 Defined property parameters are: 

9 

10 ALTREP, CN, CUTYPE, DELEGATED-FROM, DELEGATED-TO, DIR, ENCODING, FMTTYPE, 

11 FBTYPE, LANGUAGE, MEMBER, PARTSTAT, RANGE, RELATED, RELTYPE, ROLE, RSVP, 

12 SENT-BY, TZID, VALUE 

13 

144.3 Defined value data types are: 

15 

16 BINARY, BOOLEAN, CAL-ADDRESS, DATE, DATE-TIME, DURATION, FLOAT, INTEGER, 

17 PERIOD, RECUR, TEXT, TIME, URI, UTC-OFFSET 

18 

19########################################################################### 

20 

21iCalendar properties have values. The values are strongly typed. This module 

22defines these types, calling val.to_ical() on them will render them as defined 

23in rfc2445. 

24 

25If you pass any of these classes a Python primitive, you will have an object 

26that can render itself as iCalendar formatted date. 

27 

28Property Value Data Types start with a 'v'. they all have an to_ical() and 

29from_ical() method. The to_ical() method generates a text string in the 

30iCalendar format. The from_ical() method can parse this format and return a 

31primitive Python datatype. So it should always be true that: 

32 

33 x == vDataType.from_ical(VDataType(x).to_ical()) 

34 

35These types are mainly used for parsing and file generation. But you can set 

36them directly.

378class vDatetime(object): 

379 """Render and generates icalendar datetime format. 

380 

381 vDatetime is timezone aware and uses the pytz library, an implementation of 

382 the Olson database in Python. When a vDatetime object is created from an 

383 ical string, you can pass a valid pytz timezone identifier. When a 

384 vDatetime object is created from a python datetime object, it uses the 

385 tzinfo component, if present. Otherwise an timezone-naive object is 

386 created. Be aware that there are certain limitations with timezone naive 

387 DATE-TIME components in the icalendar standard. 

388 """ 

389 def __init__(self, dt): 

390 self.dt = dt 

391 self.params = Parameters() 

392 

393 def to_ical(self): 

394 dt = self.dt 

395 tzid = tzid_from_dt(dt) 

396 

397 s = "%04d%02d%02dT%02d%02d%02d" % ( 

398 dt.year, 

399 dt.month, 

400 dt.day, 

401 dt.hour, 

402 dt.minute, 

403 dt.second 

404 ) 

405 if tzid == 'UTC': 

406 s += "Z" 

407 elif tzid: 

408 self.params.update({'TZID': tzid}) 

409 return s.encode('utf-8') 

410 

411 @staticmethod 

412 def from_ical(ical, timezone=None): 

413 tzinfo = None 

414 if timezone: 

415 try: 

416 tzinfo = pytz.timezone(timezone) 

417 except pytz.UnknownTimeZoneError: 

418 if timezone in WINDOWS_TO_OLSON: 

419 tzinfo = pytz.timezone(WINDOWS_TO_OLSON.get(timezone)) 

420 else: 

421 tzinfo = _timezone_cache.get(timezone, None) 

422 

423 try: 

424 timetuple = ( 

425 int(ical[:4]), # year 

426 int(ical[4:6]), # month 

427 int(ical[6:8]), # day 

428 int(ical[9:11]), # hour 

429 int(ical[11:13]), # minute 

430 int(ical[13:15]), # second 

431 ) 

432 if tzinfo: 

433 return tzinfo.localize(datetime(*timetuple)) 

434 elif not ical[15:]: 

435 return datetime(*timetuple) 

436 elif ical[15:16] == 'Z': 

437 return pytz.utc.localize(datetime(*timetuple)) 

438 else: 

439 raise ValueError(ical) 

440 except: 

441 raise ValueError('Wrong datetime format: %s' % ical)

444class vDuration(object): 

445 """Subclass of timedelta that renders itself in the iCalendar DURATION 

446 format. 

447 """ 

448 

449 def __init__(self, td): 

450 if not isinstance(td, timedelta): 

451 raise ValueError('Value MUST be a timedelta instance') 

452 self.td = td 

453 self.params = Parameters() 

454 

455 def to_ical(self): 

456 sign = "" 

457 if self.td.days < 0: 

458 sign = "-" 

459 self.td = -self.td 

460 timepart = "" 

461 if self.td.seconds: 

462 timepart = "T" 

463 hours = self.td.seconds // 3600 

464 minutes = self.td.seconds % 3600 // 60 

465 seconds = self.td.seconds % 60 

466 if hours: 

467 timepart += "%dH" % hours 

468 if minutes or (hours and seconds): 

469 timepart += "%dM" % minutes 

470 if seconds: 

471 timepart += "%dS" % seconds 

472 if self.td.days == 0 and timepart: 

473 return (compat.unicode_type(sign).encode('utf-8') + b'P' + 

474 compat.unicode_type(timepart).encode('utf-8')) 

475 else: 

476 return (compat.unicode_type(sign).encode('utf-8') + b'P' + 

477 compat.unicode_type(abs(self.td.days)).encode('utf-8') + 

478 b'D' + compat.unicode_type(timepart).encode('utf-8')) 

479 

480 @staticmethod 

481 def from_ical(ical): 

482 try: 

483 match = DURATION_REGEX.match(ical) 

484 sign, weeks, days, hours, minutes, seconds = match.groups() 

485 if weeks: 

486 value = timedelta(weeks=int(weeks)) 

487 else: 

488 value = timedelta(days=int(days or 0), 

489 hours=int(hours or 0), 

490 minutes=int(minutes or 0), 

491 seconds=int(seconds or 0)) 

492 if sign == '-':

494 return value

500 """A precise period of time. 

501 """ 

502 def __init__(self, per): 

503 start, end_or_duration = per 

504 if not (isinstance(start, datetime) or isinstance(start, date)): 

505 raise ValueError('Start value MUST be a datetime or date instance')

510 'date or timedelta instance')

512 if isinstance(end_or_duration, timedelta): 

513 by_duration = 1 

514 duration = end_or_duration 

515 end = start + duration 

516 else: 

517 end = end_or_duration

520 raise ValueError("Start time is greater than end time")

530 self.end = end

532 self.duration = duration

688 def parse_type(cls, key, values): 

689 # integers 

690 parser = cls.types.get(key, vText) 

691 return [parser.from_ical(v) for v in values.split(',')] 

692 

693 @classmethod 

694 def from_ical(cls, ical): 

695 if isinstance(ical, cls): 

696 return ical 

697 try: 

698 recur = cls() 

699 for pairs in ical.split(';'): 

700 try: 

701 key, vals = pairs.split('=') 

702 except ValueError: 

703 # E.g. incorrect trailing semicolon, like (issue #157): 

704 # FREQ=YEARLY;BYMONTH=11;BYDAY=1SU; 

705 continue 

706 recur[key] = cls.parse_type(key, vals) 

707 return dict(recur) 

708 except: 

709 raise ValueError('Error in recurrence rule: %s' % ical)

810class vUTCOffset(object): 

811 """Renders itself as a utc offset. 

812 """ 

813 

814 ignore_exceptions = False # if True, and we cannot parse this 

815 # component, we will silently ignore 

816 # it, rather than let the exception 

817 # propagate upwards 

818 

819 def __init__(self, td): 

820 if not isinstance(td, timedelta): 

821 raise ValueError('Offset value MUST be a timedelta instance') 

822 self.td = td 

823 self.params = Parameters() 

824 

825 def to_ical(self): 

826 

827 if self.td < timedelta(0): 

828 sign = '-%s' 

829 td = timedelta(0) - self.td # get timedelta relative to 0 

830 else: 

831 # Google Calendar rejects '0000' but accepts '+0000' 

832 sign = '+%s' 

833 td = self.td 

834 

835 days, seconds = td.days, td.seconds 

836 

837 hours = abs(days * 24 + seconds // 3600) 

838 minutes = abs((seconds % 3600) // 60) 

839 seconds = abs(seconds % 60) 

840 if seconds: 

841 duration = '%02i%02i%02i' % (hours, minutes, seconds) 

842 else: 

843 duration = '%02i%02i' % (hours, minutes) 

844 return sign % duration 

845 

846 @classmethod 

847 def from_ical(cls, ical): 

848 if isinstance(ical, cls): 

849 return ical.td 

850 try: 

851 sign, hours, minutes, seconds = (ical[0:1], 

852 int(ical[1:3]), 

853 int(ical[3:5]), 

854 int(ical[5:7] or 0)) 

855 offset = timedelta(hours=hours, minutes=minutes, seconds=seconds) 

856 except: 

857 raise ValueError('Expected utc offset, got: %s' % ical) 

858 if not cls.ignore_exceptions and offset >= timedelta(hours=24): 

859 raise ValueError( 

860 'Offset must be less than 24 hours, was %s' % ical) 

861 if sign == '-': 

862 return -offset 

863 return offset
