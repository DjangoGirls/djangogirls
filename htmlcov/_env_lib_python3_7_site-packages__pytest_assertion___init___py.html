109 def callbinrepr(op, left, right): 

110 """Call the pytest_assertrepr_compare hook and prepare the result 

111 

112 This uses the first result from the hook and then ensures the 

113 following: 

114 * Overly verbose explanations are truncated unless configured otherwise 

115 (eg. if running in verbose mode). 

116 * Embedded newlines are escaped to help util.format_explanation() 

117 later. 

118 * If the rewrite mode is used embedded %-characters are replaced 

119 to protect later % formatting. 

120 

121 The result can be formatted by util.format_explanation() for 

122 pretty printing. 

123 """ 

124 hook_result = item.ihook.pytest_assertrepr_compare( 

125 config=item.config, op=op, left=left, right=right 

126 ) 

127 for new_expl in hook_result: 

128 if new_expl: 

129 new_expl = truncate.truncate_if_required(new_expl, item) 

130 new_expl = [line.replace("\n", "\\n") for line in new_expl] 

131 res = six.text_type("\n~").join(new_expl) 

132 if item.config.getvalue("assertmode") == "rewrite": 

133 res = res.replace("%", "%%") 

134 return res
