33def lookup(code, _cache={}): 

34 """Lookup an error code or class code and return its symbolic name. 

35 

36 Raise `KeyError` if the code is not found. 

37 """ 

38 if _cache: 

39 return _cache[code] 

40 

41 # Generate the lookup map at first usage. 

42 tmp = {} 

43 for k, v in globals().items(): 

44 if isinstance(v, str) and len(v) in (2, 5): 

45 tmp[v] = k 

46 

47 assert tmp 

48 

49 # Atomic update, to avoid race condition on import (bug #382) 

50 _cache.update(tmp) 

51 

52 return _cache[code]
