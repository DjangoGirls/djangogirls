538def self_generate(output_filename, filename="iso3166-1.csv"): # pragma: no cover 

539 """ 

540 The following code can be used for self-generation of this file. 

541 

542 It requires a UTF-8 CSV file containing the short ISO name and two letter 

543 country code as the first two columns. 

544 """ 

545 import csv 

546 import re 

547 

548 countries = [] 

549 alt_codes = [] 

550 with open(filename, "r") as csv_file: 

551 for row in csv.reader(csv_file): 

552 name = row[0].rstrip("*") 

553 name = re.sub(r"\(the\)", "", name) 

554 if name: 

555 countries.append((name, row[1])) 

556 alt_codes.append((row[1], row[2], int(row[3]))) 

557 with open(__file__, "r") as source_file: 

558 contents = source_file.read() 

559 # Write countries. 

560 bits = re.match("(.*\nCOUNTRIES = \{\n)(.*?)(\n\}.*)", contents, re.DOTALL).groups() 

561 country_list = [] 

562 for name, code in countries: 

563 name = name.replace('"', r"\"").strip() 

564 country_list.append(' "{code}": _("{name}"),'.format(name=name, code=code)) 

565 content = bits[0] 

566 content += "\n".join(country_list) 

567 # Write alt codes. 

568 alt_bits = re.match( 

569 "(.*\nALT_CODES = \{\n)(.*)(\n\}.*)", bits[2], re.DOTALL 

570 ).groups() 

571 alt_list = [] 

572 for code, code3, codenum in alt_codes: 

573 name = name.replace('"', r"\"").strip() 

574 alt_list.append( 

575 ' "{code}": ("{code3}", {codenum}),'.format( 

576 code=code, code3=code3, codenum=codenum 

577 ) 

578 ) 

579 content += alt_bits[0] 

580 content += "\n".join(alt_list) 

581 content += alt_bits[2] 

582 # Generate file. 

583 with open(output_filename, "w") as output_file: 

584 output_file.write(content) 

585 return countries

588def check_flags(verbosity=1): 

589 files = {} 

590 this_dir = os.path.dirname(__file__) 

591 for path in glob.glob(os.path.join(this_dir, "static", "flags", "*.gif")): 

592 files[os.path.basename(os.path.splitext(path)[0]).upper()] = path 

593 

594 flags_missing = set(COUNTRIES) - set(files) 

595 if flags_missing: # pragma: no cover 

596 print("The following country codes are missing a flag:") 

597 for code in sorted(flags_missing): 

598 print(" {0} ({1})".format(code, COUNTRIES[code])) 

599 elif verbosity: # pragma: no cover 

600 print("All country codes have flags. :)") 

601 

602 code_missing = set(files) - set(COUNTRIES) 

603 # Special-case EU and __ 

604 for special_code in ("EU", "__"): 

605 code_missing.discard(special_code) 

606 if code_missing: # pragma: no cover 

607 print("") 

608 print("The following flags don't have a matching country code:")
