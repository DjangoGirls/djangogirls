48def _save(im, fp, filename, save_all=False): 

49 is_appending = im.encoderinfo.get("append", False) 

50 if is_appending: 

51 existing_pdf = PdfParser.PdfParser(f=fp, filename=filename, mode="r+b") 

52 else: 

53 existing_pdf = PdfParser.PdfParser(f=fp, filename=filename, mode="w+b") 

54 

55 resolution = im.encoderinfo.get("resolution", 72.0) 

56 

57 info = { 

58 "title": None 

59 if is_appending 

60 else os.path.splitext(os.path.basename(filename))[0], 

61 "author": None, 

62 "subject": None, 

63 "keywords": None, 

64 "creator": None, 

65 "producer": None, 

66 "creationDate": None if is_appending else time.gmtime(), 

67 "modDate": None if is_appending else time.gmtime(), 

68 } 

69 for k, default in info.items(): 

70 v = im.encoderinfo.get(k) if k in im.encoderinfo else default 

71 if v: 

72 existing_pdf.info[k[0].upper() + k[1:]] = v 

73 

74 # 

75 # make sure image data is available 

76 im.load() 

77 

78 existing_pdf.start_writing() 

79 existing_pdf.write_header() 

80 existing_pdf.write_comment(f"created by Pillow {__version__} PDF driver") 

81 

82 # 

83 # pages 

84 ims = [im] 

85 if save_all: 

86 append_images = im.encoderinfo.get("append_images", []) 

87 for append_im in append_images: 

88 append_im.encoderinfo = im.encoderinfo.copy() 

89 ims.append(append_im) 

90 numberOfPages = 0 

91 image_refs = [] 

92 page_refs = [] 

93 contents_refs = [] 

94 for im in ims: 

95 im_numberOfPages = 1 

96 if save_all: 

97 try: 

98 im_numberOfPages = im.n_frames 

99 except AttributeError: 

100 # Image format does not have n_frames. 

101 # It is a single frame image 

102 pass 

103 numberOfPages += im_numberOfPages 

104 for i in range(im_numberOfPages): 

105 image_refs.append(existing_pdf.next_object_id(0)) 

106 page_refs.append(existing_pdf.next_object_id(0)) 

107 contents_refs.append(existing_pdf.next_object_id(0)) 

108 existing_pdf.pages.append(page_refs[-1]) 

109 

110 # 

111 # catalog and list of pages 

112 existing_pdf.write_catalog() 

113 

114 pageNumber = 0 

115 for imSequence in ims: 

116 im_pages = ImageSequence.Iterator(imSequence) if save_all else [imSequence] 

117 for im in im_pages: 

118 # FIXME: Should replace ASCIIHexDecode with RunLengthDecode 

119 # (packbits) or LZWDecode (tiff/lzw compression). Note that 

120 # PDF 1.2 also supports Flatedecode (zip compression). 

121 

122 bits = 8 

123 params = None 

124 decode = None 

125 

126 if im.mode == "1": 

127 filter = "ASCIIHexDecode" 

128 colorspace = PdfParser.PdfName("DeviceGray") 

129 procset = "ImageB" # grayscale 

130 bits = 1 

131 elif im.mode == "L": 

132 filter = "DCTDecode" 

133 # params = f"<< /Predictor 15 /Columns {width-2} >>" 

134 colorspace = PdfParser.PdfName("DeviceGray") 

135 procset = "ImageB" # grayscale 

136 elif im.mode == "P": 

137 filter = "ASCIIHexDecode" 

138 palette = im.im.getpalette("RGB") 

139 colorspace = [ 

140 PdfParser.PdfName("Indexed"), 

141 PdfParser.PdfName("DeviceRGB"), 

142 255, 

143 PdfParser.PdfBinary(palette), 

144 ] 

145 procset = "ImageI" # indexed color 

146 elif im.mode == "RGB": 

147 filter = "DCTDecode" 

148 colorspace = PdfParser.PdfName("DeviceRGB") 

149 procset = "ImageC" # color images 

150 elif im.mode == "CMYK": 

151 filter = "DCTDecode" 

152 colorspace = PdfParser.PdfName("DeviceCMYK") 

153 procset = "ImageC" # color images 

154 decode = [1, 0, 1, 0, 1, 0, 1, 0] 

155 else: 

156 raise ValueError(f"cannot save mode {im.mode}") 

157 

158 # 

159 # image 

160 

161 op = io.BytesIO() 

162 

163 if filter == "ASCIIHexDecode": 

164 if bits == 1: 

165 # FIXME: the hex encoder doesn't support packed 1-bit 

166 # images; do things the hard way... 

167 data = im.tobytes("raw", "1") 

168 im = Image.new("L", im.size) 

169 im.putdata(data) 

170 ImageFile._save(im, op, [("hex", (0, 0) + im.size, 0, im.mode)]) 

171 elif filter == "DCTDecode": 

172 Image.SAVE["JPEG"](im, op, filename) 

173 elif filter == "FlateDecode": 

174 ImageFile._save(im, op, [("zip", (0, 0) + im.size, 0, im.mode)]) 

175 elif filter == "RunLengthDecode": 

176 ImageFile._save(im, op, [("packbits", (0, 0) + im.size, 0, im.mode)]) 

177 else: 

178 raise ValueError(f"unsupported PDF filter ({filter})") 

179 

180 # 

181 # Get image characteristics 

182 

183 width, height = im.size 

184 

185 existing_pdf.write_obj( 

186 image_refs[pageNumber], 

187 stream=op.getvalue(), 

188 Type=PdfParser.PdfName("XObject"), 

189 Subtype=PdfParser.PdfName("Image"), 

190 Width=width, # * 72.0 / resolution, 

191 Height=height, # * 72.0 / resolution, 

192 Filter=PdfParser.PdfName(filter), 

193 BitsPerComponent=bits, 

194 Decode=decode, 

195 DecodeParams=params, 

196 ColorSpace=colorspace, 

197 ) 

198 

199 # 

200 # page 

201 

202 existing_pdf.write_page( 

203 page_refs[pageNumber], 

204 Resources=PdfParser.PdfDict( 

205 ProcSet=[PdfParser.PdfName("PDF"), PdfParser.PdfName(procset)], 

206 XObject=PdfParser.PdfDict(image=image_refs[pageNumber]), 

207 ), 

208 MediaBox=[ 

209 0, 

210 0, 

211 int(width * 72.0 / resolution), 

212 int(height * 72.0 / resolution), 

213 ], 

214 Contents=contents_refs[pageNumber], 

215 ) 

216 

217 # 

218 # page contents 

219 

220 page_contents = b"q %d 0 0 %d 0 0 cm /image Do Q\n" % ( 

221 int(width * 72.0 / resolution), 

222 int(height * 72.0 / resolution), 

223 ) 

224 

225 existing_pdf.write_obj(contents_refs[pageNumber], stream=page_contents) 

226 

227 pageNumber += 1 

228 

229 # 

230 # trailer 

231 existing_pdf.write_xref_and_trailer() 

232 if hasattr(fp, "flush"): 

233 fp.flush() 

234 existing_pdf.close()
