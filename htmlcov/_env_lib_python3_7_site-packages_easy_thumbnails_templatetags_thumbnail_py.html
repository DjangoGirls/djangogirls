24def split_args(args): 

25 """ 

26 Split a list of argument strings into a dictionary where each key is an 

27 argument name. 

28 

29 An argument looks like ``crop``, ``crop="some option"`` or ``crop=my_var``. 

30 Arguments which provide no value get a value of ``True``. 

31 """ 

32 args_dict = {} 

33 for arg in args: 

34 split_arg = arg.split('=', 1) 

35 if len(split_arg) > 1: 

36 value = split_arg[1] 

37 else: 

38 value = True 

39 args_dict[split_arg[0]] = value 

40 return args_dict

43class ThumbnailNode(Node): 

44 def __init__(self, source_var, opts, context_name=None): 

45 self.source_var = source_var 

46 self.opts = opts 

47 self.context_name = context_name 

48 

49 def render(self, context): 

50 # Note that this isn't a global constant because we need to change the 

51 # value for tests. 

52 raise_errors = settings.THUMBNAIL_DEBUG 

53 # Get the source file. 

54 try: 

55 source = self.source_var.resolve(context) 

56 except VariableDoesNotExist: 

57 if raise_errors: 

58 raise VariableDoesNotExist( 

59 "Variable '%s' does not exist." % self.source_var) 

60 return self.bail_out(context) 

61 if not source: 

62 if raise_errors: 

63 raise TemplateSyntaxError( 

64 "Variable '%s' is an invalid source." % self.source_var) 

65 return self.bail_out(context) 

66 # Resolve the thumbnail option values. 

67 try: 

68 opts = {} 

69 for key, value in six.iteritems(self.opts): 

70 if hasattr(value, 'resolve'): 

71 value = value.resolve(context) 

72 opts[str(key)] = value 

73 except Exception: 

74 if raise_errors: 

75 raise 

76 return self.bail_out(context) 

77 # Size variable can be either a tuple/list of two integers or a 

78 # valid string. 

79 size = opts['size'] 

80 if isinstance(size, six.string_types): 

81 m = RE_SIZE.match(size) 

82 if m: 

83 opts['size'] = (int(m.group(1)), int(m.group(2))) 

84 else: 

85 # Size variable may alternatively be referencing an alias. 

86 alias = aliases.get(size, target=source) 

87 if alias: 

88 del opts['size'] 

89 opts = dict(alias, **opts) 

90 else: 

91 if raise_errors: 

92 raise TemplateSyntaxError( 

93 "%r is not a valid size." % size) 

94 return self.bail_out(context) 

95 # Ensure the quality is an integer. 

96 if 'quality' in opts: 

97 try: 

98 opts['quality'] = int(opts['quality']) 

99 except (TypeError, ValueError): 

100 if raise_errors: 

101 raise TemplateSyntaxError( 

102 "%r is an invalid quality." % opts['quality']) 

103 return self.bail_out(context) 

104 # Ensure the subsampling level is an integer. 

105 if 'subsampling' in opts: 

106 try: 

107 opts['subsampling'] = int(opts['subsampling']) 

108 except (TypeError, ValueError): 

109 if raise_errors: 

110 raise TemplateSyntaxError( 

111 "%r is an invalid subsampling level." % 

112 opts['subsampling']) 

113 return self.bail_out(context) 

114 

115 try: 

116 thumbnail = get_thumbnailer(source).get_thumbnail(opts) 

117 except Exception: 

118 if raise_errors: 

119 raise 

120 return self.bail_out(context) 

121 # Return the thumbnail file url, or put the file on the context. 

122 if self.context_name is None: 

123 return escape(thumbnail.url) 

124 else: 

125 context[self.context_name] = thumbnail 

126 return '' 

127 

128 def bail_out(self, context): 

129 if self.context_name: 

130 context[self.context_name] = '' 

131 return ''

134@register.tag 

135def thumbnail(parser, token): 

136 """ 

137 Creates a thumbnail of an ImageField. 

138 

139 Basic tag Syntax:: 

140 

141 {% thumbnail [source] [size] [options] %} 

142 

143 *source* must be a ``File`` object, usually an Image/FileField of a model 

144 instance. 

145 

146 *size* can either be: 

147 

148 * the name of an alias 

149 

150 * the size in the format ``[width]x[height]`` (for example, 

151 ``{% thumbnail person.photo 100x50 %}``) or 

152 

153 * a variable containing a valid size (i.e. either a string in the 

154 ``[width]x[height]`` format or a tuple containing two integers): 

155 ``{% thumbnail person.photo size_var %}``. 

156 

157 *options* are a space separated list of options which are used when 

158 processing the image to a thumbnail such as ``sharpen``, ``crop`` and 

159 ``quality=90``. 

160 

161 If *size* is specified as an alias name, *options* are used to override 

162 and/or supplement the options defined in that alias. 

163 

164 The thumbnail tag can also place a 

165 :class:`~easy_thumbnails.files.ThumbnailFile` object in the context, 

166 providing access to the properties of the thumbnail such as the height and 

167 width:: 

168 

169 {% thumbnail [source] [size] [options] as [variable] %} 

170 

171 When ``as [variable]`` is used, the tag doesn't output anything. Instead, 

172 use the variable like a standard ``ImageFieldFile`` object:: 

173 

174 {% thumbnail obj.picture 200x200 upscale as thumb %} 

175 <img src="{{ thumb.url }}" 

176 width="{{ thumb.width }}" 

177 height="{{ thumb.height }}" /> 

178 

179 **Debugging** 

180 

181 By default, if there is an error creating the thumbnail or resolving the 

182 image variable then the thumbnail tag will just return an empty string (and 

183 if there was a context variable to be set then it will also be set to an 

184 empty string). 

185 

186 For example, you will not see an error if the thumbnail could not 

187 be written to directory because of permissions error. To display those 

188 errors rather than failing silently, set ``THUMBNAIL_DEBUG = True`` in 

189 your Django project's settings module. 

190 

191 """ 

192 args = token.split_contents() 

193 tag = args[0] 

194 

195 # Check to see if we're setting to a context variable. 

196 if len(args) > 4 and args[-2] == 'as': 

197 context_name = args[-1] 

198 args = args[:-2] 

199 else: 

200 context_name = None 

201 

202 if len(args) < 3: 

203 raise TemplateSyntaxError( 

204 "Invalid syntax. Expected " 

205 "'{%% %s source size [option1 option2 ...] %%}' or " 

206 "'{%% %s source size [option1 option2 ...] as variable %%}'" % 

207 (tag, tag)) 

208 

209 opts = {} 

210 

211 # The first argument is the source file. 

212 source_var = parser.compile_filter(args[1]) 

213 

214 # The second argument is the requested size. If it's the static "10x10" 

215 # format, wrap it in quotes so that it is compiled correctly. 

216 size = args[2] 

217 match = RE_SIZE.match(size) 

218 if match: 

219 size = '"%s"' % size 

220 opts['size'] = parser.compile_filter(size) 

221 

222 # All further arguments are options. 

223 args_list = split_args(args[3:]).items() 

224 for arg, value in args_list: 

225 if arg in VALID_OPTIONS: 

226 if value and value is not True: 

227 value = parser.compile_filter(value) 

228 opts[arg] = value 

229 else: 

230 raise TemplateSyntaxError("'%s' tag received a bad argument: " 

231 "'%s'" % (tag, arg)) 

232 return ThumbnailNode(source_var, opts=opts, context_name=context_name)

306def data_uri(thumbnail): 

307 """ 

308 This filter will return the base64 encoded data URI for a given thumbnail object. 

309 

310 Example usage:: 

311 

312 {% thumbnail sample_image 25x25 crop as thumb %} 

313 <img src="{{ thumb|data_uri }}"> 

314 

315 will for instance be rendered as: 

316 

317 <img src="data:image/png;base64,iVBORw0KGgo..."> 

318 """ 

319 try: 

320 thumbnail.open('rb') 

321 data = thumbnail.read() 

322 finally: 

323 thumbnail.close() 

324 mime_type = mimetypes.guess_type(str(thumbnail.file))[0] or 'application/octet-stream' 

325 data = b64encode(data).decode('utf-8') 

326 return 'data:{0};base64,{1}'.format(mime_type, data)
