54 def get_table_list(self, cursor):

56 cursor.execute("""

60 WHERE c.relkind IN ('r', 'v')

67 def get_table_description(self, cursor, table_name): 

68 """ 

69 Return a description of the table with the DB-API cursor.description 

70 interface. 

71 """ 

72 # As cursor.description does not return reliably the nullable property, 

73 # we have to query the information_schema (#7783) 

74 cursor.execute(""" 

75 SELECT column_name, is_nullable, column_default 

76 FROM information_schema.columns 

77 WHERE table_name = %s""", [table_name]) 

78 field_map = {line[0]: line[1:] for line in cursor.fetchall()} 

79 cursor.execute("SELECT * FROM %s LIMIT 1" % self.connection.ops.quote_name(table_name)) 

80 return [ 

81 FieldInfo(*(line[0:6] + (field_map[line.name][0] == 'YES', field_map[line.name][1]))) 

82 for line in cursor.description 

83 ]

141 def get_indexes(self, cursor, table_name): 

142 warnings.warn( 

143 "get_indexes() is deprecated in favor of get_constraints().", 

144 RemovedInDjango21Warning, stacklevel=2 

145 ) 

146 # This query retrieves each index on the given table, including the 

147 # first associated field name 

148 cursor.execute(self._get_indexes_query, [table_name]) 

149 indexes = {} 

150 for row in cursor.fetchall(): 

151 # row[1] (idx.indkey) is stored in the DB as an array. It comes out as 

152 # a string of space-separated integers. This designates the field 

153 # indexes (1-based) of the fields that have indexes on the table. 

154 # Here, we skip any indexes across multiple fields. 

155 if ' ' in row[1]: 

156 continue 

157 if row[0] not in indexes: 

158 indexes[row[0]] = {'primary_key': False, 'unique': False} 

159 # It's possible to have the unique and PK constraints in separate indexes. 

160 if row[3]: 

161 indexes[row[0]]['primary_key'] = True 

162 if row[2]: 

163 indexes[row[0]]['unique'] = True 

164 return indexes

166 def get_constraints(self, cursor, table_name): 

167 """ 

168 Retrieve any constraints or keys (unique, pk, fk, check, index) across 

169 one or more columns. Also retrieve the definition of expression-based 

170 indexes. 

171 """ 

172 constraints = {} 

173 # Loop over the key table, collecting things as constraints. The column 

174 # array must return column names in the same order in which they were 

175 # created. 

176 # The subquery containing generate_series can be replaced with 

177 # "WITH ORDINALITY" when support for PostgreSQL 9.3 is dropped. 

178 cursor.execute(""" 

179 SELECT 

180 c.conname, 

181 array( 

182 SELECT attname 

183 FROM ( 

184 SELECT unnest(c.conkey) AS colid, 

185 generate_series(1, array_length(c.conkey, 1)) AS arridx 

186 ) AS cols 

187 JOIN pg_attribute AS ca ON cols.colid = ca.attnum 

188 WHERE ca.attrelid = c.conrelid 

189 ORDER BY cols.arridx 

190 ), 

191 c.contype, 

192 (SELECT fkc.relname || '.' || fka.attname 

193 FROM pg_attribute AS fka 

194 JOIN pg_class AS fkc ON fka.attrelid = fkc.oid 

195 WHERE fka.attrelid = c.confrelid AND fka.attnum = c.confkey[1]), 

196 cl.reloptions 

197 FROM pg_constraint AS c 

198 JOIN pg_class AS cl ON c.conrelid = cl.oid 

199 JOIN pg_namespace AS ns ON cl.relnamespace = ns.oid 

200 WHERE ns.nspname = %s AND cl.relname = %s 

201 """, ["public", table_name]) 

202 for constraint, columns, kind, used_cols, options in cursor.fetchall(): 

203 constraints[constraint] = { 

204 "columns": columns, 

205 "primary_key": kind == "p", 

206 "unique": kind in ["p", "u"], 

207 "foreign_key": tuple(used_cols.split(".", 1)) if kind == "f" else None, 

208 "check": kind == "c", 

209 "index": False, 

210 "definition": None, 

211 "options": options, 

212 } 

213 # Now get indexes 

214 # The row_number() function for ordering the index fields can be 

215 # replaced by WITH ORDINALITY in the unnest() functions when support 

216 # for PostgreSQL 9.3 is dropped. 

217 cursor.execute(""" 

218 SELECT 

219 indexname, array_agg(attname ORDER BY rnum), indisunique, indisprimary, 

220 array_agg(ordering ORDER BY rnum), amname, exprdef, s2.attoptions 

221 FROM ( 

222 SELECT 

223 row_number() OVER () as rnum, c2.relname as indexname, 

224 idx.*, attr.attname, am.amname, 

225 CASE 

226 WHEN idx.indexprs IS NOT NULL THEN 

227 pg_get_indexdef(idx.indexrelid) 

228 END AS exprdef, 

229 CASE am.amname 

230 WHEN 'btree' THEN 

231 CASE (option & 1) 

232 WHEN 1 THEN 'DESC' ELSE 'ASC' 

233 END 

234 END as ordering, 

235 c2.reloptions as attoptions 

236 FROM ( 

237 SELECT 

238 *, unnest(i.indkey) as key, unnest(i.indoption) as option 

239 FROM pg_index i 

240 ) idx 

241 LEFT JOIN pg_class c ON idx.indrelid = c.oid 

242 LEFT JOIN pg_class c2 ON idx.indexrelid = c2.oid 

243 LEFT JOIN pg_am am ON c2.relam = am.oid 

244 LEFT JOIN pg_attribute attr ON attr.attrelid = c.oid AND attr.attnum = idx.key 

245 WHERE c.relname = %s 

246 ) s2 

247 GROUP BY indexname, indisunique, indisprimary, amname, exprdef, attoptions; 

248 """, [table_name]) 

249 for index, columns, unique, primary, orders, type_, definition, options in cursor.fetchall(): 

250 if index not in constraints: 

251 constraints[index] = { 

252 "columns": columns if columns != [None] else [], 

253 "orders": orders if orders != [None] else [], 

254 "primary_key": primary, 

255 "unique": unique, 

256 "foreign_key": None, 

257 "check": False, 

258 "index": True, 

259 "type": Index.suffix if type_ == 'btree' else type_, 

260 "definition": definition, 

261 "options": options, 

262 } 

263 return constraints
