82def gen_filenames(only_new=False): 

83 """ 

84 Return a list of filenames referenced in sys.modules and translation files. 

85 """ 

86 # N.B. ``list(...)`` is needed, because this runs in parallel with 

87 # application code which might be mutating ``sys.modules``, and this will 

88 # fail with RuntimeError: cannot mutate dictionary while iterating 

89 global _cached_modules, _cached_filenames 

90 module_values = set(sys.modules.values()) 

91 _cached_filenames = clean_files(_cached_filenames) 

92 if _cached_modules == module_values: 

93 # No changes in module list, short-circuit the function 

94 if only_new: 

95 return [] 

96 else: 

97 return _cached_filenames + clean_files(_error_files) 

98 

99 new_modules = module_values - _cached_modules 

100 new_filenames = clean_files( 

101 [filename.__file__ for filename in new_modules 

102 if hasattr(filename, '__file__')]) 

103 

104 if not _cached_filenames and settings.USE_I18N: 

105 # Add the names of the .mo files that can be generated 

106 # by compilemessages management command to the list of files watched. 

107 basedirs = [os.path.join(os.path.dirname(os.path.dirname(__file__)), 

108 'conf', 'locale'), 

109 'locale'] 

110 for app_config in reversed(list(apps.get_app_configs())): 

111 basedirs.append(os.path.join(app_config.path, 'locale')) 

112 basedirs.extend(settings.LOCALE_PATHS) 

113 basedirs = [os.path.abspath(basedir) for basedir in basedirs 

114 if os.path.isdir(basedir)] 

115 for basedir in basedirs: 

116 for dirpath, dirnames, locale_filenames in os.walk(basedir): 

117 for filename in locale_filenames: 

118 if filename.endswith('.mo'): 

119 new_filenames.append(os.path.join(dirpath, filename)) 

120 

121 _cached_modules = _cached_modules.union(new_modules) 

122 _cached_filenames += new_filenames 

123 if only_new: 

124 return new_filenames + clean_files(_error_files) 

125 else: 

126 return _cached_filenames + clean_files(_error_files)

129def clean_files(filelist): 

130 filenames = [] 

131 for filename in filelist: 

132 if not filename: 

133 continue 

134 if filename.endswith(".pyc") or filename.endswith(".pyo"): 

135 filename = filename[:-1] 

136 if filename.endswith("$py.class"): 

137 filename = filename[:-9] + ".py" 

138 if os.path.exists(filename): 

139 filenames.append(filename) 

140 return filenames

152def inotify_code_changed(): 

153 """ 

154 Check for changed code using inotify. After being called 

155 it blocks until a change event has been fired. 

156 """ 

157 class EventHandler(pyinotify.ProcessEvent): 

158 modified_code = None 

159 

160 def process_default(self, event): 

161 if event.path.endswith('.mo'): 

162 EventHandler.modified_code = I18N_MODIFIED 

163 else: 

164 EventHandler.modified_code = FILE_MODIFIED 

165 

166 wm = pyinotify.WatchManager() 

167 notifier = pyinotify.Notifier(wm, EventHandler()) 

168 

169 def update_watch(sender=None, **kwargs): 

170 if sender and getattr(sender, 'handles_files', False): 

171 # No need to update watches when request serves files. 

172 # (sender is supposed to be a django.core.handlers.BaseHandler subclass) 

173 return 

174 mask = ( 

175 pyinotify.IN_MODIFY | 

176 pyinotify.IN_DELETE | 

177 pyinotify.IN_ATTRIB | 

178 pyinotify.IN_MOVED_FROM | 

179 pyinotify.IN_MOVED_TO | 

180 pyinotify.IN_CREATE | 

181 pyinotify.IN_DELETE_SELF | 

182 pyinotify.IN_MOVE_SELF 

183 ) 

184 for path in gen_filenames(only_new=True): 

185 wm.add_watch(path, mask) 

186 

187 # New modules may get imported when a request is processed. 

188 request_finished.connect(update_watch) 

189 

190 # Block until an event happens. 

191 update_watch() 

192 notifier.check_events(timeout=None) 

193 notifier.read_events() 

194 notifier.process_events() 

195 notifier.stop() 

196 

197 # If we are here the code must have changed. 

198 return EventHandler.modified_code

201def code_changed(): 

202 global _mtimes, _win 

203 for filename in gen_filenames(): 

204 stat = os.stat(filename) 

205 mtime = stat.st_mtime 

206 if _win: 

207 mtime -= stat.st_ctime 

208 if filename not in _mtimes: 

209 _mtimes[filename] = mtime 

210 continue 

211 if mtime != _mtimes[filename]: 

212 _mtimes = {} 

213 try: 

214 del _error_files[_error_files.index(filename)] 

215 except ValueError: 

216 pass 

217 return I18N_MODIFIED if filename.endswith('.mo') else FILE_MODIFIED 

218 return False

221def check_errors(fn): 

222 def wrapper(*args, **kwargs): 

223 global _exception 

224 try: 

225 fn(*args, **kwargs) 

226 except Exception: 

227 _exception = sys.exc_info() 

228 

229 et, ev, tb = _exception 

230 

231 if getattr(ev, 'filename', None) is None: 

232 # get the filename from the last item in the stack 

233 filename = traceback.extract_tb(tb)[-1][0] 

234 else: 

235 filename = ev.filename 

236 

237 if filename not in _error_files: 

238 _error_files.append(filename) 

239 

240 raise 

241 

242 return wrapper

251def ensure_echo_on(): 

252 if termios: 

253 fd = sys.stdin 

254 if fd.isatty(): 

255 attr_list = termios.tcgetattr(fd) 

256 if not attr_list[3] & termios.ECHO: 

257 attr_list[3] |= termios.ECHO 

258 if hasattr(signal, 'SIGTTOU'): 

259 old_handler = signal.signal(signal.SIGTTOU, signal.SIG_IGN) 

260 else: 

261 old_handler = None 

262 termios.tcsetattr(fd, termios.TCSANOW, attr_list) 

263 if old_handler is not None: 

264 signal.signal(signal.SIGTTOU, old_handler)

267def reloader_thread(): 

268 ensure_echo_on() 

269 if USE_INOTIFY: 

270 fn = inotify_code_changed 

271 else: 

272 fn = code_changed 

273 while RUN_RELOADER: 

274 change = fn() 

275 if change == FILE_MODIFIED: 

276 sys.exit(3) # force reload 

277 elif change == I18N_MODIFIED: 

278 reset_translations() 

279 time.sleep(1)

282def restart_with_reloader(): 

283 while True: 

284 args = [sys.executable] + ['-W%s' % o for o in sys.warnoptions] + sys.argv 

285 new_environ = os.environ.copy() 

286 new_environ["RUN_MAIN"] = 'true' 

287 exit_code = subprocess.call(args, env=new_environ) 

288 if exit_code != 3: 

289 return exit_code

292def python_reloader(main_func, args, kwargs): 

293 if os.environ.get("RUN_MAIN") == "true": 

294 _thread.start_new_thread(main_func, args, kwargs) 

295 try: 

296 reloader_thread() 

297 except KeyboardInterrupt: 

298 pass 

299 else: 

300 try: 

301 exit_code = restart_with_reloader() 

302 if exit_code < 0: 

303 os.kill(os.getpid(), -exit_code) 

304 else: 

305 sys.exit(exit_code) 

306 except KeyboardInterrupt: 

307 pass
