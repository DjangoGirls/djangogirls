90 """Adds a user-agent to the headers.

92 Args:

110def clean_headers(headers): 

111 """Forces header keys and values to be strings, i.e not unicode. 

112 

113 The httplib module just concats the header keys and values in a way that 

114 may make the message header a unicode string, which, if it then tries to 

115 contatenate to a binary request body may result in a unicode decode error. 

116 

117 Args: 

118 headers: dict, A dictionary of headers. 

119 

120 Returns: 

121 The same dictionary but with all the keys converted to strings. 

122 """ 

123 clean = {} 

124 try: 

125 for k, v in six.iteritems(headers): 

126 if not isinstance(k, six.binary_type): 

127 k = str(k) 

128 if not isinstance(v, six.binary_type): 

129 v = str(v) 

130 clean[_helpers._to_bytes(k)] = _helpers._to_bytes(v) 

131 except UnicodeEncodeError: 

132 from oauth2client.client import NonAsciiHeaderError 

133 raise NonAsciiHeaderError(k, ': ', v) 

134 return clean

137def wrap_http_for_auth(credentials, http): 

138 """Prepares an HTTP object's request method for auth. 

139 

140 Wraps HTTP requests with logic to catch auth failures (typically 

141 identified via a 401 status code). In the event of failure, tries 

142 to refresh the token used and then retry the original request. 

143 

144 Args: 

145 credentials: Credentials, the credentials used to identify 

146 the authenticated user. 

147 http: httplib2.Http, an http object to be used to make 

148 auth requests. 

149 """ 

150 orig_request_method = http.request 

151 

152 # The closure that will replace 'httplib2.Http.request'. 

153 def new_request(uri, method='GET', body=None, headers=None, 

154 redirections=httplib2.DEFAULT_MAX_REDIRECTS, 

155 connection_type=None): 

156 if not credentials.access_token: 

157 _LOGGER.info('Attempting refresh to obtain ' 

158 'initial access_token') 

159 credentials._refresh(orig_request_method) 

160 

161 # Clone and modify the request headers to add the appropriate 

162 # Authorization header. 

163 headers = _initialize_headers(headers) 

164 credentials.apply(headers) 

165 _apply_user_agent(headers, credentials.user_agent) 

166 

167 body_stream_position = None 

168 # Check if the body is a file-like stream. 

169 if all(getattr(body, stream_prop, None) for stream_prop in 

170 _STREAM_PROPERTIES): 

171 body_stream_position = body.tell() 

172 

173 resp, content = request(orig_request_method, uri, method, body, 

174 clean_headers(headers), 

175 redirections, connection_type) 

176 

177 # A stored token may expire between the time it is retrieved and 

178 # the time the request is made, so we may need to try twice. 

179 max_refresh_attempts = 2 

180 for refresh_attempt in range(max_refresh_attempts): 

181 if resp.status not in REFRESH_STATUS_CODES: 

182 break 

183 _LOGGER.info('Refreshing due to a %s (attempt %s/%s)', 

184 resp.status, refresh_attempt + 1, 

185 max_refresh_attempts) 

186 credentials._refresh(orig_request_method) 

187 credentials.apply(headers) 

188 if body_stream_position is not None: 

189 body.seek(body_stream_position) 

190 

191 resp, content = request(orig_request_method, uri, method, body, 

192 clean_headers(headers), 

193 redirections, connection_type) 

194 

195 return resp, content 

196 

197 # Replace the request method with our own closure. 

198 http.request = new_request 

199 

200 # Set credentials as a property of the request method. 

201 http.request.credentials = credentials

204def wrap_http_for_jwt_access(credentials, http): 

205 """Prepares an HTTP object's request method for JWT access. 

206 

207 Wraps HTTP requests with logic to catch auth failures (typically 

208 identified via a 401 status code). In the event of failure, tries 

209 to refresh the token used and then retry the original request. 

210 

211 Args: 

212 credentials: _JWTAccessCredentials, the credentials used to identify 

213 a service account that uses JWT access tokens. 

214 http: httplib2.Http, an http object to be used to make 

215 auth requests. 

216 """ 

217 orig_request_method = http.request 

218 wrap_http_for_auth(credentials, http) 

219 # The new value of ``http.request`` set by ``wrap_http_for_auth``. 

220 authenticated_request_method = http.request 

221 

222 # The closure that will replace 'httplib2.Http.request'. 

223 def new_request(uri, method='GET', body=None, headers=None, 

224 redirections=httplib2.DEFAULT_MAX_REDIRECTS, 

225 connection_type=None): 

226 if 'aud' in credentials._kwargs: 

227 # Preemptively refresh token, this is not done for OAuth2 

228 if (credentials.access_token is None or 

229 credentials.access_token_expired): 

230 credentials.refresh(None) 

231 return request(authenticated_request_method, uri, 

232 method, body, headers, redirections, 

233 connection_type) 

234 else: 

235 # If we don't have an 'aud' (audience) claim, 

236 # create a 1-time token with the uri root as the audience 

237 headers = _initialize_headers(headers) 

238 _apply_user_agent(headers, credentials.user_agent) 

239 uri_root = uri.split('?', 1)[0] 

240 token, unused_expiry = credentials._create_token({'aud': uri_root}) 

241 

242 headers['Authorization'] = 'Bearer ' + token 

243 return request(orig_request_method, uri, method, body, 

244 clean_headers(headers), 

245 redirections, connection_type) 

246 

247 # Replace the request method with our own closure. 

248 http.request = new_request 

249 

250 # Set credentials as a property of the request method. 

251 http.request.credentials = credentials

254def request(http, uri, method='GET', body=None, headers=None, 

255 redirections=httplib2.DEFAULT_MAX_REDIRECTS, 

256 connection_type=None): 

257 """Make an HTTP request with an HTTP object and arguments. 

258 

259 Args: 

260 http: httplib2.Http, an http object to be used to make requests. 

261 uri: string, The URI to be requested. 

262 method: string, The HTTP method to use for the request. Defaults 

263 to 'GET'. 

264 body: string, The payload / body in HTTP request. By default 

265 there is no payload. 

266 headers: dict, Key-value pairs of request headers. By default 

267 there are no headers. 

268 redirections: int, The number of allowed 203 redirects for 

269 the request. Defaults to 5. 

270 connection_type: httplib.HTTPConnection, a subclass to be used for 

271 establishing connection. If not set, the type 

272 will be determined from the ``uri``. 

273 

274 Returns: 

275 tuple, a pair of a httplib2.Response with the status code and other 

276 headers and the bytes of the content returned. 

277 """ 

278 # NOTE: Allowing http or http.request is temporary (See Issue 601). 

279 http_callable = getattr(http, 'request', http) 

280 return http_callable(uri, method=method, body=body, headers=headers, 

281 redirections=redirections, 

282 connection_type=connection_type)
