2 

3This is run via a scheduled task every hour and checks if there are any emails that need sending.

20def send_event_emails( 

21 events, subject_template, plain_template, html_template, timestamp_field, email_type, 

22 ignore_approximate_events=False): 

23 """Send out any that need sending (thank yous, information request, ...).""" 

24 

25 for event in events: 

26 # Some emails can only be sent if the event has a proper date set, ignore approximate dates in those cases. 

27 if ignore_approximate_events and event.date_is_approximate: 

28 continue 

29 

30 recipients = list(set([event.email] + list(event.team.all().values_list('email', flat=True)))) 

31 context = { 

32 'event': event, 

33 'settings': settings, 

34 } 

35 html_content = render_to_string(html_template, context) 

36 plain_content = render_to_string(plain_template, context) 

37 subject_content = render_to_string(subject_template, context) 

38 

39 try: 

40 send_mail( 

41 subject=subject_content, 

42 message=plain_content, 

43 from_email=settings.DEFAULT_FROM_EMAIL, 

44 recipient_list=recipients, 

45 html_message=html_content 

46 ) 

47 except SMTPException: 

48 logging.exception("Couldn't send {} email to {}".format(email_type, "".join(recipients))) 

49 else: 

50 setattr(event, timestamp_field, timezone.now()) 

51 event.save(update_fields=[timestamp_field])

54def send_thank_you_emails(): 

55 """ Send thank you emails a day after the event happened. """ 

56 send_event_emails( 

57 events=Event.objects.filter( 

58 is_on_homepage=True, 

59 date__lte=timezone.now() - timezone.timedelta(days=1),

71def send_submit_information_emails(): 

72 """ Send "please provide numbers" emails.

92def send_offer_help_emails(): 

93 """ Send "do you need help?" emails to events. 

94 

95 This sends emails to events that are in the future but less than 6 weeks away. 

96 Because of dates being approximate, we sometimes can't tell if an event in the current month 

97 has happened or not on the db-level. So we need to catch it while it's more than a month away. 

98 """ 

99 events = Event.objects.filter( 

100 is_on_homepage=True, 

101 date__gt=timezone.now(), 

102 date__lte=timezone.now() + timezone.timedelta(weeks=6), 

103 created_at__lte=timezone.now() - timezone.timedelta(weeks=2), 

104 offer_help_email_sent__isnull=True 

105 ).filter( 

106 Q(is_page_live=False) | Q(form=None) 

107 ) 

108 send_event_emails( 

109 events=events, 

110 subject_template="emails/offer_help_subject.txt", 

111 plain_template="emails/offer_help_email.txt", 

112 html_template="emails/offer_help_email.html", 

113 timestamp_field='offer_help_email_sent', 

114 email_type="offer help email", 

115 ) 

116 

117 if events: 

118 send_mail( 

119 subject="Check-in email summary", 

120 message="Hi there!\n Check-in emails were automatically send to events with no live website " 

121 "or application form open. If the organizers don't contact you back soon, you should " 

122 "remove those events from the events page:\n" + "\n".join(e.city for e in events), 

123 from_email=settings.DEFAULT_FROM_EMAIL, 

124 recipient_list=["hello@djangogirls.org"] 

125 ) 

126 

127

129def command():
