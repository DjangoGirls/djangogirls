154 def load(self): 

155 """Load image data based on tile list""" 

156 

157 if self.tile is None: 

158 raise OSError("cannot load this image") 

159 

160 pixel = Image.Image.load(self) 

161 if not self.tile: 

162 return pixel 

163 

164 self.map = None 

165 use_mmap = self.filename and len(self.tile) == 1 

166 # As of pypy 2.1.0, memory mapping was failing here. 

167 use_mmap = use_mmap and not hasattr(sys, "pypy_version_info") 

168 

169 readonly = 0 

170 

171 # look for read/seek overrides 

172 try: 

173 read = self.load_read 

174 # don't use mmap if there are custom read/seek functions 

175 use_mmap = False 

176 except AttributeError: 

177 read = self.fp.read 

178 

179 try: 

180 seek = self.load_seek 

181 use_mmap = False 

182 except AttributeError: 

183 seek = self.fp.seek 

184 

185 if use_mmap: 

186 # try memory mapping 

187 decoder_name, extents, offset, args = self.tile[0] 

188 if ( 

189 decoder_name == "raw" 

190 and len(args) >= 3 

191 and args[0] == self.mode 

192 and args[0] in Image._MAPMODES 

193 ): 

194 try: 

195 # use mmap, if possible 

196 import mmap 

197 

198 with open(self.filename) as fp: 

199 self.map = mmap.mmap(fp.fileno(), 0, access=mmap.ACCESS_READ) 

200 self.im = Image.core.map_buffer( 

201 self.map, self.size, decoder_name, offset, args 

202 ) 

203 readonly = 1 

204 # After trashing self.im, 

205 # we might need to reload the palette data. 

206 if self.palette: 

207 self.palette.dirty = 1 

208 except (AttributeError, OSError, ImportError): 

209 self.map = None 

210 

211 self.load_prepare() 

212 err_code = -3 # initialize to unknown error 

213 if not self.map: 

214 # sort tiles in file order 

215 self.tile.sort(key=_tilesort) 

216 

217 try: 

218 # FIXME: This is a hack to handle TIFF's JpegTables tag. 

219 prefix = self.tile_prefix 

220 except AttributeError: 

221 prefix = b"" 

222 

223 for decoder_name, extents, offset, args in self.tile: 

224 decoder = Image._getdecoder( 

225 self.mode, decoder_name, args, self.decoderconfig 

226 ) 

227 try: 

228 seek(offset) 

229 decoder.setimage(self.im, extents) 

230 if decoder.pulls_fd: 

231 decoder.setfd(self.fp) 

232 status, err_code = decoder.decode(b"") 

233 else: 

234 b = prefix 

235 while True: 

236 try: 

237 s = read(self.decodermaxblock) 

238 except (IndexError, struct.error) as e: 

239 # truncated png/gif 

240 if LOAD_TRUNCATED_IMAGES: 

241 break 

242 else: 

243 raise OSError("image file is truncated") from e 

244 

245 if not s: # truncated jpeg 

246 if LOAD_TRUNCATED_IMAGES: 

247 break 

248 else: 

249 raise OSError( 

250 "image file is truncated " 

251 f"({len(b)} bytes not processed)" 

252 ) 

253 

254 b = b + s 

255 n, err_code = decoder.decode(b) 

256 if n < 0: 

257 break 

258 b = b[n:] 

259 finally: 

260 # Need to cleanup here to prevent leaks 

261 decoder.cleanup() 

262 

263 self.tile = [] 

264 self.readonly = readonly 

265 

266 self.load_end() 

267 

268 if self._exclusive_fp and self._close_exclusive_fp_after_loading: 

269 self.fp.close() 

270 self.fp = None 

271 

272 if not self.map and not LOAD_TRUNCATED_IMAGES and err_code < 0: 

273 # still raised if decoder fails to return anything 

274 raise_oserror(err_code) 

275 

276 return Image.Image.load(self)

298 def _seek_check(self, frame): 

299 if ( 

300 frame < self._min_frame 

301 # Only check upper limit on frames if additional seek operations 

302 # are not required to do so 

303 or (

339class Parser: 

340 """ 

341 Incremental image parser. This class implements the standard 

342 feed/close consumer interface. 

343 """ 

344 

345 incremental = None 

346 image = None 

347 data = None 

348 decoder = None 

349 offset = 0 

350 finished = 0 

351 

352 def reset(self): 

353 """ 

354 (Consumer) Reset the parser. Note that you can only call this 

355 method immediately after you've created a parser; parser 

356 instances cannot be reused. 

357 """ 

358 assert self.data is None, "cannot reuse parsers" 

359 

360 def feed(self, data): 

361 """ 

362 (Consumer) Feed data to the parser. 

363 

364 :param data: A string buffer. 

365 :exception OSError: If the parser failed to parse the image file. 

366 """ 

367 # collect data 

368 

369 if self.finished: 

370 return 

371 

372 if self.data is None: 

373 self.data = data 

374 else: 

375 self.data = self.data + data 

376 

377 # parse what we have 

378 if self.decoder: 

379 

380 if self.offset > 0: 

381 # skip header 

382 skip = min(len(self.data), self.offset) 

383 self.data = self.data[skip:] 

384 self.offset = self.offset - skip 

385 if self.offset > 0 or not self.data: 

386 return 

387 

388 n, e = self.decoder.decode(self.data) 

389 

390 if n < 0: 

391 # end of stream 

392 self.data = None 

393 self.finished = 1 

394 if e < 0: 

395 # decoding error 

396 self.image = None 

397 raise_oserror(e) 

398 else: 

399 # end of image 

400 return 

401 self.data = self.data[n:] 

402 

403 elif self.image: 

404 

405 # if we end up here with no decoder, this file cannot 

406 # be incrementally parsed. wait until we've gotten all 

407 # available data 

408 pass 

409 

410 else: 

411 

412 # attempt to open this file 

413 try: 

414 with io.BytesIO(self.data) as fp: 

415 im = Image.open(fp) 

416 except OSError: 

417 # traceback.print_exc() 

418 pass # not enough data 

419 else: 

420 flag = hasattr(im, "load_seek") or hasattr(im, "load_read") 

421 if flag or len(im.tile) != 1: 

422 # custom load code, or multiple tiles 

423 self.decode = None 

424 else: 

425 # initialize decoder 

426 im.load_prepare() 

427 d, e, o, a = im.tile[0] 

428 im.tile = [] 

429 self.decoder = Image._getdecoder(im.mode, d, a, im.decoderconfig) 

430 self.decoder.setimage(im.im, e) 

431 

432 # calculate decoder offset 

433 self.offset = o 

434 if self.offset <= len(self.data): 

435 self.data = self.data[self.offset :] 

436 self.offset = 0 

437 

438 self.image = im 

439 

440 def __enter__(self): 

441 return self 

442 

443 def __exit__(self, *args): 

444 self.close() 

445 

446 def close(self): 

447 """ 

448 (Consumer) Close the stream. 

449 

450 :returns: An image object. 

451 :exception OSError: If the parser failed to parse the image file either 

452 because it cannot be identified or cannot be 

453 decoded. 

454 """ 

455 # finish decoding 

456 if self.decoder: 

457 # get rid of what's left in the buffers 

458 self.feed(b"") 

459 self.data = self.decoder = None 

460 if not self.finished: 

461 raise OSError("image was incomplete") 

462 if not self.image: 

463 raise OSError("cannot parse this image") 

464 if self.data: 

465 # incremental parsing not possible; reopen the file 

466 # not that we have all data 

467 with io.BytesIO(self.data) as fp: 

468 try: 

469 self.image = Image.open(fp) 

470 finally: 

471 self.image.load() 

472 return self.image

478def _save(im, fp, tile, bufsize=0): 

479 """Helper to save image based on tile list 

480 

481 :param im: Image object. 

482 :param fp: File object. 

483 :param tile: Tile list. 

484 :param bufsize: Optional buffer size 

485 """ 

486 

487 im.load() 

488 if not hasattr(im, "encoderconfig"): 

489 im.encoderconfig = () 

490 tile.sort(key=_tilesort) 

491 # FIXME: make MAXBLOCK a configuration parameter 

492 # It would be great if we could have the encoder specify what it needs 

493 # But, it would need at least the image size in most cases. RawEncode is 

494 # a tricky case. 

495 bufsize = max(MAXBLOCK, bufsize, im.size[0] * 4) # see RawEncode.c 

496 if fp == sys.stdout: 

497 fp.flush() 

498 return 

499 try: 

500 fh = fp.fileno() 

501 fp.flush() 

502 except (AttributeError, io.UnsupportedOperation) as exc: 

503 # compress to Python file-compatible object 

504 for e, b, o, a in tile: 

505 e = Image._getencoder(im.mode, e, a, im.encoderconfig) 

506 if o > 0: 

507 fp.seek(o) 

508 e.setimage(im.im, b) 

509 if e.pushes_fd: 

510 e.setfd(fp) 

511 l, s = e.encode_to_pyfd() 

512 else: 

513 while True: 

514 l, s, d = e.encode(bufsize) 

515 fp.write(d) 

516 if s: 

517 break 

518 if s < 0: 

519 raise OSError(f"encoder error {s} when writing image file") from exc 

520 e.cleanup() 

521 else: 

522 # slight speedup: compress to real file object 

523 for e, b, o, a in tile: 

524 e = Image._getencoder(im.mode, e, a, im.encoderconfig) 

525 if o > 0: 

526 fp.seek(o) 

527 e.setimage(im.im, b) 

528 if e.pushes_fd: 

529 e.setfd(fp) 

530 l, s = e.encode_to_pyfd() 

531 else: 

532 s = e.encode_to_file(fh, bufsize) 

533 if s < 0: 

534 raise OSError(f"encoder error {s} when writing image file") 

535 e.cleanup() 

536 if hasattr(fp, "flush"): 

537 fp.flush()

540def _safe_read(fp, size): 

541 """ 

542 Reads large blocks in a safe way. Unlike fp.read(n), this function 

543 doesn't trust the user. If the requested size is larger than 

544 SAFEBLOCK, the file is read block by block. 

545 

546 :param fp: File handle. Must implement a <b>read</b> method. 

547 :param size: Number of bytes to read. 

548 :returns: A string containing <i>size</i> bytes of data. 

549 

550 Raises an OSError if the file is truncated and the read cannot be completed 

551 

552 """ 

553 if size <= 0: 

554 return b"" 

555 if size <= SAFEBLOCK: 

556 data = fp.read(size) 

557 if len(data) < size: 

558 raise OSError("Truncated File Read") 

559 return data 

560 data = [] 

561 while size > 0: 

562 block = fp.read(min(size, SAFEBLOCK)) 

563 if not block: 

564 break 

565 data.append(block) 

566 size -= len(block) 

567 if sum(len(d) for d in data) < size: 

568 raise OSError("Truncated File Read") 

569 return b"".join(data)
