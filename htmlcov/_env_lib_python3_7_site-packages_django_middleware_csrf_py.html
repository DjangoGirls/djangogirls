44def _salt_cipher_secret(secret): 

45 """ 

46 Given a secret (assumed to be a string of CSRF_ALLOWED_CHARS), generate a 

47 token by adding a salt and using it to encrypt the secret. 

48 """ 

49 salt = _get_new_csrf_string() 

50 chars = CSRF_ALLOWED_CHARS 

51 pairs = zip((chars.index(x) for x in secret), (chars.index(x) for x in salt)) 

52 cipher = ''.join(chars[(x + y) % len(chars)] for x, y in pairs) 

53 return salt + cipher

56def _unsalt_cipher_token(token): 

57 """ 

58 Given a token (assumed to be a string of CSRF_ALLOWED_CHARS, of length 

59 CSRF_TOKEN_LENGTH, and that its first half is a salt), use it to decrypt 

60 the second half to produce the original secret. 

61 """ 

62 salt = token[:CSRF_SECRET_LENGTH] 

63 token = token[CSRF_SECRET_LENGTH:] 

64 chars = CSRF_ALLOWED_CHARS 

65 pairs = zip((chars.index(x) for x in token), (chars.index(x) for x in salt)) 

66 secret = ''.join(chars[x - y] for x, y in pairs) # Note negative values are ok 

67 return secret
