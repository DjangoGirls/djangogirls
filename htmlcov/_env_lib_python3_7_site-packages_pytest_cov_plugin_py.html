18def validate_report(arg): 

19 file_choices = ['annotate', 'html', 'xml'] 

20 term_choices = ['term', 'term-missing'] 

21 term_modifier_choices = ['skip-covered'] 

22 all_choices = term_choices + file_choices 

23 values = arg.split(":", 1) 

24 report_type = values[0] 

25 if report_type not in all_choices + ['']: 

26 msg = 'invalid choice: "{}" (choose from "{}")'.format(arg, all_choices) 

27 raise argparse.ArgumentTypeError(msg) 

28 

29 if len(values) == 1: 

30 return report_type, None 

31 

32 report_modifier = values[1] 

33 if report_type in term_choices and report_modifier in term_modifier_choices: 

34 return report_type, report_modifier 

35 

36 if report_type not in file_choices: 

37 msg = 'output specifier not supported for: "{}" (choose from "{}")'.format(arg, 

38 file_choices) 

39 raise argparse.ArgumentTypeError(msg) 

40 

41 return values
