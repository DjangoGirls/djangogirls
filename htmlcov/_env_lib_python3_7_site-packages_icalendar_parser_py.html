20def escape_char(text): 

21 """Format value according to iCalendar TEXT escaping rules. 

22 """ 

23 assert isinstance(text, (compat.unicode_type, compat.bytes_type)) 

24 # NOTE: ORDER MATTERS! 

25 return text.replace(r'\N', '\n')\ 

26 .replace('\\', '\\\\')\ 

27 .replace(';', r'\;')\ 

28 .replace(',', r'\,')\ 

29 .replace('\r\n', r'\n')\ 

30 .replace('\n', r'\n')

65def foldline(line, limit=75, fold_sep='\r\n '): 

66 """Make a string folded as defined in RFC5545 

67 Lines of text SHOULD NOT be longer than 75 octets, excluding the line 

68 break. Long content lines SHOULD be split into a multiple line 

69 representations using a line "folding" technique. That is, a long 

70 line can be split between any two characters by inserting a CRLF 

71 immediately followed by a single linear white-space character (i.e., 

72 SPACE or HTAB). 

73 """ 

74 assert isinstance(line, compat.unicode_type) 

75 assert '\n' not in line 

76 

77 # Use a fast and simple variant for the common case that line is all ASCII. 

78 try: 

79 line.encode('ascii') 

80 except (UnicodeEncodeError, UnicodeDecodeError): 

81 pass 

82 else: 

83 return fold_sep.join( 

84 line[i:i + limit - 1] for i in range(0, len(line), limit - 1) 

85 ) 

86 

87 ret_chars = [] 

88 byte_count = 0 

89 for char in line: 

90 char_byte_len = len(char.encode(DEFAULT_ENCODING)) 

91 byte_count += char_byte_len 

92 if byte_count >= limit: 

93 ret_chars.append(fold_sep) 

94 byte_count = char_byte_len 

95 ret_chars.append(char) 

96 

97 return ''.join(ret_chars)

142def dquote(val): 

143 """Enclose parameter values containing [,;:] in double quotes. 

144 """ 

145 # a double-quote character is forbidden to appear in a parameter value 

146 # so replace it with a single-quote character 

147 val = val.replace('"', "'") 

148 if QUOTABLE.search(val): 

149 return '"%s"' % val 

150 return val

154def q_split(st, sep=',', maxsplit=-1): 

155 """Splits a string on char, taking double (q)uotes into considderation. 

156 """ 

157 if maxsplit == 0: 

158 return [st] 

159 

160 result = [] 

161 cursor = 0 

162 length = len(st) 

163 inquote = 0 

164 splits = 0 

165 for i in range(length): 

166 ch = st[i] 

167 if ch == '"': 

168 inquote = not inquote 

169 if not inquote and ch == sep: 

170 result.append(st[cursor:i]) 

171 cursor = i + 1 

172 splits += 1 

173 if i + 1 == length or splits == maxsplit: 

174 result.append(st[cursor:]) 

175 break 

176 return result

180 """Joins a list on sep, quoting strings with QUOTABLE chars.

215 def to_ical(self, sorted=True): 

216 result = [] 

217 items = list(self.items()) 

218 if sorted: 

219 items.sort() 

220 

221 for key, value in items: 

222 value = param_value(value) 

223 if isinstance(value, compat.unicode_type): 

224 value = value.encode(DEFAULT_ENCODING) 

225 # CaselessDict keys are always unicode 

226 key = key.upper().encode(DEFAULT_ENCODING) 

227 result.append(key + b'=' + value) 

228 return b';'.join(result)

230 @classmethod 

231 def from_ical(cls, st, strict=False): 

232 """Parses the parameter format from ical text format.""" 

233 

234 # parse into strings 

235 result = cls() 

236 for param in q_split(st, ';'): 

237 try: 

238 key, val = q_split(param, '=', maxsplit=1) 

239 validate_token(key) 

240 # Property parameter values that are not in quoted 

241 # strings are case insensitive. 

242 vals = [] 

243 for v in q_split(val, ','): 

244 if v.startswith('"') and v.endswith('"'): 

245 v = v.strip('"') 

246 validate_param_value(v, quoted=True) 

247 vals.append(v) 

248 else: 

249 validate_param_value(v, quoted=False) 

250 if strict: 

251 vals.append(v.upper()) 

252 else: 

253 vals.append(v) 

254 if not vals: 

255 result[key] = val 

256 else: 

257 if len(vals) == 1: 

258 result[key] = vals[0] 

259 else: 

260 result[key] = vals 

261 except ValueError as exc: 

262 raise ValueError('%r is not a valid parameter string: %s' 

263 % (param, exc)) 

264 return result

288class Contentline(compat.unicode_type): 

289 """A content line is basically a string that can be folded and parsed into 

290 parts. 

291 """ 

292 def __new__(cls, value, strict=False, encoding=DEFAULT_ENCODING): 

293 value = to_unicode(value, encoding=encoding) 

294 assert '\n' not in value, ('Content line can not contain unescaped ' 

295 'new line characters.') 

296 self = super(Contentline, cls).__new__(cls, value) 

297 self.strict = strict 

298 return self

300 @classmethod 

301 def from_parts(cls, name, params, values, sorted=True): 

302 """Turn a parts into a content line. 

303 """ 

304 assert isinstance(params, Parameters) 

305 if hasattr(values, 'to_ical'): 

306 values = values.to_ical() 

307 else: 

308 values = vText(values).to_ical() 

309 # elif isinstance(values, basestring): 

310 # values = escape_char(values) 

311 

312 # TODO: after unicode only, remove this 

313 # Convert back to unicode, after to_ical encoded it. 

314 name = to_unicode(name) 

315 values = to_unicode(values) 

316 if params: 

317 params = to_unicode(params.to_ical(sorted=sorted)) 

318 return cls('%s;%s:%s' % (name, params, values)) 

319 return cls('%s:%s' % (name, values))

321 def parts(self): 

322 """Split the content line up into (name, parameters, values) parts. 

323 """ 

324 try: 

325 st = escape_string(self) 

326 name_split = None 

327 value_split = None 

328 in_quotes = False 

329 for i, ch in enumerate(st): 

330 if not in_quotes: 

331 if ch in ':;' and not name_split: 

332 name_split = i 

333 if ch == ':' and not value_split: 

334 value_split = i 

335 if ch == '"': 

336 in_quotes = not in_quotes 

337 name = unescape_string(st[:name_split]) 

338 if not name: 

339 raise ValueError('Key name is required') 

340 validate_token(name) 

341 if not name_split or name_split + 1 == value_split: 

342 raise ValueError('Invalid content line') 

343 params = Parameters.from_ical(st[name_split + 1: value_split], 

344 strict=self.strict) 

345 params = Parameters( 

346 (unescape_string(key), unescape_list_or_string(value)) 

347 for key, value in compat.iteritems(params) 

348 ) 

349 values = unescape_string(st[value_split + 1:]) 

350 return (name, params, values) 

351 except ValueError as exc: 

352 raise ValueError( 

353 "Content line could not be parsed into parts: '%s': %s" 

354 % (self, exc) 

355 )

382 @classmethod 

383 def from_ical(cls, st): 

384 """Parses a string into content lines. 

385 """ 

386 st = to_unicode(st) 

387 try: 

388 # a fold is carriage return followed by either a space or a tab 

389 unfolded = uFOLD.sub('', st) 

390 lines = cls(Contentline(line) for 

391 line in NEWLINE.split(unfolded) if line) 

392 lines.append('') # '\r\n' at the end of every content line 

393 return lines 

394 except: 

395 raise ValueError('Expected StringType with content lines')
